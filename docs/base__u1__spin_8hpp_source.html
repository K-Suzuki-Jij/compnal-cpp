145 static std::vector<std::int64_t> GenerateBasis(const int system_size,
146 const HalfInt magnitude_spin,
147 const HalfInt total_sz,
148 const bool flag_display_info = true) {
149 if (!isValidQNumber(system_size, magnitude_spin, total_sz) || system_size <= 0) {
150 std::stringstream ss;
151 ss << "Error in " << __FUNCTION__ << " at " << __LINE__ << std::endl;
152 ss << "Invalid parameters (system_size or magnitude_spin or total_sz)" << std::endl;
153 throw std::runtime_error(ss.str());
154 }
155
156 const auto start = std::chrono::system_clock::now();
157
158 const int dim_onsite = 2*magnitude_spin + 1;
159 std::vector<std::int64_t> basis;
160
161 if (flag_display_info) {
162 std::cout << "Generating Basis..." << std::flush;
163 }
164
165 const int shifted_2sz = static_cast<int>(2*(system_size*magnitude_spin - total_sz));
166 const std::int64_t dim_target = CalculateTargetDim(system_size, magnitude_spin, total_sz);
167 std::vector<std::vector<int>> partition_integers;
168 utility::GenerateIntegerPartition(&partition_integers, shifted_2sz, magnitude_spin);
169
170 std::vector<std::int64_t> site_constant(system_size);
171 for (int site = 0; site < system_size; ++site) {
172 site_constant[site] = static_cast<std::int64_t>(std::pow(dim_onsite, site));
173 }
174
175#ifdef _OPENMP
176 const int num_threads = omp_get_max_threads();
177 std::vector<std::vector<std::int64_t>> temp_basis(num_threads);
178 for (auto &&integer_list: partition_integers) {
179 const bool condition1 = (0 < integer_list.size()) && (static_cast<int>(integer_list.size()) <= system_size);
180 const bool condition2 = (integer_list.size() == 0) && (shifted_2sz == 0);
181 if (condition1 || condition2) {
182 for (int j = static_cast<int>(integer_list.size()); j < system_size; ++j) {
183 integer_list.push_back(0);
184 }
185
186 const std::int64_t size = utility::CalculateNumCombination(integer_list);
187 std::vector<std::vector<int>> temp_partition_integer(num_threads);
188
189#pragma omp parallel num_threads (num_threads)
190 {
191 const int thread_num = omp_get_thread_num();
192 const std::int64_t loop_begin = thread_num*size/num_threads;
193 const std::int64_t loop_end = (thread_num + 1)*size/num_threads;
194 temp_partition_integer[thread_num] = integer_list;
195 utility::CalculateNthPermutation(&temp_partition_integer[thread_num], loop_begin);
196
197 for (std::int64_t j = loop_begin; j < loop_end; ++j) {
198 std::int64_t basis_global = 0;
199 const auto iter_begin = temp_partition_integer[thread_num].begin();
200 const auto iter_end = temp_partition_integer[thread_num].end();
201 for (auto itr = iter_begin; itr != iter_end; ++itr) {
202 basis_global += *itr*site_constant[std::distance(iter_begin, itr)];
203 }
204 temp_basis[thread_num].push_back(basis_global);
205 std::next_permutation(temp_partition_integer[thread_num].begin(), temp_partition_integer[thread_num].end());
206 }
207 }
208 }
209 }
210
211 for (auto &&it: temp_basis) {
212 basis.insert(basis.end(), it.begin(), it.end());
213 std::vector<std::int64_t>().swap(it);
214 }
215
216#else
217 basis.reserve(dim_target);
218
219 for (auto &&integer_list: partition_integers) {
220 const bool condition1 = (0 < integer_list.size()) && (static_cast<int>(integer_list.size()) <= system_size_);
221 const bool condition2 = (integer_list.size() == 0) && (shifted_2sz == 0);
222 if (condition1 || condition2) {
223
224 for (std::int64_t j = integer_list.size(); j < system_size_; ++j) {
225 integer_list.push_back(0);
226 }
227
228 std::sort(integer_list.begin(), integer_list.end());
229
230 do {
231 std::int64_t basis_global = 0;
232 for (std::size_t j = 0; j < integer_list.size(); ++j) {
233 basis_global += integer_list[j]*site_constant[j];
234 }
235 basis.push_back(basis_global);
236 } while (std::next_permutation(integer_list.begin(), integer_list.end()));
237 }
238 }
239
240#endif
241
242 if (static_cast<std::int64_t>(basis.size()) != dim_target) {
243 std::stringstream ss;
244 ss << "Unknown error detected in " << __FUNCTION__ << " at " << __LINE__ << std::endl;
245 throw std::runtime_error(ss.str());
246 }
247
248 std::sort(basis.begin(), basis.end());
249
250 if (flag_display_info) {
251 const auto time_count = std::chrono::duration_cast<std::chrono::microseconds>(std::chrono::system_clock::now() - start).count();
252 const double time_sec = static_cast<double>(time_count)/sparse_matrix::TIME_UNIT_CONSTANT;
253 std::cout << "\rElapsed time of generating basis:" << time_sec << "[sec]" << std::endl;
254 }
255 return basis;
256 }

263 static bool isValidQNumber(const int system_size, const HalfInt magnitude_spin, const HalfInt total_sz) {
264 const int total_2sz = 2*total_sz;
265 const int magnitude_2spin = 2*magnitude_spin;
266 const bool c1 = ((system_size*magnitude_2spin - total_2sz)%2 == 0);
267 const bool c2 = (-system_size*magnitude_2spin <= total_2sz);
268 const bool c3 = (total_2sz <= system_size*magnitude_2spin);
269 if (c1 && c2 && c3) {
270 return true;
271 }
272 else {
273 return false;
274 }
275 }

283 static std::int64_t CalculateTargetDim(const int system_size, const HalfInt magnitude_spin, const HalfInt total_sz) {
284 if (!isValidQNumber(system_size, magnitude_spin, total_sz)) {
285 return 0;
286 }
287 if (system_size <= 0) {
288 return 0;
289 }
290 const int magnitude_2spin = 2*magnitude_spin;
291 const int total_2sz = 2*total_sz;
292 const int max_total_2sz = system_size*magnitude_2spin;
293 std::vector<std::vector<std::int64_t>> dim(system_size, std::vector<std::int64_t>(max_total_2sz + 1));
294 for (int s = -magnitude_2spin; s <= magnitude_2spin; s += 2) {
295 dim[0][(s + magnitude_2spin)/2] = 1;
296 }
297 for (int site = 1; site < system_size; site++) {
298 for (int s = -magnitude_2spin; s <= magnitude_2spin; s += 2) {
299 for (int s_prev = -magnitude_2spin*site; s_prev <= magnitude_2spin*site; s_prev += 2) {
300 const std::int64_t a = dim[site ][(s + s_prev + magnitude_2spin*(site + 1))/2];
301 const std::int64_t b = dim[site - 1][(s_prev + magnitude_2spin*site)/2];
302 if (a >= INT64_MAX - b) {
303 throw std::overflow_error("Overflow detected for sumation using uint64_t");
304 }
305 dim[site][(s + s_prev + magnitude_2spin*(site + 1))/2] = a + b;
306 }
307 }
308 }
309 return dim[system_size - 1][(total_2sz + max_total_2sz)/2];
310 }

315 static CRS CreateOnsiteOperatorSx(const HalfInt magnitude_spin) {
316 return 0.5*(CreateOnsiteOperatorSp(magnitude_spin) + CreateOnsiteOperatorSm(magnitude_spin));

330 static CRS CreateOnsiteOperatorSz(const HalfInt magnitude_spin) {
331 const int dim_onsite = 2*magnitude_spin + 1;
332 CRS matrix(dim_onsite, dim_onsite);
333
334 for (int row = 0; row < dim_onsite; ++row) {
335 const RealType val = magnitude_spin - row;
336 if (val != 0.0) {
337 matrix.val.push_back(val);
338 matrix.col.push_back(row);
339 }
340 matrix.row[row + 1] = matrix.col.size();
341 }
342 return matrix;
343 }
344
348 static CRS CreateOnsiteOperatorSp(const HalfInt magnitude_spin) {
349 const int dim_onsite = 2*magnitude_spin + 1;
350 CRS matrix(dim_onsite, dim_onsite);
351 for (int row = 1; row < dim_onsite; ++row) {
352 matrix.val.push_back(std::sqrt((magnitude_spin + 1)*2.0*row - row*(row + 1)));
353 matrix.col.push_back(row);
354 matrix.row[row] = matrix.col.size();
355 }
356 matrix.row[dim_onsite] = matrix.col.size();
357 return matrix;
358 }
359
363 static CRS CreateOnsiteOperatorSm(const HalfInt magnitude_spin) {
364 const int dim_onsite = 2*magnitude_spin + 1;
365 CRS matrix(dim_onsite, dim_onsite);
366 for (int row = 1; row < dim_onsite; ++row) {
367 matrix.val.push_back(std::sqrt((magnitude_spin + 1)*2.0*row - row*(row + 1)));
368 matrix.col.push_back(row - 1);
369 matrix.row[row + 1] = matrix.col.size();
370 }
371 return matrix;
372 }
