174 std::vector<std::int64_t> GenerateBasis(const HalfInt total_sz, const bool flag_display_info = false) const {
175 if (!isValidQNumber(total_sz)) {
176 std::stringstream ss;
177 ss << "Error in " << __FUNCTION__ << " at " << __LINE__ << std::endl;
178 ss << "Invalid parameters (system_size or magnitude_spin or total_sz)" << std::endl;
179 throw std::runtime_error(ss.str());
180 }
181
182 const auto start = std::chrono::system_clock::now();
183
184 std::vector<std::int64_t> basis;
185
186 if (flag_display_info) {
187 std::cout << "Generating Basis..." << std::flush;
188 }
189
190 const int shifted_2sz = static_cast<int>(2*(system_size_*magnitude_spin_ - total_sz));
191 const std::int64_t dim_target = CalculateTargetDim(total_sz);
192 std::vector<std::vector<int>> partition_integers;
193 utility::GenerateIntegerPartition(&partition_integers, shifted_2sz, magnitude_spin_);
194
195 std::vector<std::int64_t> site_constant(system_size_);
196 for (int site = 0; site < system_size_; ++site) {
197 site_constant[site] = static_cast<std::int64_t>(std::pow(dim_onsite_, site));
198 }
199
200#ifdef _OPENMP
201 const int num_threads = omp_get_max_threads();
202 std::vector<std::vector<std::int64_t>> temp_basis(num_threads);
203 for (auto &&integer_list: partition_integers) {
204 const bool condition1 = (0 < integer_list.size()) && (static_cast<int>(integer_list.size()) <= system_size_);
205 const bool condition2 = (integer_list.size() == 0) && (shifted_2sz == 0);
206 if (condition1 || condition2) {
207 for (int j = static_cast<int>(integer_list.size()); j < system_size_; ++j) {
208 integer_list.push_back(0);
209 }
210
211 const std::int64_t size = utility::CalculateNumCombination(integer_list);
212 std::vector<std::vector<int>> temp_partition_integer(num_threads);
213
214#pragma omp parallel num_threads (num_threads)
215 {
216 const int thread_num = omp_get_thread_num();
217 const std::int64_t loop_begin = thread_num*size/num_threads;
218 const std::int64_t loop_end = (thread_num + 1)*size/num_threads;
219 temp_partition_integer[thread_num] = integer_list;
220 utility::CalculateNthPermutation(&temp_partition_integer[thread_num], loop_begin);
221
222 for (std::int64_t j = loop_begin; j < loop_end; ++j) {
223 std::int64_t basis_global = 0;
224 const auto iter_begin = temp_partition_integer[thread_num].begin();
225 const auto iter_end = temp_partition_integer[thread_num].end();
226 for (auto itr = iter_begin; itr != iter_end; ++itr) {
227 basis_global += *itr*site_constant[std::distance(iter_begin, itr)];
228 }
229 temp_basis[thread_num].push_back(basis_global);
230 std::next_permutation(temp_partition_integer[thread_num].begin(), temp_partition_integer[thread_num].end());
231 }
232 }
233 }
234 }
235
236 for (auto &&it: temp_basis) {
237 basis.insert(basis.end(), it.begin(), it.end());
238 std::vector<std::int64_t>().swap(it);
239 }
240
241#else
242 basis.reserve(dim_target);
243
244 for (auto &&integer_list: partition_integers) {
245 const bool condition1 = (0 < integer_list.size()) && (static_cast<int>(integer_list.size()) <= system_size_);
246 const bool condition2 = (integer_list.size() == 0) && (shifted_2sz == 0);
247 if (condition1 || condition2) {
248
249 for (std::int64_t j = integer_list.size(); j < system_size_; ++j) {
250 integer_list.push_back(0);
251 }
252
253 std::sort(integer_list.begin(), integer_list.end());
254
255 do {
256 std::int64_t basis_global = 0;
257 for (std::size_t j = 0; j < integer_list.size(); ++j) {
258 basis_global += integer_list[j]*site_constant[j];
259 }
260 basis.push_back(basis_global);
261 } while (std::next_permutation(integer_list.begin(), integer_list.end()));
262 }
263 }
264
265#endif
266
267 if (static_cast<std::int64_t>(basis.size()) != dim_target) {
268 std::stringstream ss;
269 ss << "Unknown error detected in " << __FUNCTION__ << " at " << __LINE__ << std::endl;
270 throw std::runtime_error(ss.str());
271 }
272
273 std::sort(basis.begin(), basis.end());
274
275 if (flag_display_info) {
276 const auto time_count = std::chrono::duration_cast<std::chrono::microseconds>(std::chrono::system_clock::now() - start).count();
277 const double time_sec = static_cast<double>(time_count)/sparse_matrix::TIME_UNIT_CONSTANT;
278 std::cout << "\rElapsed time of generating basis:" << time_sec << "[sec]" << std::endl;
279 }
280 return basis;
281 }

288 static bool isValidQNumber(const int system_size, const HalfInt magnitude_spin, const HalfInt total_sz) {
289 const int total_2sz = 2*total_sz;
290 const int magnitude_2spin = 2*magnitude_spin;
291 const bool c1 = ((system_size*magnitude_2spin - total_2sz)%2 == 0);
292 const bool c2 = (-system_size*magnitude_2spin <= total_2sz);
293 const bool c3 = (total_2sz <= system_size*magnitude_2spin);
294 if (c1 && c2 && c3) {
295 return true;
296 }
297 else {
298 return false;
299 }
300 }

307 static std::int64_t CalculateTargetDim(const int system_size, const HalfInt magnitude_spin, const HalfInt total_sz) {
308 if (!isValidQNumber(system_size, magnitude_spin, total_sz)) {
309 return 0;
310 }
311 if (system_size <= 0) {
312 return 0;
313 }
314 const int magnitude_2spin = 2*magnitude_spin;
315 const int total_2sz = 2*total_sz;
316 const int max_total_2sz = system_size*magnitude_2spin;
317 std::vector<std::vector<std::int64_t>> dim(system_size, std::vector<std::int64_t>(max_total_2sz + 1));
318 for (int s = -magnitude_2spin; s <= magnitude_2spin; s += 2) {
319 dim[0][(s + magnitude_2spin)/2] = 1;
320 }
321 for (int site = 1; site < system_size; site++) {
322 for (int s = -magnitude_2spin; s <= magnitude_2spin; s += 2) {
323 for (int s_prev = -magnitude_2spin*site; s_prev <= magnitude_2spin*site; s_prev += 2) {
324 const std::int64_t a = dim[site ][(s + s_prev + magnitude_2spin*(site + 1))/2];
325 const std::int64_t b = dim[site - 1][(s_prev + magnitude_2spin*site)/2];
326 if (a >= INT64_MAX - b) {
327 throw std::runtime_error("Overflow detected for sumation using uint64_t");
328 }
329 dim[site][(s + s_prev + magnitude_2spin*(site + 1))/2] = a + b;
330 }
331 }
332 }
333 return dim[system_size - 1][(total_2sz + max_total_2sz)/2];
334 }

339 static CRS CreateOnsiteOperatorSx(const HalfInt magnitude_spin) {
340 return 0.5*(CreateOnsiteOperatorSp(magnitude_spin) + CreateOnsiteOperatorSm(magnitude_spin));

353 static CRS CreateOnsiteOperatorSz(const HalfInt magnitude_spin) {
354 const int dim_onsite = 2*magnitude_spin + 1;
355 CRS matrix(dim_onsite, dim_onsite);
356
357 for (int row = 0; row < dim_onsite; ++row) {
358 const RealType val = magnitude_spin - row;
359 if (val != 0.0) {
360 matrix.val.push_back(val);
361 matrix.col.push_back(row);
362 }
363 matrix.row[row + 1] = matrix.col.size();
364 }
365 return matrix;
366 }
367
371 static CRS CreateOnsiteOperatorSp(const HalfInt magnitude_spin) {
372 const int dim_onsite = 2*magnitude_spin + 1;
373 CRS matrix(dim_onsite, dim_onsite);
374 for (int row = 1; row < dim_onsite; ++row) {
375 matrix.val.push_back(std::sqrt((magnitude_spin + 1)*2.0*row - row*(row + 1)));
376 matrix.col.push_back(row);
377 matrix.row[row] = matrix.col.size();
378 }
379 matrix.row[dim_onsite] = matrix.col.size();
380 return matrix;
381 }
382
386 static CRS CreateOnsiteOperatorSm(const HalfInt magnitude_spin) {
387 const int dim_onsite = 2*magnitude_spin + 1;
388 CRS matrix(dim_onsite, dim_onsite);
389 for (int row = 1; row < dim_onsite; ++row) {
390 matrix.val.push_back(std::sqrt((magnitude_spin + 1)*2.0*row - row*(row + 1)));
391 matrix.col.push_back(row - 1);
392 matrix.row[row + 1] = matrix.col.size();
393 }
394 return matrix;
