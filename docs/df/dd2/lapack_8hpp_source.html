39template <typename RealType>
40void LapackDsyev(RealType *gs_value,
41 std::vector<RealType> *gs_vector,
42 const type::CRS<RealType> &matrix_in,
43 const int target_level = 0
44 ) {
45
46 if (matrix_in.row_dim != matrix_in.col_dim || matrix_in.row_dim < 1 || matrix_in.col_dim < 1) {
47 std::stringstream ss;
48 ss << "Error in " << __func__ << std::endl;
49 ss << "The input matrix is not a square one" << std::endl;
50 ss << "row=" << matrix_in.row_dim << ", col=" << matrix_in.col_dim << std::endl;
51 throw std::runtime_error(ss.str());
52 }
53
54 if (target_level < 0) {
55 std::stringstream ss;
56 ss << "Invalid target_level" << std::endl;
57 throw std::runtime_error(ss.str());
58 }
59
60 double matrix_array[matrix_in.row_dim][matrix_in.col_dim];
61
62 for (std::int64_t i = 0; i < matrix_in.row_dim; ++i) {
63 for (std::int64_t j = 0; j < matrix_in.col_dim; ++j) {
64 matrix_array[i][j] = 0.0;
65 }
66 }
67
68 for (std::int64_t i = 0; i < matrix_in.row_dim; ++i) {
69 for (std::int64_t j = matrix_in.row[i]; j < matrix_in.row[i + 1]; ++j) {
70 matrix_array[i][matrix_in.col[j]] = static_cast<double>(matrix_in.val[j]);
71 matrix_array[matrix_in.col[j]][i] = static_cast<double>(matrix_in.val[j]);
72 }
73 }
74
75 int info;
76 double val_array[matrix_in.row_dim];
77 double work[3 * matrix_in.row_dim];
78
79 dsyev_('V', 'L', static_cast<int>(matrix_in.row_dim),
80 (double**)matrix_array,
81 static_cast<int>(matrix_in.row_dim), val_array, work,
82 static_cast<int>(3 * matrix_in.row_dim), info);
83
84 gs_vector->resize(matrix_in.row_dim);
85
86 for (std::int64_t i = 0; i < matrix_in.row_dim; ++i) {
87 (*gs_vector)[i] = static_cast<RealType>(matrix_array[target_level][i]);
88 }
89
90 *gs_value = static_cast<RealType>(val_array[target_level]);
91}

93template <typename RealType>
94void LapackDstev(RealType *gs_value,
95 std::vector<RealType> *gs_vector,
96 const std::vector<RealType> &diag,
97 const std::vector<RealType> &off_diag) {
98
99 if (off_diag.size() + 1 != diag.size()) {
100 std::stringstream ss;
101 ss << "Error in " << __func__ << std::endl;
102 ss << "diag size=" << diag.size() << ", off_diag size=" << off_diag.size() << std::endl;
103 throw std::runtime_error(ss.str());
104 }
105
106 int dim = static_cast<int>(diag.size());
107
108 int info;
109 double Lap_D[dim];
110 double Lap_E[dim - 1];
111 double Lap_Vec[dim][dim];
112 double Lap_Work[2*dim];
113
114 for (int i = 0; i < dim; i++) {
115 Lap_D[i] = static_cast<double>(diag[i]);
116 }
117
118 for (int i = 0; i < dim - 1; i++) {
119 Lap_E[i] = static_cast<double>(off_diag[i]);
120 }
121
122 dstev_('V', dim, Lap_D, Lap_E, (double**)Lap_Vec, dim, Lap_Work, info);
123
124 gs_vector->resize(dim);
125
126 for (int i = 0; i < dim; ++i) {
127 (*gs_vector)[i] = static_cast<RealType>(Lap_Vec[0][i]);
128 }
129
130 *gs_value = static_cast<RealType>(Lap_D[0]);
131}

133template <typename RealType>
134void LapackDspgv(std::vector<RealType> *eigenvalues,
135 std::vector<RealType> *eigenvectors,
136 const int i_type,
137 const int dim,
138 const std::vector<RealType> &mat_a,
139 const std::vector<RealType> &mat_b) {
140
141 const int size = dim*(dim + 1)/2;
142
143 if (static_cast<int>(mat_a.size()) < size || static_cast<int>(mat_b.size()) < size) {
144 std::stringstream ss;
145 ss << "Error in " << __func__ << std::endl;
146 throw std::runtime_error(ss.str());
147 }
148
149 double lap_ap[size];
150 double lap_bp[size];
151 double lap_w[dim];
152 double lap_z[dim][dim];
153 double lap_work[3*dim];
154 int lap_ldz = dim;
155 int lap_info;
156
157 for (int i = 0; i < size; ++i) {
158 lap_ap[i] = mat_a[i];
159 lap_bp[i] = mat_b[i];
160 }
161
162 dspgv_(i_type, 'V', 'U', dim, lap_ap, lap_bp, lap_w, (double**)lap_z, lap_ldz, lap_work, lap_info);
163
164 eigenvalues->resize(dim);
165 eigenvectors->resize(dim*dim);
166
167 for (int i = 0; i < dim; ++i) {
168 (*eigenvalues)[i] = static_cast<RealType>(lap_w[i]);
169 for (int j = 0; j < dim; ++j) {
170 (*eigenvectors)[i*dim + j] = static_cast<RealType>(lap_z[i][j]);
171 }
172 }
173
174}
