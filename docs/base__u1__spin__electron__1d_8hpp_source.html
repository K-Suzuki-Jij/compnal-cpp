315 q_number_n_vac_vec.push_back(n_vac);
316 q_number_n_up_vec.push_back(n_up );
317 q_number_n_down_vec.push_back(n_down);
318 q_number_n_up_down_vec.push_back(n_up_down);

353 const bool condition2 = (integer_list.size() == 0) && (shifted_2sz == 0);
354 if (condition1 || condition2) {
355 for (int j = static_cast<int>(integer_list.size()); j < system_size_; ++j) {

379#pragma omp parallel for
380 for (std::int64_t i = 0; i < loop_size; ++i) {

382 const int n_up = q_number_n_up_vec[i];
383 const int n_down = q_number_n_down_vec[i];

471 if (!isValidQNumber(system_size, magnitude_lspin, total_electron, total_sz)) {

476 std::int64_t dim = 0;

485 dim += dim_electron*dim_lspin;
486 }
487 }

667 // 0 <-> [ ] -- (0, 0 )

669 // 2 <-> [down ] -- (1, -1 )

687 }
688 for (int element = 0; element < dim_onsite_lspin; ++element) {
689 matrix.row[element + 1 + 3*dim_onsite_lspin] = matrix.col.size();
