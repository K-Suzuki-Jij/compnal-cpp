296 void GenerateBasis(const int total_electron, const double total_sz) {
297 if (!isValidQNumber(total_electron, total_sz)) {
298 std::stringstream ss;
299 ss << "Error in " << __FUNCTION__ << " at " << __LINE__ << std::endl;
300 ss << "Invalid parameters (system_size or magnitude_spin or total_sz)" << std::endl;
301 throw std::runtime_error(ss.str());
302 }
303 const auto start = std::chrono::system_clock::now();
304 const int total_2sz = utility::DoubleHalfInteger(total_sz);
305
306 if (bases_.count({total_electron, total_2sz}) != 0) {
307 return;
308 }
309
310 std::cout << "Generating Basis..." << std::flush;
311 const int max_n_up_down = static_cast<int>(total_electron/2);
312 const std::int64_t dim_target_global = CalculateTargetDim(total_electron, total_sz);
313
314 std::vector<std::int64_t>().swap(bases_[{total_electron, total_2sz}]);
315 auto &global_basis_ref = bases_.at({total_electron, total_2sz});
316
317 std::vector<std::int64_t> site_constant_global(system_size_);
318 std::vector<std::int64_t> site_constant_electron(system_size_);
319 std::vector<std::int64_t> site_constant_lspin(system_size_);
320 for (int site = 0; site < system_size_; ++site) {
321 site_constant_global[site] = static_cast<std::int64_t>(std::pow(dim_onsite_ , site));
322 site_constant_electron[site] = static_cast<std::int64_t>(std::pow(dim_onsite_electron_, site));
323 site_constant_lspin[site] = static_cast<std::int64_t>(std::pow(dim_onsite_lspin_ , site));
324 }
325
326 const std::vector<std::vector<std::int64_t>> binom = utility::CalculateBinomialTable(system_size_);
327 std::vector<int> q_number_spin_vec;
328 std::vector<int> q_number_n_vac_vec;
329 std::vector<int> q_number_n_up_vec;
330 std::vector<int> q_number_n_down_vec;
331 std::vector<int> q_number_n_up_down_vec;
332 std::vector<std::int64_t> bias_basis;
333 bias_basis.push_back(0);
334 for (int n_up_down = 0; n_up_down <= max_n_up_down; ++n_up_down) {
335 for (int n_up = 0; n_up <= total_electron - 2*n_up_down; ++n_up) {
336 const int n_down = total_electron - 2*n_up_down - n_up;
337 const int n_vac = system_size_ - n_up - n_down - n_up_down;
338 if (0 <= n_up && 0 <= n_down && 0 <= n_vac) {
339 const double total_sz_electron = 0.5*(n_up - n_down);
340 if (BaseU1Spin_1D<RealType>::isValidQNumber(system_size_, 0.5*magnitude_2lspin_, total_sz - total_sz_electron)) {
341 q_number_spin_vec.push_back(total_2sz - (n_up - n_down));
342 q_number_n_vac_vec.push_back(n_vac);
343 q_number_n_up_vec.push_back(n_up );
344 q_number_n_down_vec.push_back(n_down);
345 q_number_n_up_down_vec.push_back(n_up_down);
346 const std::int64_t dim_electron = binom[system_size_][n_up]*binom[system_size_ - n_up][n_down]*binom[system_size_ - n_up - n_down][n_up_down];
347 const std::int64_t dim_lspin = BaseU1Spin_1D<RealType>::CalculateTargetDim(system_size_, 0.5*magnitude_2lspin_, total_sz - total_sz_electron);
348 bias_basis.push_back(dim_electron*dim_lspin);
349 }
350 }
351 }
352 }
353
354 for (std::size_t i = 1; i < bias_basis.size(); ++i) {
355 bias_basis[i] += bias_basis[i - 1];
356 }
357
358 if (bias_basis.back() != dim_target_global) {
359 std::stringstream ss;
360 ss << "Unknown error detected in " << __FUNCTION__ << " at " << __LINE__ << std::endl;
361 throw std::runtime_error(ss.str());
362 }
363
364 //Generate spin bases
365 std::vector<int> temp_q_number_spin_vec = q_number_spin_vec;
366 std::sort(temp_q_number_spin_vec.begin(), temp_q_number_spin_vec.end());
367 temp_q_number_spin_vec.erase(std::unique(temp_q_number_spin_vec.begin(), temp_q_number_spin_vec.end()), temp_q_number_spin_vec.end());
368 std::unordered_map<int, std::vector<std::int64_t>> spin_bases;
369#pragma omp parallel for
370 for (std::int64_t i = 0; i < static_cast<std::int64_t>(temp_q_number_spin_vec.size()); ++i) {
371 const int total_2_sz_lspin = temp_q_number_spin_vec[i];
372 const int shifted_2sz = (system_size_*magnitude_2lspin_ - total_2_sz_lspin)/2;
373 const std::int64_t dim_target_lspin = BaseU1Spin_1D<RealType>::CalculateTargetDim(system_size_, 0.5*magnitude_2lspin_, 0.5*total_2_sz_lspin);
374 std::vector<std::vector<int>> partition_integers;
375 utility::GenerateIntegerPartition(&partition_integers, shifted_2sz, magnitude_2lspin_);
376 auto &spin_basis = spin_bases[total_2_sz_lspin];
377 spin_basis.reserve(dim_target_lspin);
378 for (auto &&integer_list: partition_integers) {
379 const bool condition1 = (0 < integer_list.size()) && (static_cast<int>(integer_list.size()) <= system_size_);
380 const bool condition2 = (integer_list.size() == 0) && (shifted_2sz == 0);
381 if (condition1 || condition2) {
382 for (int j = static_cast<int>(integer_list.size()); j < system_size_; ++j) {
383 integer_list.push_back(0);
384 }
385 std::sort(integer_list.begin(), integer_list.end());
386 do {
387 std::int64_t basis_global = 0;
388 for (std::size_t j = 0; j < integer_list.size(); ++j) {
389 basis_global += integer_list[j]*site_constant_global[j];
390 }
391 spin_basis.push_back(basis_global);
392 } while (std::next_permutation(integer_list.begin(), integer_list.end()));
393 }
394 }
395 if (static_cast<std::int64_t>(spin_basis.size()) != dim_target_lspin) {
396 std::stringstream ss;
397 ss << "Unknown error detected in " << __FUNCTION__ << " at " << __LINE__ << std::endl;
398 throw std::runtime_error(ss.str());
399 }
400 std::sort(spin_basis.begin(), spin_basis.end());
401 }
402
403 //Generate global bases
404 const std::int64_t loop_size = static_cast<std::int64_t>(q_number_spin_vec.size());
405 global_basis_ref.resize(dim_target_global);
406#pragma omp parallel for
407 for (std::int64_t i = 0; i < loop_size; ++i) {
408 const int n_vac = q_number_n_vac_vec[i];
409 const int n_up = q_number_n_up_vec[i];
410 const int n_down = q_number_n_down_vec[i];
411 const int n_up_down = q_number_n_up_down_vec[i];
412 const int total_2sz_lspin = q_number_spin_vec[i];
413 std::int64_t count = bias_basis[i];
414 std::vector<int> basis_list_electron(system_size_);
415 for (int s = 0; s < n_vac; ++s) {
416 basis_list_electron[s] = 0;
417 }
418 for (int s = 0; s < n_up; ++s) {
419 basis_list_electron[s + n_vac] = 1;
420 }
421 for (int s = 0; s < n_down; ++s) {
422 basis_list_electron[s + n_vac + n_up] = 2;
423 }
424 for (int s = 0; s < n_up_down; ++s) {
425 basis_list_electron[s + n_vac + n_up + n_down] = 3;
426 }
427
428 do {
429 std::int64_t basis_global_electron = 0;
430 for (std::size_t j = 0; j < basis_list_electron.size(); ++j) {
431 basis_global_electron += basis_list_electron[j]*site_constant_global[j]*dim_onsite_lspin_;
432 }
433 for (const auto &basis_global_lspin: spin_bases.at(total_2sz_lspin)) {
434 global_basis_ref[count++] = basis_global_electron + basis_global_lspin;
435 }
436 } while (std::next_permutation(basis_list_electron.begin(), basis_list_electron.end()));
437 }
438
439 if (static_cast<std::int64_t>(bases_.at({total_electron, total_2sz}).size()) != dim_target_global) {
440 std::stringstream ss;
441 ss << "Unknown error detected in " << __FUNCTION__ << " at " << __LINE__ << std::endl;
442 throw std::runtime_error(ss.str());
443 }
444
445 std::sort(global_basis_ref.begin(), global_basis_ref.end());
446
447 bases_inv_[{total_electron, total_2sz}].clear();
448
449 auto &basis_inv_ref = bases_inv_.at({total_electron, total_2sz});
450 for (std::int64_t i = 0; i < dim_target_global; ++i) {
451 basis_inv_ref[global_basis_ref[i]] = i;
452 }
453
454 if (basis_inv_ref.size() != global_basis_ref.size()) {
455 std::stringstream ss;
456 ss << "Unknown error detected in " << __FUNCTION__ << " at " << __LINE__ << std::endl;
457 ss << "The same basis has been detected" << std::endl;
458 throw std::runtime_error(ss.str());
459 }
460 const auto time_count = std::chrono::duration_cast<std::chrono::microseconds>(std::chrono::system_clock::now() - start).count();
461 const double time_sec = static_cast<double>(time_count)/sparse_matrix::TIME_UNIT_CONSTANT;
462 std::cout << "\rElapsed time of generating basis:" << time_sec << "[sec]" << std::endl;
463
464 }

472 static bool isValidQNumber(const int system_size, const double magnitude_lspin, const int total_electron, const double total_sz) {
473 const int total_2sz = utility::DoubleHalfInteger(total_sz);
474 const int magnitude_2lspin = utility::DoubleHalfInteger(magnitude_lspin);
475 const bool c1 = (0 <= total_electron && total_electron <= 2*system_size);
476 const bool c2 = ((total_electron + system_size*magnitude_2lspin - total_2sz)%2 == 0);
477 const bool c3 = (-(total_electron + system_size*magnitude_2lspin) <= total_2sz);
478 const bool c4 = (total_2sz <= total_electron + system_size*magnitude_2lspin);
479 if (c1 && c2 && c3 && c4) {
480 return true;
481 }
482 else {
483 return false;
484 }
485 }

496 static std::int64_t CalculateTargetDim(const int system_size, const double magnitude_lspin, const int total_electron, const double total_sz) {
497 if (!isValidQNumber(system_size, magnitude_lspin, total_electron, total_sz)) {
498 return 0;
499 }
500 const std::vector<std::vector<std::int64_t>> binom = utility::CalculateBinomialTable(system_size);
501 const int max_n_up_down = static_cast<int>(total_electron/2);
502 std::int64_t dim = 0;
503 for (int n_up_down = 0; n_up_down <= max_n_up_down; ++n_up_down) {
504 for (int n_up = 0; n_up <= total_electron - 2*n_up_down; ++n_up) {
505 const int n_down = total_electron - 2*n_up_down - n_up;
506 const int n_vac = system_size - n_up - n_down - n_up_down;
507 if (0 <= n_up && 0 <= n_down && 0 <= n_vac) {
508 const double total_sz_electron = 0.5*(n_up - n_down);
509 const std::int64_t dim_electron = binom[system_size][n_up]*binom[system_size - n_up][n_down]*binom[system_size - n_up - n_down][n_up_down];
510 const std::int64_t dim_lspin = BaseU1Spin_1D<RealType>::CalculateTargetDim(system_size, magnitude_lspin, total_sz - total_sz_electron);
511 dim += dim_electron*dim_lspin;
512 }
513 }
514 }
515 return dim;
516 }

617 GenerateTargetSector(const CRS &m_1_bra, const CRS &m_2_bra, const CRS &m_3_ket, const CRS &m_4_ket) const {
618 std::unordered_set<QType, utility::PairHash> delta_sector_set_m1;
619 std::unordered_set<QType, utility::PairHash> delta_sector_set_m2;
620 std::unordered_set<QType, utility::PairHash> delta_sector_set_m3;
621 std::unordered_set<QType, utility::PairHash> delta_sector_set_m4;
622
623 for (std::int64_t i = 0; i < m_1_bra.row_dim; ++i) {
624 for (std::int64_t j = m_1_bra.row[i]; j < m_1_bra.row[i + 1]; ++j) {
625 if (m_1_bra.val[j] != 0.0) {
626 delta_sector_set_m1.emplace(CalculateQuntumNumberDifference(i, m_1_bra.col[j]));
627 }
628 }
629 }
630
631 for (std::int64_t i = 0; i < m_2_bra.row_dim; ++i) {
632 for (std::int64_t j = m_2_bra.row[i]; j < m_2_bra.row[i + 1]; ++j) {
633 if (m_2_bra.val[j] != 0.0) {
634 delta_sector_set_m2.emplace(CalculateQuntumNumberDifference(i, m_2_bra.col[j]));
635 }
636 }
637 }
638
639 for (std::int64_t i = 0; i < m_3_ket.row_dim; ++i) {
640 for (std::int64_t j = m_3_ket.row[i]; j < m_3_ket.row[i + 1]; ++j) {
641 if (m_3_ket.val[j] != 0.0) {
642 delta_sector_set_m3.emplace(CalculateQuntumNumberDifference(i, m_3_ket.col[j]));
643 }
644 }
645 }
646
647 for (std::int64_t i = 0; i < m_4_ket.row_dim; ++i) {
648 for (std::int64_t j = m_4_ket.row[i]; j < m_4_ket.row[i + 1]; ++j) {
649 if (m_4_ket.val[j] != 0.0) {
650 delta_sector_set_m4.emplace(CalculateQuntumNumberDifference(i, m_4_ket.col[j]));
651 }
652 }
653 }
654
655 std::vector<std::tuple<QType, QType, QType>> target_sector_set;
656 for (const auto &del_sec_m1: delta_sector_set_m1) {
657 for (const auto &del_sec_m2: delta_sector_set_m2) {
658 for (const auto &del_sec_m3: delta_sector_set_m3) {
659 for (const auto &del_sec_m4: delta_sector_set_m4) {
660 const QType del_sec_m1_m2 = {del_sec_m1.first + del_sec_m2.first, del_sec_m1.second + del_sec_m2.second};
661 const QType del_sec_m3_m4 = {del_sec_m3.first + del_sec_m4.first, del_sec_m3.second + del_sec_m4.second};
662 const bool c1 = isValidQNumber(del_sec_m1.first + total_electron_, del_sec_m1.second + 0.5*total_2sz_);
663 const bool c2 = isValidQNumber(del_sec_m1_m2.first + total_electron_, del_sec_m1_m2.second + 0.5*total_2sz_);
664 const bool c3 = isValidQNumber(del_sec_m4.first + total_electron_, del_sec_m4.second + 0.5*total_2sz_);
665 if (del_sec_m1_m2 == del_sec_m3_m4 && c1 && c2 && c3) {
666 target_sector_set.push_back({
667 {del_sec_m1.first + total_electron_, del_sec_m1.second + 0.5*total_2sz_},
668 {del_sec_m1_m2.first + total_electron_, del_sec_m1_m2.second + 0.5*total_2sz_},
669 {del_sec_m4.first + total_electron_, del_sec_m4.second + 0.5*total_2sz_}
670 });
671 }
672 }
673 }
674 }
675 }
676
677 std::sort(target_sector_set.begin(), target_sector_set.end());
678 target_sector_set.erase(std::unique(target_sector_set.begin(), target_sector_set.end()), target_sector_set.end());
679 return target_sector_set;
680 }

688 const int dim_onsite_lspin = magnitude_2lspin + 1;

691 //--------------------------------
692 // # <-> [Cherge ] -- (N, 2*sz)
693 // 0 <-> [ ] -- (0, 0 )
694 // 1 <-> [up ] -- (1, 1 )
695 // 2 <-> [down ] -- (1, -1 )
696 // 3 <-> [up&down ] -- (2, 0 )
697 //--------------------------------
698
699 CRS matrix(dim_onsite, dim_onsite);

701 for (int element = 0; element < dim_onsite_lspin; ++element) {
702 matrix.val.push_back(1.0);
703 matrix.col.push_back(dim_onsite_lspin + element);
