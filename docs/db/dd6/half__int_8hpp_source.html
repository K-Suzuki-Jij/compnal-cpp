40 template<typename ValueType>
41 HalfInt(const ValueType value) {
42 if (std::floor(2*value) != 2*value) {
43 std::stringstream ss;
44 ss << "The input number " << value << " is not half-integer." << std::endl;
45 throw std::runtime_error(ss.str());
46 }
47 integer_ = static_cast<int>(2*value);
48 };
