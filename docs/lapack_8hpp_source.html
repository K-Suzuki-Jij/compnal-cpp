29template <typename RealType>
30void LapackDsyev(RealType *gs_value,
31 std::vector<RealType> *gs_vector,
32 const CRS<RealType> &matrix_in,
33 const int target_level = 0
34 ) {
35
36 if (matrix_in.row_dim != matrix_in.col_dim || matrix_in.row_dim < 1 || matrix_in.col_dim < 1) {
37 std::stringstream ss;
38 ss << "Error in " << __func__ << std::endl;
39 ss << "The input matrix is not a square one" << std::endl;
40 ss << "row=" << matrix_in.row_dim << ", col=" << matrix_in.col_dim << std::endl;
41 throw std::runtime_error(ss.str());
42 }
43
44 if (target_level < 0) {
45 std::stringstream ss;
46 ss << "Invalid target_level" << std::endl;
47 throw std::runtime_error(ss.str());
48 }
49
50 double matrix_array[matrix_in.row_dim][matrix_in.col_dim];
51
52 for (std::int64_t i = 0; i < matrix_in.row_dim; ++i) {
53 for (std::int64_t j = 0; j < matrix_in.col_dim; ++j) {
54 matrix_array[i][j] = 0.0;
55 }
56 }
57
58 for (std::int64_t i = 0; i < matrix_in.row_dim; ++i) {
59 for (std::int64_t j = matrix_in.row[i]; j < matrix_in.row[i + 1]; ++j) {
60 matrix_array[i][matrix_in.col[j]] = static_cast<double>(matrix_in.val[j]);
61 matrix_array[matrix_in.col[j]][i] = static_cast<double>(matrix_in.val[j]);
62 }
63 }
64
65 int info;
66 double val_array[matrix_in.row_dim];
67 double work[3 * matrix_in.row_dim];
68
69 dsyev_('V', 'L', static_cast<int>(matrix_in.row_dim),
70 (double**)matrix_array,
71 static_cast<int>(matrix_in.row_dim), val_array, work,
72 static_cast<int>(3 * matrix_in.row_dim), info);
73
74 gs_vector->resize(matrix_in.row_dim);
75
76 for (std::int64_t i = 0; i < matrix_in.row_dim; ++i) {
77 (*gs_vector)[i] = static_cast<RealType>(matrix_array[target_level][i]);
78 }
79
80 *gs_value = static_cast<RealType>(val_array[target_level]);
81}

83template <typename RealType>
84void LapackDstev(RealType *gs_value,
85 std::vector<RealType> *gs_vector,
86 const std::vector<RealType> &diag,
87 const std::vector<RealType> &off_diag) {
88
89 if (off_diag.size() + 1 != diag.size()) {
90 std::stringstream ss;
91 ss << "Error in " << __func__ << std::endl;
92 ss << "diag size=" << diag.size() << ", off_diag size=" << off_diag.size() << std::endl;
93 throw std::runtime_error(ss.str());
94 }
95
96 int dim = static_cast<int>(diag.size());
97
98 int info;
99 double Lap_D[dim];
100 double Lap_E[dim - 1];
101 double Lap_Vec[dim][dim];
102 double Lap_Work[2*dim];
103
104 for (int i = 0; i < dim; i++) {
105 Lap_D[i] = static_cast<double>(diag[i]);
106 }
107
108 for (int i = 0; i < dim - 1; i++) {
109 Lap_E[i] = static_cast<double>(off_diag[i]);
110 }
111
112 dstev_('V', dim, Lap_D, Lap_E, (double**)Lap_Vec, dim, Lap_Work, info);
113
114 gs_vector->resize(dim);
115
116 for (int i = 0; i < dim; ++i) {
117 (*gs_vector)[i] = static_cast<RealType>(Lap_Vec[0][i]);
118 }
119
120 *gs_value = static_cast<RealType>(Lap_D[0]);
121}

126 const int i_type,

145 int lap_info;
146

149 lap_bp[i] = mat_b[i];

151
152 dspgv_(i_type, 'V', 'U', dim, lap_ap, lap_bp, lap_w, (double**)lap_z, lap_ldz, lap_work, lap_info);
153
154 eigenvalues->resize(dim);

156
157 for (int i = 0; i < dim; ++i) {
158 (*eigenvalues)[i] = static_cast<RealType>(lap_w[i]);
159 for (int j = 0; j < dim; ++j) {
160 (*eigenvectors)[i*dim + j] = static_cast<RealType>(lap_z[i][j]);
161 }
162 }
163
164}
