119 int CalculateNumElectron(const int basis_onsite) const {
120
121 //--------------------------------
122 // # <-> [Cherge ] -- (N, 2*sz)
123 // 0 <-> [ ] -- (0, 0 )
124 // 1 <-> [up ] -- (1, 1 )
125 // 2 <-> [down ] -- (1, -1 )
126 // 3 <-> [up&down ] -- (2, 0 )
127 //--------------------------------
128
129 if (basis_onsite == 0) {
130 return 0;
131 }
132 else if (basis_onsite == 1 || basis_onsite == 2) {
133 return 1;
134 }
135 else if (basis_onsite == 3) {
136 return 2;
137 }
138 else {
139 std::stringstream ss;
140 ss << "Error at " << __LINE__ << " in " << __func__ << " in "<< __FILE__ << std::endl;
141 ss << "Invalid onsite basis" << std::endl;
142 throw std::runtime_error(ss.str());
143 }

159 template<typename IntegerType>
160 QType CalculateQuntumNumberDifference(const IntegerType row, const IntegerType col) const {
161 static_assert(std::is_integral<IntegerType>::value, "Template parameter IntegerType must be integer type");
162 if (row < 0 || col < 0 || dim_onsite_ <= row || dim_onsite_ <= col) {
163 std::stringstream ss;
164 ss << "Error at " << __LINE__ << " in " << __func__ << " in "<< __FILE__ << std::endl;
165 ss << "Invalid parameters" << std::endl;
166 throw std::runtime_error(ss.str());
167 }
168 if (row == col && 0 <= row && row < 4 && 0 <= col && col < 4) {
169 return {+0 + total_electron_, +0.0 + total_sz_};
170 }
171 else if (row == 0 && col == 1) {
172 return {-1 + total_electron_, -0.5 + total_sz_};
173 }
174 else if (row == 0 && col == 2) {
175 return {-1 + total_electron_, +0.5 + total_sz_};
176 }
177 else if (row == 0 && col == 3) {
178 return {-2 + total_electron_, +0.0 + total_sz_};
179 }
180 else if (row == 1 && col == 0) {
181 return {+1 + total_electron_, +0.5 + total_sz_};
182 }
183 else if (row == 1 && col == 2) {
184 return {+0 + total_electron_, +1.0 + total_sz_};
185 }
186 else if (row == 1 && col == 3) {
187 return {-1 + total_electron_, +0.5 + total_sz_};
188 }
189 else if (row == 2 && col == 0) {
190 return {+1 + total_electron_, -0.5 + total_sz_};
191 }
192 else if (row == 2 && col == 1) {
193 return {+0 + total_electron_, -1.0 + total_sz_};
194 }
195 else if (row == 2 && col == 3) {
196 return {-1 + total_electron_, -0.5 + total_sz_};
197 }
198 else if (row == 3 && col == 0) {
199 return {+2 + total_electron_, +0.0 + total_sz_};
200 }
201 else if (row == 3 && col == 1) {
202 return {+1 + total_electron_, -0.5 + total_sz_};
203 }
204 else if (row == 3 && col == 2) {
205 return {+1 + total_electron_, +0.5 + total_sz_};
206 }
207 else {
208 std::stringstream ss;
209 ss << "Error in " << __func__ << std::endl;
210 ss << "The dimenstion of the matrix must be 4";

220 std::vector<std::int64_t> GenerateBasis(const int system_size,
221 const QType &quantum_number,
222 const bool flag_display_info = true) const {
223 const auto start = std::chrono::system_clock::now();
224
225 if (!ValidateQNumber(quantum_number)) {
226 std::stringstream ss;
227 ss << "Error at " << __LINE__ << " in " << __func__ << " in "<< __FILE__ << std::endl;
228 ss << "Invalid parameters (system_size or total_electron or total_sz)" << std::endl;
229 throw std::runtime_error(ss.str());
230 }
231
232 if (flag_display_info) {
233 std::cout << "Generating Basis..." << std::flush;
234 }
235
236 const int total_electron = quantum_number.first;
237 const HalfInt total_sz = quantum_number.second;
238 const int total_2sz = 2*total_sz;
239
240 std::vector<std::int64_t> site_constant(system_size);
241 for (int site = 0; site < system_size; ++site) {
242 site_constant[site] = static_cast<std::int64_t>(std::pow(dim_onsite_, site));
243 }
244
245 const int max_n_up_down = static_cast<int>(total_electron/2);
246
247 std::vector<std::vector<int>> partition_integers;
248 for (int n_up_down = 0; n_up_down <= max_n_up_down; ++n_up_down) {
249 const int n_up = (total_electron - 2*n_up_down + total_2sz)/2;
250 const int n_down = (total_electron - 2*n_up_down - total_2sz)/2;
251 const int n_vac = system_size - total_electron + n_up_down;
252 if (0 <= n_up && 0 <= n_down && 0 <= n_vac) {
253 std::vector<int> integer_list(system_size);
254 for (int s = 0; s < n_vac; ++s) {
255 integer_list[s] = 0;
256 }
257 for (int s = 0; s < n_up; ++s) {
258 integer_list[s + n_vac] = 1;
259 }
260 for (int s = 0; s < n_down; ++s) {
261 integer_list[s + n_vac + n_up] = 2;
262 }
263 for (int s = 0; s < n_up_down; ++s) {
264 integer_list[s + n_vac + n_up + n_down] = 3;
265 }
266 partition_integers.push_back(integer_list);
267 }
268 }
269
270 const std::int64_t dim_target = CalculateTargetDim(system_size, quantum_number.first, quantum_number.second);
271 std::vector<std::int64_t> basis;
272 basis.reserve(dim_target);
273
274#ifdef _OPENMP
275 const int num_threads = omp_get_max_threads();
276 std::vector<std::vector<std::int64_t>> temp_basis(num_threads);
277 for (const auto &integer_list: partition_integers) {
278 const std::int64_t size = utility::CalculateNumPermutation(integer_list);
279#pragma omp parallel num_threads (num_threads)
280 {
281 const int thread_num = omp_get_thread_num();
282 const std::int64_t loop_begin = thread_num*size/num_threads;
283 const std::int64_t loop_end = (thread_num + 1)*size/num_threads;
284 std::vector<int> n_th_integer_list = utility::GenerateNthPermutation(integer_list, loop_begin);
285 for (std::int64_t j = loop_begin; j < loop_end; ++j) {
286 std::int64_t basis_global = 0;
287 for (std::size_t k = 0; k < n_th_integer_list.size(); ++k) {
288 basis_global += n_th_integer_list[k]*site_constant[k];
289 }
290 temp_basis[thread_num].push_back(basis_global);
291 std::next_permutation(n_th_integer_list.begin(), n_th_integer_list.end());
292 }
293 }
294 }
295 for (auto &&it: temp_basis) {
296 basis.insert(basis.end(), it.begin(), it.end());
297 std::vector<std::int64_t>().swap(it);
298 }
299#else
300 for (std::size_t i = 0; i < partition_integers.size(); ++i) {
301 auto &integer_list = partition_integers[i];
302 std::sort(integer_list.begin(), integer_list.end());
303 do {
304 std::int64_t basis_global = 0;
305 for (std::size_t j = 0; j < integer_list.size(); ++j) {
306 basis_global += integer_list[j]*site_constant[j];
307 }
308 basis.push_back(basis_global);
309 } while (std::next_permutation(integer_list.begin(), integer_list.end()));
310 }
311#endif
312 basis.shrink_to_fit();
313
314 if (static_cast<std::int64_t>(basis.size()) != dim_target) {
315 std::stringstream ss;
316 ss << "Unknown error at " << __LINE__ << " in " << __func__ << " in "<< __FILE__ << std::endl;
317 ss << "basis.size()=" << basis.size() << ", but dim_target=" << dim_target << std::endl;
318 throw std::runtime_error(ss.str());
319 }
320
321 std::sort(basis.begin(), basis.end());
322
323 if (flag_display_info) {
324 const auto time_count = std::chrono::duration_cast<std::chrono::microseconds>(std::chrono::system_clock::now() - start).count();
325 const double time_sec = static_cast<double>(time_count)/blas::TIME_UNIT_CONSTANT;
326 std::cout << "\rElapsed time of generating basis:" << time_sec << "[sec]" << std::endl;
327 }
328 return basis;
329 }

351 static bool ValidateQNumber(const int system_size, const int total_electron, const HalfInt total_sz) {
352 if (system_size < 0 || total_electron < 0) {
353 std::stringstream ss;
354 ss << "Error at " << __LINE__ << " in " << __func__ << " in "<< __FILE__ << std::endl;
355 ss << "Invalid parameters (system_size or total_electron)" << std::endl;
356 }
357 const int total_2sz = 2*total_sz;
358 const bool c1 = (0 <= total_electron && total_electron <= 2*system_size);
359 const bool c2 = ((total_electron - total_2sz)%2 == 0);
360 const bool c3 = (-total_electron <= total_2sz && total_2sz <= total_electron);
361 if (c1 && c2 && c3) {
362 return true;
363 }
364 else {
365 return false;
366 }
367 }

374 static std::int64_t CalculateTargetDim(const int system_size, const int total_electron, const HalfInt total_sz) {
375 if (!ValidateQNumber(system_size, total_electron, total_sz)) {
376 return 0;
377 }
378 if (system_size <= 0) {
379 return 0;
380 }
381 const int total_2sz = 2*total_sz;
382 const std::vector<std::vector<std::int64_t>> binom = utility::GenerateBinomialTable(system_size);
383 const int max_n_up_down = static_cast<int>(total_electron/2);
384 std::int64_t dim = 0;
385 for (int n_up_down = 0; n_up_down <= max_n_up_down; ++n_up_down) {
386 const int n_up = static_cast<int>((total_electron - 2*n_up_down + total_2sz)/2);
387 const int n_down = static_cast<int>((total_electron - 2*n_up_down - total_2sz)/2);
388 const int n_vac = system_size - total_electron + n_up_down;
389 if (0 <= n_up && 0 <= n_down && 0 <= n_vac) {
390 // TODO: Detect Overflow
391 dim += binom[system_size][n_up]*binom[system_size - n_up][n_down]*binom[system_size - n_up - n_down][n_up_down];
392 }
393 }
394 return dim;
395 }

399 static CRS CreateOnsiteOperatorCUp() {
400
401 //--------------------------------
402 // # <-> [Cherge ] -- (N, 2*sz)
403 // 0 <-> [ ] -- (0, 0 )
404 // 1 <-> [up ] -- (1, 1 )
405 // 2 <-> [down ] -- (1, -1 )
406 // 3 <-> [up&down ] -- (2, 0 )
407 //--------------------------------
408
409 const RealType val = RealType{1.0};
410 const int dim_onsite = 4;
411 CRS matrix(dim_onsite, dim_onsite);
412 for (int row = 0; row < dim_onsite; row++) {
413 for (int col = 0; col < dim_onsite; col++) {
414 if ((col == 1 && row == 0) || (col == 3 && row == 2)) {
415 matrix.col.push_back(col);
416 matrix.val.push_back(val);
417 }
418 }
419 matrix.row[row + 1] = matrix.col.size();
420 }
421
422 matrix.tag = type::CRSTag::FERMION;
423
424 return matrix;
425 }

429 static CRS CreateOnsiteOperatorCDown() {
430
431 //--------------------------------
432 // # <-> [Cherge ] -- (N, 2*sz)
433 // 0 <-> [ ] -- (0, 0 )
434 // 1 <-> [up ] -- (1, 1 )
435 // 2 <-> [down ] -- (1, -1 )
436 // 3 <-> [up&down ] -- (2, 0 )
437 //--------------------------------
438
439 const RealType val = RealType{1.0};
440 const int dim_onsite = 4;
441 CRS matrix(dim_onsite, dim_onsite);
442 for (int row = 0; row < dim_onsite; row++) {
443 for (int col = 0; col < dim_onsite; col++) {
444 if (col == 2 && row == 0) {
445 matrix.col.push_back(col);
446 matrix.val.push_back(val);
447 }
448 else if (col == 3 && row == 1) {
449 matrix.col.push_back(col);
450 matrix.val.push_back(-val);
451 }
452 }
453 matrix.row[row + 1] = matrix.col.size();
454 }
455
456 matrix.tag = type::CRSTag::FERMION;
457
458 return matrix;
