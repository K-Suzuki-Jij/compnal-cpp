34template<typename IntegerType>
35std::vector<std::vector<IntegerType>> GenerateIntegerPartition(const IntegerType partitioned_number,
36 IntegerType max_number,
37 const IntegerType max_size) {
38 static_assert(std::is_integral<IntegerType>::value, "Template parameter IntegerType must be integer type");
39
40 if (partitioned_number < 0 || max_number <= 0 || max_size <= 0) {
41 std::stringstream ss;
42 ss << "Error at " << __LINE__ << " in " << __func__ << " in "<< __FILE__ << std::endl;
43 ss << "Invalid input parameters" << std::endl;
44 ss << "partitioned_number=" << partitioned_number << std::endl;
45 ss << "max_number=" << max_number << std::endl;
46 ss << "max_size=" << max_size << std::endl;
47 throw std::runtime_error(ss.str());
48 }
49
50 if (partitioned_number == 0) {
51 std::vector<std::vector<IntegerType>>{std::vector<IntegerType>(max_size)};
52 }
53
54 auto generate_next = [](const std::vector<IntegerType> &vec) -> std::vector<IntegerType> {
55 std::int64_t size = static_cast<std::int64_t>(vec.size());
56 std::vector<IntegerType> out = vec;
57 for (std::int64_t i = size - 1; i >= 0; --i) {
58 if (vec[i] > 1) {
59 if (i + 1 < size) {
60 if (vec[i] - 1 >= vec[i + 1] + 1) {
61 out[i]--;
62 out[i + 1]++;
63 return out;
64 }
65 else {
66 out[i]--;
67 out.push_back(1);
68 return out;
69 }
70 }
71 else {
72 out[i]--;
73 out.push_back(1);
74 return out;
75 }
76 }
77 }
78 return std::vector<IntegerType>();
79 };
80
81 std::vector<std::vector<IntegerType>> out;
82
83 if (max_number >= partitioned_number) {
84 max_number = partitioned_number;
85 out.push_back({max_number});
86 }
87 else {
88 std::vector<IntegerType> temp = {max_number};
89 IntegerType rem = partitioned_number - max_number;
90 while (rem != 0) {
91 if (rem <= max_number) {
92 temp.push_back(rem);
93 rem = 0;
94 }
95 else {
96 temp.push_back(max_number);
97 rem -= max_number;
98 }
99 }
100 if (static_cast<IntegerType>(temp.size()) > max_size) {
101 return std::vector<std::vector<IntegerType>>();
102 }
103 else {
104 out.push_back(temp);
105 }
106 }
107
108 while (true) {
109 auto next_partition = generate_next(out.back());
110 if (0 < next_partition.size() && static_cast<IntegerType>(next_partition.size()) <= max_size) {
111 out.push_back(next_partition);
112 }
113 else {
114 break;
115 }
116 }
117
118 return out;
119
120}

137template<typename IntegerType>
138std::int64_t CalculateBinomialCoefficient(IntegerType n, const IntegerType k) {
139 static_assert(std::is_integral<IntegerType>::value, "Template parameter IntegerType must be integer type");
140
141 if (n < 0 || k < 0 || n < k) {
142 std::stringstream ss;

144 ss << "Invalid input parameters" << std::endl;
145 throw std::runtime_error(ss.str());
146 }
147
148 std::int64_t r = 1;
149
150 for (IntegerType d = 1; d <= k; d++) {
151 r *= n--;
152 r /= d;
153 }

162template<typename IntegerType>
163std::vector<std::vector<std::int64_t>> GenerateBinomialTable(const IntegerType n) {
164 static_assert(std::is_integral<IntegerType>::value, "Template parameter IntegerType must be integer type");
165
166 if (n < 0) {
167 std::stringstream ss;
168 ss << "Error at " << __LINE__ << " in " << __func__ << " in "<< __FILE__ << std::endl;
169 ss << "Invalid input parameters" << std::endl;
170 throw std::runtime_error(ss.str());
171 }
172
173 std::vector<std::vector<std::int64_t>> vec(n + 1);
174 for (IntegerType i = 0; i <= n; ++i) {
175 vec[i].resize(i + 1);
176 }
177
178 for (IntegerType i = 0; i <= n; ++i) {
179 for (IntegerType j = 0; j <= i; j++) {
180 if (j == 0 || j == i) {
181 vec[i][j] = 1;
182 }
183 else {
184 vec[i][j] = vec[i - 1][j - 1] + vec[i - 1][j];
185 }
186 }
187 }
188 return vec;
189}

197template<typename T>
198std::int64_t CalculateNumPermutation(const std::vector<T> &list) {
199 static_assert(!std::is_floating_point<T>::value, "Template parameter T must not be floating point");
200
201 std::unordered_map<T, std::int64_t> u_map;
202
203 for (const auto &it: list) {
204 u_map[it]++;
205 }
206
207 std::int64_t result = 1;
208 std::int64_t size_list = static_cast<std::int64_t>(list.size());
209
210 for (const auto &it: u_map) {
211 result *= CalculateBinomialCoefficient(size_list, it.second);
212 size_list -= it.second;
213 }
214
215 return result;
216
217}

227template<typename T>
228std::vector<T> GenerateNthPermutation(const std::vector<T> &list, const std::int64_t n) {
229 static_assert(!std::is_floating_point<T>::value, "Template parameter T must not be floating point");
230
231 if (n < 0) {
232 std::stringstream ss;
233 ss << "Error at " << __LINE__ << " in " << __func__ << " in "<< __FILE__ << std::endl;
234 ss << "Invalid input parameters" << std::endl;
235 throw std::runtime_error(ss.str());
236 }
237
238 std::int64_t size_list = static_cast<std::int64_t>(list.size());
239 std::int64_t rem = n + 1;
240 std::int64_t num_perm = CalculateNumPermutation(list);
241
242 if (n >= num_perm) {
243 std::stringstream ss;
244 ss << "Error at " << __LINE__ << " in " << __func__ << " in "<< __FILE__ << std::endl;
245 ss << "Invalid input parameters" << std::endl;
246 throw std::runtime_error(ss.str());
247 }
248
249 std::map<T, std::int64_t> map;
250
251 for (const auto &it: list) {
252 map[it]++;
253 }
254
255 std::vector<T> out;
256
257 while (true) {
258 std::size_t negative_count = 0;
259 for (auto &&it: map) {
260 if (it.second > 0) {
261 const std::int64_t count = (num_perm*it.second)/size_list;
262 if (rem <= count) {
263 out.push_back(it.first);
264 num_perm = count;
265 it.second--;
266 size_list--;
267 break;
268 }
269 rem -= count;
270 }
271 else {
272 negative_count++;
273 }
274 }
275 if (negative_count == map.size()) {
276 break;
277 }
278 }
279
280 return out;
