256 static CRS CreateOnsiteOperatorHam(const double magnitude_lspin, const RealType J_z, const RealType J_xy, const RealType h_z, const RealType D_z) {
257 const CRS spc = BaseU1SpinElectron_1D<RealType>::CreateOnsiteOperatorSpC(magnitude_lspin);
258 const CRS smc = BaseU1SpinElectron_1D<RealType>::CreateOnsiteOperatorSmC(magnitude_lspin);
259 const CRS szc = BaseU1SpinElectron_1D<RealType>::CreateOnsiteOperatorSzC(magnitude_lspin);
260 const CRS spl = BaseU1SpinElectron_1D<RealType>::CreateOnsiteOperatorSpL(magnitude_lspin);
261 const CRS sml = BaseU1SpinElectron_1D<RealType>::CreateOnsiteOperatorSmL(magnitude_lspin);
262 const CRS szl = BaseU1SpinElectron_1D<RealType>::CreateOnsiteOperatorSzL(magnitude_lspin);
263 return J_z*szc*szl + J_xy*0.5*(spc*sml + smc*spl) + h_z*(szc + szl) * D_z*szl*szl;
264 }

277 static CRS CreateOnsiteOperatorHam(const double magnitude_lspin, const RealType J, const RealType h_z, const RealType D_z) {
278 return CreateOnsiteOperatorHam(magnitude_lspin, J, J, h_z, D_z);
279 }
