174 template<typename ReturnType>
175 ReturnType L1Norm() const {
176 ReturnType norm = 0.0;
177#pragma omp parallel for reduction (+:norm)
178 for (std::size_t i = 0; i < this->value_list.size(); ++i) {
179 norm += std::abs(this->value_list[i]);
180 }
181 return norm;
182 }
183
184 template<typename ReturnType>
185 ReturnType L2Norm() const {
186 ReturnType inner_product = 0.0;
187#pragma omp parallel for reduction (+:inner_product)
188 for (std::size_t i = 0; i < this->value_list.size(); ++i) {
189 inner_product += this->value_list[i]*this->value_list[i];
190 }
191 return std::sqrt(inner_product);
192 }

285template<typename T1, typename T2, typename T3, typename T4>
286BraketVector<decltype(T1{0}*T2{0}+T3{0}*T4{0})> CalculateVectorVectorSum(const T1 coeef_1,
287 const BraketVector<T2> &braket_vector_1,
288 const T3 coeef_2,
289 const BraketVector<T4> &braket_vector_2) {
290 if (braket_vector_1.value_list.size() != braket_vector_2.value_list.size()) {
291 std::stringstream ss;
292 ss << "Error at " << __LINE__ << " in " << __func__ << " in "<< __FILE__ << std::endl;
293 ss << "BraketVector types do not match each other" << std::endl;
294 ss << "dim_1 = " << braket_vector_1.value_list.size() << ", dim_2 = " << braket_vector_2.value_list.size() << std::endl;
295 throw std::runtime_error(ss.str());
296 }
297 BraketVector<decltype(T1{0}*T2{0}+T3{0}*T4{0})> vector_out;
298 vector_out.Resize(braket_vector_1.value_list.size());
299#pragma omp parallel for
300 for (std::int64_t i = 0; i < braket_vector_1.value_list.size(); ++i) {
301 vector_out.value_list[i] = coeef_1*braket_vector_1.value_list[i] + coeef_2*braket_vector_2.value_list[i];
302 }
303 return vector_out;
304}

306template<typename T1, typename T2>
307decltype(T1{0}*T2{0}) CalculateVectorVectorProduct(const BraketVector<T1> &braket_vector_1,
308 const BraketVector<T2> &braket_vector_2) {
309 if (braket_vector_1.value_list.size() != braket_vector_2.value_list.size()) {
310 std::stringstream ss;
311 ss << "Error at " << __LINE__ << " in " << __func__ << " in "<< __FILE__ << std::endl;
312 ss << "BraketVector types do not match each other" << std::endl;
313 ss << "dim_1 = " << braket_vector_1.value_list.size() << ", dim_2 = " << braket_vector_2.value_list.size() << std::endl;
314 throw std::runtime_error(ss.str());
315 }
316 decltype(T1{0}*T2{0}) val_out = 0.0;
317#pragma omp parallel for reduction (+: val_out)
318 for (std::int64_t i = 0; i < braket_vector_1.value_list.size(); ++i) {
319 val_out += braket_vector_1.value_list[i]*braket_vector_2.value_list[i];
320 }
321 return val_out;
322}

338decltype(T1{0}*T2{0}-T3{0}*T4{0}) CalculateL1Norm(const T1 coeef_1,
339 const BraketVector<T2> &braket_vector_1,
