87 void CalculateGroundState(const bool flag_display_info = false) {
88 if (calculated_eigenvector_set_.count(0) != 0) {
89 return;
90 }
91
92 GenerateBasis(model.GetQNumber(), flag_display_info);
93
94 CRS ham;
95 GenerateHamiltonian(&ham);
96
97 if (eigenvalues_.size() == 0) {
98 eigenvalues_.emplace_back();
99 }
100 if (eigenvectors_.size() == 0) {
101 eigenvectors_.emplace_back();
102 }
103
104 if (diag_method_ == DiagMethod::LANCZOS) {
105 blas::EigenvalueDecompositionLanczos(&eigenvalues_[0], &eigenvectors_[0], ham, params.lanczos);
106 }
107 else if (diag_method_ == DiagMethod::LOBPCG) {
108 blas::EigenvalueDecompositionLOBPCG(&eigenvalues_[0], &eigenvectors_[0], ham, params.lanczos);
109 }
110 else {
111 std::stringstream ss;
112 ss << "Error in " << __func__ << std::endl;
113 ss << "Invalid diagonalization method detected." << std::endl;
114 throw std::runtime_error(ss.str());
115 }
116
117 blas::InverseIteration(&ham, &eigenvectors_[0], eigenvalues_[0], params.ii);
118
119 calculated_eigenvector_set_.emplace(0);
120 }

122 void CalculateTargetState(const int target_sector) {
123 if (target_sector < 0) {
124 std::stringstream ss;
125 ss << "Error in " << __func__ << std::endl;
126 ss << "Invalid target_sector: " << target_sector << std::endl;
127 throw std::runtime_error(ss.str());
128 }
129 CalculateGroundState();
130 if (model.GetCalculatedEigenvectorSet().count(target_sector) != 0) {
131 return;
132 }
133
134 model.GenerateBasis();
135 CRS ham;
136 GenerateHamiltonian(&ham);
137 if (diag_method_ == DiagMethod::LANCZOS) {
138 for (int sector = 1; sector <= target_sector; ++sector) {
139 if (model.GetCalculatedEigenvectorSet().count(sector) == 0) {
140 if (static_cast<int>(eigenvectors_.size()) != sector) {
141 std::stringstream ss;
142 ss << "Unknown Error in " << __func__ << std::endl;
143 ss << "sector: " << sector << ", size: " << eigenvectors_.size() << std::endl;
144 throw std::runtime_error(ss.str());
145 }
146 BraketVector temp_vector(ham.row_dim);
147 RealType temp_value = 0.0;
148 blas::EigenvalueDecompositionLanczos(&temp_value, &temp_vector, ham, sector, eigenvectors_, params.lanczos);
149 blas::InverseIteration(&ham, &temp_vector, temp_value, params.ii, eigenvectors_);
150 eigenvalues_.push_back(temp_value);
151 eigenvectors_.push_back(temp_vector);
152 model.SetCalculatedEigenvectorSet(sector);
153 }
154 }
155 }
156 else if (diag_method_ == DiagMethod::LOBPCG) {
157 std::stringstream ss;
158 ss << "Error in " << __func__ << std::endl;
159 ss << "LOBPCG is under construction: " << target_sector << std::endl;
160 throw std::runtime_error(ss.str());
161 }
162 else {
163 std::stringstream ss;
164 ss << "Error in " << __func__ << std::endl;
165 ss << "Invalid diag_method: " << target_sector << std::endl;
166 throw std::runtime_error(ss.str());
167 }

171 RealType CalculateExpectationValue(const CRS &m, const int site, const int level = 0) const {
172 if (model.GetCalculatedEigenvectorSet().count(level) == 0) {
173 std::stringstream ss;
174 ss << "Error in " << __func__ << std::endl;
175 ss << "An eigenvector of the energy level: " << level << " has not been calculated" << std::endl;
176 throw std::runtime_error(ss.str());
177 }
178
179 const auto &basis = model.GetTargetBasis();
180 const auto &basis_inv = model.GetTargetBasisInv();
181
182 const std::int64_t dim = eigenvectors_.at(level).val.size();
183 if (static_cast<std::int64_t>(basis.size()) != dim || static_cast<std::int64_t>(basis_inv.size()) != dim) {
184 std::stringstream ss;
185 ss << "Error in " << __func__ << std::endl;
186 ss << "Unknown error detected" << std::endl;
187 throw std::runtime_error(ss.str());
188 }
189
190 const int dim_onsite = static_cast<int>(m.row_dim);
191 const std::int64_t site_constant = static_cast<std::int64_t>(std::pow(dim_onsite, site));
192 const BraketVector &eigenvector = eigenvectors_.at(level);
193 RealType val = 0.0;
194
195#pragma omp parallel for reduction (+: val)
196 for (std::int64_t i = 0; i < dim; ++i) {
197 const std::int64_t global_basis = basis[i];
198 const int local_basis = CalculateLocalBasis(global_basis, site, dim_onsite);
199 RealType temp_val = 0.0;
200 for (std::int64_t j = m.row[local_basis]; j < m.row[local_basis + 1]; ++j){
201 const std::int64_t a_basis = global_basis - (local_basis - m.col[j])*site_constant;
202 if (basis_inv.count(a_basis) != 0) {
203 temp_val += eigenvector.val[basis_inv.at(a_basis)]*m.val[j];
204 }
205 }
206 val += eigenvector.val[i]*temp_val;
207 }
208 return val;
209 }

215 RealType CalculateCorrelationFunction(const CRS &m_1, const int site_1, const CRS &m_2, const int site_2, const int target_level = 0) {
216 if (model.GetCalculatedEigenvectorSet().count(target_level) == 0) {
217 std::stringstream ss;
218 ss << "Error in " << __func__ << std::endl;
219 ss << "An eigenvector of the energy level: " << target_level << " has not been calculated" << std::endl;
220 throw std::runtime_error(ss.str());
221 }
222
223 if (m_1.row_dim != m_2.row_dim || m_1.row_dim != m_1.col_dim || m_2.row_dim != m_2.col_dim) {
224 std::stringstream ss;
225 ss << "Error in " << __func__ << std::endl;
226 ss << "Invalid input of the local operators" << std::endl;
227 throw std::runtime_error(ss.str());
228 }
229
230 const CRS m1_dagger = blas::CalculateTransposedMatrix(m_1);
231
232 const auto target_sectors = model.GenerateTargetSector(m1_dagger, m_2);
233 const auto &basis_inv = model.GetTargetBasisInv();
234 const int dim_onsite = static_cast<int>(m1_dagger.row_dim);
235 const std::int64_t site_constant_m1 = static_cast<std::int64_t>(std::pow(dim_onsite, site_1));
236 const std::int64_t site_constant_m2 = static_cast<std::int64_t>(std::pow(dim_onsite, site_2));
237 const BraketVector &eigenvector = eigenvectors_.at(target_level);
238 BraketVector vector_work_m1;
239 BraketVector vector_work_m2;
240 RealType val = 0.0;
241
242 for (const auto &sector: target_sectors) {
243 model.GenerateBasis(sector);
244 const auto &basis = model.GetBasis(sector);
245 const std::int64_t dim_target = static_cast<std::int64_t>(basis.size());
246 vector_work_m1.val.resize(dim_target);
247 vector_work_m2.val.resize(dim_target);
248#pragma omp parallel for
249 for (std::int64_t i = 0; i < dim_target; ++i) {
250 const std::int64_t global_basis = basis[i];
251 const int local_basis_m1 = CalculateLocalBasis(global_basis, site_1, dim_onsite);
252 const int local_basis_m2 = CalculateLocalBasis(global_basis, site_2, dim_onsite);
253 RealType temp_val_m1 = 0.0;
254 RealType temp_val_m2 = 0.0;
255
256 int fermion_sign_m1 = 1;
257 if (m_1.tag == type::CRSTag::FERMION) {
258 int num_electron = 0;
259 for (int site = 0; site < site_1; site++) {
260 num_electron += model.CalculateNumElectron(CalculateLocalBasis(global_basis, site, dim_onsite));
261 }
262 if (num_electron%2 == 1) {
263 fermion_sign_m1 = -1;
264 }
265 }
266
267 for (std::int64_t j = m1_dagger.row[local_basis_m1]; j < m1_dagger.row[local_basis_m1 + 1]; ++j){
268 const std::int64_t a_basis = global_basis - (local_basis_m1 - m1_dagger.col[j])*site_constant_m1;
269 if (basis_inv.count(a_basis) != 0) {
270 temp_val_m1 += eigenvector.val[basis_inv.at(a_basis)]*m1_dagger.val[j];
271 }
272 }
273 vector_work_m1.val[i] = temp_val_m1*fermion_sign_m1;
274
275 int fermion_sign_m2 = 1;
276 if (m_2.tag == type::CRSTag::FERMION) {
277 int num_electron = 0;
278 for (int site = 0; site < site_2; site++) {
279 num_electron += model.CalculateNumElectron(CalculateLocalBasis(global_basis, site, dim_onsite));
280 }
281 if (num_electron%2 == 1) {
282 fermion_sign_m2 = -1;
283 }
284 }
285
286 for (std::int64_t j = m_2.row[local_basis_m2]; j < m_2.row[local_basis_m2 + 1]; ++j){
287 const std::int64_t a_basis = global_basis - (local_basis_m2 - m_2.col[j])*site_constant_m2;
288 if (basis_inv.count(a_basis) != 0) {
289 temp_val_m2 += eigenvector.val[basis_inv.at(a_basis)]*m_2.val[j];
290 }
291 }
292 vector_work_m2.val[i] = temp_val_m2*fermion_sign_m2;
293 }
294 val += blas::CalculateInnerProduct(vector_work_m1, vector_work_m2);
295 }
296 return val;
297 }

299 RealType CalculateCorrelationFunction(const CRS &m_1, const int site_1,
300 const CRS &m_2, const int site_2,
301 const CRS &m_3, const int site_3,
302 const int target_level = 0) {
303
304 if (model.GetCalculatedEigenvectorSet().count(target_level) == 0) {
305 std::stringstream ss;
306 ss << "Error in " << __func__ << std::endl;
307 ss << "An eigenvector of the energy level: " << target_level << " has not been calculated" << std::endl;
308 throw std::runtime_error(ss.str());
309 }
310
311 const bool c1 = (m_1.row_dim == m_1.col_dim);
312 const bool c2 = (m_2.row_dim == m_2.col_dim);
313 const bool c3 = (m_3.row_dim == m_3.col_dim);
314 const bool c4 = (m_1.row_dim == m_2.row_dim && m_2.row_dim == m_3.row_dim);
315
316 if (!(c1 && c2 && c3 && c4)) {
317 std::stringstream ss;
318 ss << "Error in " << __func__ << std::endl;
319 ss << "Invalid input of the local operators" << std::endl;
320 throw std::runtime_error(ss.str());
321 }
322
323 if (site_1 == site_2) {
324 return CalculateCorrelationFunction(m_1*m_2, site_1, m_3, site_3, target_level);
325 }
326 if (site_2 == site_3) {
327 return CalculateCorrelationFunction(m_1, site_1, m_2*m_3, site_3, target_level);
328 }
329
330 const CRS m1_dagger = blas::CalculateTransposedMatrix(m_1);
331 const auto target_sectors = model.GenerateTargetSector(m1_dagger, m_2, m_3);
332
333 const auto &basis_gs_sector_inv = model.GetTargetBasisInv();
334 const int dim_onsite = static_cast<int>(m1_dagger.row_dim);
335 const std::int64_t site_constant_m1 = static_cast<std::int64_t>(std::pow(dim_onsite, site_1));
336 const std::int64_t site_constant_m2 = static_cast<std::int64_t>(std::pow(dim_onsite, site_2));
337 const std::int64_t site_constant_m3 = static_cast<std::int64_t>(std::pow(dim_onsite, site_3));
338 const BraketVector &eigenvector = eigenvectors_.at(target_level);
339 BraketVector vector_work_a;
340 BraketVector vector_work_b;
341 BraketVector vector_work_c;
342 RealType val = 0.0;
343
344 for (const auto &it: target_sectors) {
345 const auto sector_m1 = std::get<0>(it);
346 const auto sector_m3 = std::get<1>(it);
347 model.GenerateBasis(sector_m1);
348 model.GenerateBasis(sector_m3);
349 const auto &basis_m1 = model.GetBasis(sector_m1);
350 const auto &basis_m3 = model.GetBasis(sector_m3);
351 const std::int64_t dim_target_m1 = static_cast<std::int64_t>(basis_m1.size());
352 const std::int64_t dim_target_m3 = static_cast<std::int64_t>(basis_m3.size());
353
354 //m3|gs>
355 vector_work_a.val.resize(dim_target_m3);
356#pragma omp parallel for
357 for (std::int64_t i = 0; i < dim_target_m3; ++i) {
358 const std::int64_t global_basis = basis_m3[i];
359 const int local_basis = CalculateLocalBasis(global_basis, site_3, dim_onsite);
360 RealType temp_val = 0.0;
361
362 int fermion_sign_m3 = 1;
363 if (m_3.tag == type::CRSTag::FERMION) {
364 int num_electron = 0;
365 for (int site = 0; site < site_1; site++) {
366 num_electron += model.CalculateNumElectron(CalculateLocalBasis(global_basis, site, dim_onsite));
367 }
368 if (num_electron%2 == 1) {
369 fermion_sign_m3 = -1;
370 }
371 }
372
373 for (std::int64_t j = m_3.row[local_basis]; j < m_3.row[local_basis + 1]; ++j){
374 const std::int64_t a_basis = global_basis - (local_basis - m_3.col[j])*site_constant_m3;
375 if (basis_gs_sector_inv.count(a_basis) != 0) {
376 temp_val += eigenvector.val[basis_gs_sector_inv.at(a_basis)]*m_3.val[j];
377 }
378 }
379 vector_work_a.val[i] = temp_val*fermion_sign_m3;
380 }
381
382 //m2 * m3|gs> and m1_dag|gs>
383 vector_work_b.val.resize(dim_target_m1);
384 vector_work_c.val.resize(dim_target_m1);
385 const auto &basis_m3_sector_inv = model.GetBasisInv(sector_m3);
386#pragma omp parallel for
387 for (std::int64_t i = 0; i < dim_target_m1; ++i) {
388 const std::int64_t global_basis = basis_m1[i];
389 const int local_basis_m1 = CalculateLocalBasis(global_basis, site_1, dim_onsite);
390 const int local_basis_m2 = CalculateLocalBasis(global_basis, site_2, dim_onsite);
391 RealType temp_val_m1 = 0.0;
392 RealType temp_val_m2 = 0.0;
393
394 int fermion_sign_m2 = 1;
395 if (m_2.tag == type::CRSTag::FERMION) {
396 int num_electron = 0;
397 for (int site = 0; site < site_1; site++) {
398 num_electron += model.CalculateNumElectron(CalculateLocalBasis(global_basis, site, dim_onsite));
399 }
400 if (num_electron%2 == 1) {
401 fermion_sign_m2 = -1;
402 }
403 }
404
405 for (std::int64_t j = m_2.row[local_basis_m2]; j < m_2.row[local_basis_m2 + 1]; ++j){
406 const std::int64_t a_basis = global_basis - (local_basis_m2 - m_2.col[j])*site_constant_m2;
407 if (basis_m3_sector_inv.count(a_basis) != 0) {
408 temp_val_m2 += vector_work_a.val[basis_m3_sector_inv.at(a_basis)]*m_2.val[j];
409 }
410 }
411 vector_work_b.val[i] = temp_val_m2*fermion_sign_m2;
412
413 int fermion_sign_m1 = 1;
414 if (m_1.tag == type::CRSTag::FERMION) {
415 int num_electron = 0;
416 for (int site = 0; site < site_1; site++) {
417 num_electron += model.CalculateNumElectron(CalculateLocalBasis(global_basis, site, dim_onsite));
418 }
419 if (num_electron%2 == 1) {
420 fermion_sign_m1 = -1;
421 }
422 }
423
424 for (std::int64_t j = m_1.row[local_basis_m1]; j < m_1.row[local_basis_m1 + 1]; ++j){
425 const std::int64_t a_basis = global_basis - (local_basis_m1 - m_1.col[j])*site_constant_m1;
426 if (basis_gs_sector_inv.count(a_basis) != 0) {
427 temp_val_m1 += eigenvector.val[basis_gs_sector_inv.at(a_basis)]*m_1.val[j];
428 }
429 }
430 vector_work_c.val[i] = temp_val_m1*fermion_sign_m1;
431 }
432 val += blas::CalculateInnerProduct(vector_work_b, vector_work_c);
433 }
434 return val;
435 }

437 RealType CalculateCorrelationFunction(const CRS &m_1, const int site_1,
438 const CRS &m_2, const int site_2,
439 const CRS &m_3, const int site_3,
440 const CRS &m_4, const int site_4,
441 const int target_level = 0) {
442
443 if (model.GetCalculatedEigenvectorSet().count(target_level) == 0) {
444 std::stringstream ss;
445 ss << "Error in " << __func__ << std::endl;
446 ss << "An eigenvector of the energy level: " << target_level << " has not been calculated" << std::endl;
447 throw std::runtime_error(ss.str());
448 }
449
450 const bool c1 = (m_1.row_dim == m_1.col_dim);
451 const bool c2 = (m_2.row_dim == m_2.col_dim);
452 const bool c3 = (m_3.row_dim == m_3.col_dim);
453 const bool c4 = (m_4.row_dim == m_4.col_dim);
454 const bool c5 = (m_1.row_dim == m_2.row_dim && m_2.row_dim == m_3.row_dim && m_3.row_dim == m_4.row_dim);
455
456 if (!(c1 && c2 && c3 && c4 && c5)) {
457 std::stringstream ss;
458 ss << "Error in " << __func__ << std::endl;
459 ss << "Invalid input of the local operators" << std::endl;
460 throw std::runtime_error(ss.str());
461 }
462
463 if (site_1 == site_2) {
464 return CalculateCorrelationFunction(m_1*m_2, site_1, m_3, site_3, m_4, site_4);
465 }
466 if (site_2 == site_3) {
467 return CalculateCorrelationFunction(m_1, site_1, m_2*m_3, site_2, m_4, site_4);
468 }
469 if (site_3 == site_4) {
470 return CalculateCorrelationFunction(m_1, site_1, m_2, site_2, m_3*m_4, site_3);
471 }
472
473 const CRS m1_dagger = blas::CalculateTransposedMatrix(m_1);
474 const CRS m2_dagger = blas::CalculateTransposedMatrix(m_2);
475 const auto target_sectors = model.GenerateTargetSector(m1_dagger, m2_dagger, m_3, m_4);
476
477 const auto &basis_gs_sector_inv = model.GetTargetBasisInv();
478 const int dim_onsite = static_cast<int>(m1_dagger.row_dim);
479 const std::int64_t site_constant_m1 = static_cast<std::int64_t>(std::pow(dim_onsite, site_1));
480 const std::int64_t site_constant_m2 = static_cast<std::int64_t>(std::pow(dim_onsite, site_2));
481 const std::int64_t site_constant_m3 = static_cast<std::int64_t>(std::pow(dim_onsite, site_3));
482 const std::int64_t site_constant_m4 = static_cast<std::int64_t>(std::pow(dim_onsite, site_4));
483 const BraketVector &eigenvector = eigenvectors_.at(target_level);
484 BraketVector vector_work_m1;
485 BraketVector vector_work_m2m1;
486 BraketVector vector_work_m4;
487 BraketVector vector_work_m3m4;
488 RealType val = 0.0;
489
490 for (const auto &it: target_sectors) {
491 const auto sector_bra_change_1 = std::get<0>(it);
492 const auto sector_bra_change_2 = std::get<1>(it);
493 const auto sector_ket_change_1 = std::get<2>(it);
494 model.GenerateBasis(sector_bra_change_1);
495 model.GenerateBasis(sector_bra_change_2);
496 model.GenerateBasis(sector_ket_change_1);
497 const auto &basis_bra_1 = model.GetBasis(sector_bra_change_1);
498 const auto &basis_bra_2 = model.GetBasis(sector_bra_change_2);
499 const auto &basis_ket_1 = model.GetBasis(sector_ket_change_1);
500 const auto &basis_ket_1_inv = model.GetBasisInv(sector_ket_change_1);
501 const auto &basis_bra_1_inv = model.GetBasisInv(sector_bra_change_1);
502 const std::int64_t dim_target_bra_1 = static_cast<std::int64_t>(basis_bra_1.size());
503 const std::int64_t dim_target_bra_2 = static_cast<std::int64_t>(basis_bra_2.size());
504 const std::int64_t dim_target_ket_1 = static_cast<std::int64_t>(basis_ket_1.size());
505
506 //m4|gs>
507 vector_work_m4.val.resize(dim_target_ket_1);
508#pragma omp parallel for
509 for (std::int64_t i = 0; i < dim_target_ket_1; ++i) {
510 const std::int64_t global_basis = basis_ket_1[i];
511 const int local_basis = CalculateLocalBasis(global_basis, site_4, dim_onsite);
512 RealType temp_val = 0.0;
513
514 int fermion_sign_m4 = 1;
515 if (m_4.tag == type::CRSTag::FERMION) {
516 int num_electron = 0;
517 for (int site = 0; site < site_1; site++) {
518 num_electron += model.CalculateNumElectron(CalculateLocalBasis(global_basis, site, dim_onsite));
519 }
520 if (num_electron%2 == 1) {
521 fermion_sign_m4 = -1;
522 }
523 }
524
525 for (std::int64_t j = m_4.row[local_basis]; j < m_4.row[local_basis + 1]; ++j){
526 const std::int64_t a_basis = global_basis - (local_basis - m_4.col[j])*site_constant_m4;
527 if (basis_gs_sector_inv.count(a_basis) != 0) {
528 temp_val += eigenvector.val[basis_gs_sector_inv.at(a_basis)]*m_4.val[j];
529 }
530 }
531 vector_work_m4.val[i] = temp_val*fermion_sign_m4;
532 }
533
534 //m1_dag|gs>
535 vector_work_m1.val.resize(dim_target_bra_1);
536#pragma omp parallel for
537 for (std::int64_t i = 0; i < dim_target_bra_1; ++i) {
538 const std::int64_t global_basis = basis_bra_1[i];
539 const int local_basis = CalculateLocalBasis(global_basis, site_1, dim_onsite);
540 RealType temp_val = 0.0;
541
542 int fermion_sign_m1 = 1;
543 if (m_1.tag == type::CRSTag::FERMION) {
544 int num_electron = 0;
545 for (int site = 0; site < site_1; site++) {
546 num_electron += model.CalculateNumElectron(CalculateLocalBasis(global_basis, site, dim_onsite));
547 }
548 if (num_electron%2 == 1) {
549 fermion_sign_m1 = -1;
550 }
551 }
552
553 for (std::int64_t j = m1_dagger.row[local_basis]; j < m1_dagger.row[local_basis + 1]; ++j){
554 const std::int64_t a_basis = global_basis - (local_basis - m1_dagger.col[j])*site_constant_m1;
555 if (basis_gs_sector_inv.count(a_basis) != 0) {
556 temp_val += eigenvector.val[basis_gs_sector_inv.at(a_basis)]*m1_dagger.val[j];
557 }
558 }
559 vector_work_m1.val[i] = temp_val*fermion_sign_m1;
560 }
561
562 //m3 * m4|gs> and m2_dag * m1_dag|gs>
563 vector_work_m3m4.val.resize(dim_target_bra_2);
564 vector_work_m2m1.val.resize(dim_target_bra_2);
565
566#pragma omp parallel for
567 for (std::int64_t i = 0; i < dim_target_bra_2; ++i) {
568 const std::int64_t global_basis = basis_bra_2[i];
569 const int local_basis_m2 = CalculateLocalBasis(global_basis, site_2, dim_onsite);
570 const int local_basis_m3 = CalculateLocalBasis(global_basis, site_3, dim_onsite);
571 RealType temp_val = 0.0;
572
573 int fermion_sign_m3 = 1;
574 if (m_3.tag == type::CRSTag::FERMION) {
575 int num_electron = 0;
576 for (int site = 0; site < site_1; site++) {
577 num_electron += model.CalculateNumElectron(CalculateLocalBasis(global_basis, site, dim_onsite));
578 }
579 if (num_electron%2 == 1) {
580 fermion_sign_m3 = -1;
581 }
582 }
583
584 for (std::int64_t j = m_3.row[local_basis_m3]; j < m_3.row[local_basis_m3 + 1]; ++j){
585 const std::int64_t a_basis = global_basis - (local_basis_m3 - m_3.col[j])*site_constant_m3;
586 if (basis_ket_1_inv.count(a_basis) != 0) {
587 temp_val += vector_work_m4.val[basis_ket_1_inv.at(a_basis)]*m_3.val[j];
588 }
589 }
590 vector_work_m3m4.val[i] = temp_val*fermion_sign_m3;
591
592 temp_val = 0.0;
593
594 int fermion_sign_m2 = 1;
595 if (m_2.tag == type::CRSTag::FERMION) {
596 int num_electron = 0;
597 for (int site = 0; site < site_1; site++) {
598 num_electron += model.CalculateNumElectron(CalculateLocalBasis(global_basis, site, dim_onsite));
599 }
600 if (num_electron%2 == 1) {
601 fermion_sign_m2 = -1;
602 }
603 }
604
605 for (std::int64_t j = m2_dagger.row[local_basis_m2]; j < m2_dagger.row[local_basis_m2 + 1]; ++j){
606 const std::int64_t a_basis = global_basis - (local_basis_m2 - m2_dagger.col[j])*site_constant_m2;
607 if (basis_bra_1_inv.count(a_basis) != 0) {
608 temp_val += vector_work_m1.val[basis_bra_1_inv.at(a_basis)]*m2_dagger.val[j];
609 }
610 }
611 vector_work_m2m1.val[i] = temp_val*fermion_sign_m2;
612 }
613 val += blas::CalculateInnerProduct(vector_work_m2m1, vector_work_m3m4);
614 }
615 return val;

774 }
775
776 const int basis_onsite = edmc->basis_onsite[site];

781 if (edmc->inv_basis_affected.count(a_basis) == 0) {
782 edmc->inv_basis_affected[a_basis] = edmc->basis_affected.size();
783 edmc->val.push_back(coeef*matrix_onsite.val[i]);
784 edmc->basis_affected.push_back(a_basis);
785 }
786 else {
787 edmc->val[edmc->inv_basis_affected.at(a_basis)] += coeef*matrix_onsite.val[i];

796 const int site_2,
797 const CRS &matrix_onsite_2,
798 const RealType coeef,
799 const int fermion_sign = 1.0) const {
800
801 if (std::abs(coeef) <= edmc->zero_precision) {
802 return;
803 }
804
805 const int basis_onsite_1 = edmc->basis_onsite[site_1];

815 if (edmc->inv_basis_affected.count(a_basis) == 0) {
816 edmc->inv_basis_affected[a_basis] = edmc->basis_affected.size();
817 edmc->val.push_back(fermion_sign*coeef*val_1*matrix_onsite_2.val[i2]);
818 edmc->basis_affected.push_back(a_basis);
819 }
820 else {
821 edmc->val[edmc->inv_basis_affected.at(a_basis)] += fermion_sign*coeef*val_1*matrix_onsite_2.val[i2];
822 }
823 }

995 void GenerateMatrixComponents(ExactDiagMatrixComponents<RealType> *edmc,
996 const std::int64_t basis,
997 const model::Hubbard_1D<RealType> &model_input) const {
998
999 const auto &nc = model_input.GetOnsiteOperatorNC();
1000 const auto &c_up = model_input.GetOnsiteOperatorCUp();
1001 const auto &c_up_dagger = model_input.GetOnsiteOperatorCUpDagger();
1002 const auto &c_down = model_input.GetOnsiteOperatorCDown();
1003 const auto &c_down_dagger = model_input.GetOnsiteOperatorCDownDagger();
1004
1005 for (int site = 0; site < model_input.GetSystemSize(); ++site) {
1006 edmc->basis_onsite[site] = CalculateLocalBasis(basis, site, model_input.GetDimOnsite());
1007 }
1008
1009 //Onsite elements
1010 for (int site = 0; site < model_input.GetSystemSize(); ++site) {
1011 GenerateMatrixComponentsOnsite(edmc, basis, site, model_input.GetOnsiteOperatorHam(), 1.0);
1012 }
1013
1014 //Intersite Coulomb
1015 for (int distance = 1; distance <= static_cast<int>(model_input.GetIntersiteCoulomb().size()); ++distance) {
1016 for (int site = 0; site < model_input.GetSystemSize() - distance; ++site) {
1017 GenerateMatrixComponentsIntersite(edmc, basis, site, nc, site + distance, nc, model_input.GetIntersiteCoulomb(distance - 1));
1018 }
1019 }
1020
1021 //Intersite Hopping
1022 for (int distance = 1; distance <= static_cast<int>(model_input.GetHopping().size()); ++distance) {
1023 for (int site = 0; site < model_input.GetSystemSize() - distance; ++site) {
1024 const auto t = model_input.GetHopping(distance - 1);
1025 int n_electron = 0;
1026 for (int s = site; s < site + distance; ++s) {
1027 n_electron += model_input.CalculateNumElectron(edmc->basis_onsite[s]);
1028 }
1029 GenerateMatrixComponentsIntersite(edmc, basis, site, c_up_dagger , site + distance, c_up , +1.0*t, 2*(n_electron%2) - 1);
1030 GenerateMatrixComponentsIntersite(edmc, basis, site, c_up , site + distance, c_up_dagger , -1.0*t, 2*(n_electron%2) - 1);
1031 GenerateMatrixComponentsIntersite(edmc, basis, site, c_down_dagger, site + distance, c_down , +1.0*t, 2*(n_electron%2) - 1);
1032 GenerateMatrixComponentsIntersite(edmc, basis, site, c_down , site + distance, c_down_dagger, -1.0*t, 2*(n_electron%2) - 1);
1033 }
1034 }
1035
1036 if (model_input.GetBoundaryCondition() == model::BoundaryCondition::PBC) {
1037 //Intersite Coulomb
1038 for (int distance = 1; distance <= static_cast<int>(model_input.GetIntersiteCoulomb().size()); ++distance) {
1039 for (int i = 0; i < distance; ++i) {
1040 const auto d1 = model_input.GetSystemSize() - distance + i;
1041 const auto d2 = i;
1042 GenerateMatrixComponentsIntersite(edmc, basis, d1, nc, d2, nc, model_input.GetIntersiteCoulomb(distance - 1));
1043 }
1044 }
1045
1046 //Intersite Hopping
1047 for (int distance = 1; distance <= static_cast<int>(model_input.GetHopping().size()); ++distance) {
1048 for (int i = 0; i < distance; ++i) {
1049 const auto d1 = model_input.GetSystemSize() - distance + i;
1050 const auto d2 = i;
1051 const auto t = model_input.GetHopping(distance - 1);
