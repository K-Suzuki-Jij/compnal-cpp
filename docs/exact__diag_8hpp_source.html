50 void CalculateGroundState(const std::string &diag_method = "Lanczos") {
51 if (model.GetCalculatedEigenvectorSet().count(0) != 0) {
52 return;
53 }
54 model.GenerateBasis();
55 CRS ham;
56 GenerateHamiltonian(&ham);
57 if (eigenvalues_.size() == 0) {
58 eigenvalues_.emplace_back();
59 }
60 if (eigenvectors_.size() == 0) {
61 eigenvectors_.emplace_back();
62 }
63 if (diag_method == "Lanczos") {
64 sparse_matrix::EigenvalueDecompositionLanczos(&eigenvalues_[0], &eigenvectors_[0], ham, params.lanczos);
65 }
66 else if (diag_method == "LOBPCG") {
67 sparse_matrix::EigenvalueDecompositionLOBPCG(&eigenvalues_[0], &eigenvectors_[0], ham, params.lanczos);
68 }
69 else {
70 std::stringstream ss;
71 ss << "Error in " << __func__ << std::endl;
72 ss << "Invalid diag_method: " << diag_method << std::endl;
73 throw std::runtime_error(ss.str());
74 }
75
76 sparse_matrix::InverseIteration(&ham, &eigenvectors_[0], eigenvalues_[0], params.ii);
77
78 model.SetCalculatedEigenvectorSet(0);
79 }

81 void CalculateTargetState(const int target_sector, const std::string &diag_method = "Lanczos") {
82 if (target_sector < 0) {
83 std::stringstream ss;
84 ss << "Error in " << __func__ << std::endl;
85 ss << "Invalid target_sector: " << target_sector << std::endl;
86 throw std::runtime_error(ss.str());
87 }
88 CalculateGroundState(diag_method);
89 if (model.GetCalculatedEigenvectorSet().count(target_sector) != 0) {
90 return;
91 }
92
93 model.GenerateBasis();
94 CRS ham;
95 GenerateHamiltonian(&ham);
96 if (diag_method == "Lanczos") {
97 for (int sector = 1; sector <= target_sector; ++sector) {
98 if (model.GetCalculatedEigenvectorSet().count(sector) == 0) {
99 if (static_cast<int>(eigenvectors_.size()) != sector) {
100 std::stringstream ss;
101 ss << "Unknown Error in " << __func__ << std::endl;
102 ss << "sector: " << sector << ", size: " << eigenvectors_.size() << std::endl;
103 throw std::runtime_error(ss.str());
104 }
105 BraketVector temp_vector(ham.row_dim);
106 RealType temp_value = 0.0;
107 sparse_matrix::EigenvalueDecompositionLanczos(&temp_value, &temp_vector, ham, sector, eigenvectors_, params.lanczos);
108 sparse_matrix::InverseIteration(&ham, &temp_vector, temp_value, params.ii, eigenvectors_);
109 eigenvalues_.push_back(temp_value);
110 eigenvectors_.push_back(temp_vector);
111 model.SetCalculatedEigenvectorSet(sector);
112 }
113 }
114 }
115 else if (diag_method == "LOBPCG") {
116 std::stringstream ss;
117 ss << "Error in " << __func__ << std::endl;
118 ss << "LOBPCG is under construction: " << target_sector << std::endl;
119 throw std::runtime_error(ss.str());
120 }
121 else {
122 std::stringstream ss;
123 ss << "Error in " << __func__ << std::endl;
124 ss << "Invalid diag_method: " << target_sector << std::endl;
125 throw std::runtime_error(ss.str());
126 }

130 RealType CalculateExpectationValue(const CRS &m, const int site, const int level = 0) const {
131 if (model.GetCalculatedEigenvectorSet().count(level) == 0) {
132 std::stringstream ss;
133 ss << "Error in " << __func__ << std::endl;
134 ss << "An eigenvector of the energy level: " << level << " has not been calculated" << std::endl;
135 throw std::runtime_error(ss.str());
136 }
137
138 const auto &basis = model.GetTargetBasis();
139 const auto &basis_inv = model.GetTargetBasisInv();
140
141 const std::int64_t dim = eigenvectors_.at(level).val.size();
142 if (static_cast<std::int64_t>(basis.size()) != dim || static_cast<std::int64_t>(basis_inv.size()) != dim) {
143 std::stringstream ss;
144 ss << "Error in " << __func__ << std::endl;
145 ss << "Unknown error detected" << std::endl;
146 throw std::runtime_error(ss.str());
147 }
148
149 const int dim_onsite = static_cast<int>(m.row_dim);
150 const std::int64_t site_constant = static_cast<std::int64_t>(std::pow(dim_onsite, site));
151 const BraketVector &eigenvector = eigenvectors_.at(level);
152 RealType val = 0.0;
153
154#pragma omp parallel for reduction (+: val)
155 for (std::int64_t i = 0; i < dim; ++i) {
156 const std::int64_t global_basis = basis[i];
157 const int local_basis = CalculateLocalBasis(global_basis, site, dim_onsite);
158 RealType temp_val = 0.0;
159 for (std::int64_t j = m.row[local_basis]; j < m.row[local_basis + 1]; ++j){
160 const std::int64_t a_basis = global_basis - (local_basis - m.col[j])*site_constant;
161 if (basis_inv.count(a_basis) != 0) {
162 temp_val += eigenvector.val[basis_inv.at(a_basis)]*m.val[j];
163 }
164 }
165 val += eigenvector.val[i]*temp_val;
166 }
167 return val;
168 }
