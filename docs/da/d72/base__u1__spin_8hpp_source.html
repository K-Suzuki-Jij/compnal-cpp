153 std::vector<std::int64_t> GenerateBasis(const int system_size,
154 const QType total_sz,
155 const bool flag_display_info = true) const {
156
157 const auto start = std::chrono::system_clock::now();
158
159 if (!ValidateQNumber(system_size, magnitude_spin_, total_sz)) {
160 std::stringstream ss;
161 ss << "Error at " << __LINE__ << " in " << __func__ << " in "<< __FILE__ << std::endl;
162 ss << "Invalid parameters (system_size or magnitude_spin or total_sz)" << std::endl;
163 throw std::runtime_error(ss.str());
164 }
165
166 const int dim_onsite = 2*magnitude_spin_ + 1;
167
168 if (flag_display_info) {
169 std::cout << "Generating Basis..." << std::flush;
170 }
171
172 auto partition_integers = utility::GenerateIntegerPartition(static_cast<int>(system_size*magnitude_spin_ - total_sz),
173 static_cast<int>(2*magnitude_spin_),
174 system_size);
175
176 for (auto &&it: partition_integers) {
177 it.resize(system_size, 0);
178 }
179
180 std::vector<std::int64_t> site_constant(system_size);
181 for (int site = 0; site < system_size; ++site) {
182 site_constant[site] = static_cast<std::int64_t>(std::pow(dim_onsite, site));
183 }
184
185 std::vector<std::int64_t> basis;
186
187#ifdef _OPENMP
188 const int num_threads = omp_get_max_threads();
189 std::vector<std::vector<std::int64_t>> temp_basis(num_threads);
190 for (const auto &integer_list: partition_integers) {
191 const std::int64_t size = utility::CalculateNumPermutation(integer_list);
192#pragma omp parallel num_threads (num_threads)
193 {
194 const int thread_num = omp_get_thread_num();
195 const std::int64_t loop_begin = thread_num*size/num_threads;
196 const std::int64_t loop_end = (thread_num + 1)*size/num_threads;
197 std::vector<int> n_th_integer_list = utility::GenerateNthPermutation(integer_list, loop_begin);
198 for (std::int64_t j = loop_begin; j < loop_end; ++j) {
199 std::int64_t basis_global = 0;
200 for (std::size_t k = 0; k < n_th_integer_list.size(); ++k) {
201 basis_global += n_th_integer_list[k]*site_constant[k];
202 }
203 temp_basis[thread_num].push_back(basis_global);
204 std::next_permutation(n_th_integer_list.begin(), n_th_integer_list.end());
205 }
206 }
207 }
208 for (auto &&it: temp_basis) {
209 basis.insert(basis.end(), it.begin(), it.end());
210 std::vector<std::int64_t>().swap(it);
211 }
212#else
213 for (std::size_t i = 0; i < partition_integers.size(); ++i) {
214 auto &integer_list = partition_integers[i];
215 std::sort(integer_list.begin(), integer_list.end());
216 do {
217 std::int64_t basis_global = 0;
218 for (std::size_t j = 0; j < integer_list.size(); ++j) {
219 basis_global += integer_list[j]*site_constant[j];
220 }
221 basis.push_back(basis_global);
222 } while (std::next_permutation(integer_list.begin(), integer_list.end()));
223 }
224#endif
225
226 const std::int64_t dim_target = CalculateTargetDim(system_size, magnitude_spin_, total_sz);
227
228 basis.shrink_to_fit();
229
230 if (static_cast<std::int64_t>(basis.size()) != dim_target) {
231 std::stringstream ss;
232 ss << "Unknown error at " << __LINE__ << " in " << __func__ << " in "<< __FILE__ << std::endl;
233 ss << "basis.size()=" << basis.size() << ", but dim_target=" << dim_target << std::endl;
234 throw std::runtime_error(ss.str());
235 }
236
237 std::sort(basis.begin(), basis.end());
238
239 if (flag_display_info) {
240 const auto time_count = std::chrono::duration_cast<std::chrono::microseconds>(std::chrono::system_clock::now() - start).count();
241 const double time_sec = static_cast<double>(time_count)/blas::TIME_UNIT_CONSTANT;
242 std::cout << "\rElapsed time of generating basis:" << time_sec << "[sec]" << std::endl;
243 }
244 return basis;
245 }

272 static bool ValidateQNumber(const int system_size, const HalfInt magnitude_spin, const HalfInt total_sz) {
273 if (system_size <= 0 || magnitude_spin <= 0) {
274 std::stringstream ss;
275 ss << "Error at " << __LINE__ << " in " << __func__ << " in "<< __FILE__ << std::endl;
276 ss << "Invalid parameters (system_size or magnitude_spin)" << std::endl;
277 }
278 const int total_2sz = 2*total_sz;
279 const int magnitude_2spin = 2*magnitude_spin;
280 const bool c1 = ((system_size*magnitude_2spin - total_2sz)%2 == 0);
281 const bool c2 = (-system_size*magnitude_2spin <= total_2sz);
282 const bool c3 = (total_2sz <= system_size*magnitude_2spin);
283 if (c1 && c2 && c3) {
284 return true;
285 }
286 else {
287 return false;
288 }
289 }

297 static std::int64_t CalculateTargetDim(const int system_size, const HalfInt magnitude_spin, const HalfInt total_sz) {
298 if (!ValidateQNumber(system_size, magnitude_spin, total_sz)) {
299 return 0;
300 }
301 if (system_size <= 0) {
302 return 0;
303 }
304 const int magnitude_2spin = 2*magnitude_spin;
305 const int total_2sz = 2*total_sz;
306 const int max_total_2sz = system_size*magnitude_2spin;
307 std::vector<std::vector<std::int64_t>> dim(system_size, std::vector<std::int64_t>(max_total_2sz + 1));
308 for (int s = -magnitude_2spin; s <= magnitude_2spin; s += 2) {
309 dim[0][(s + magnitude_2spin)/2] = 1;
310 }
311 for (int site = 1; site < system_size; site++) {
312 for (int s = -magnitude_2spin; s <= magnitude_2spin; s += 2) {
313 for (int s_prev = -magnitude_2spin*site; s_prev <= magnitude_2spin*site; s_prev += 2) {
314 const std::int64_t a = dim[site ][(s + s_prev + magnitude_2spin*(site + 1))/2];
315 const std::int64_t b = dim[site - 1][(s_prev + magnitude_2spin*site)/2];
316 if (a >= INT64_MAX - b) {
317 throw std::overflow_error("Overflow detected for sumation using uint64_t");
318 }
319 dim[site][(s + s_prev + magnitude_2spin*(site + 1))/2] = a + b;
320 }
321 }
322 }
323 return dim[system_size - 1][(total_2sz + max_total_2sz)/2];
324 }

329 static CRS CreateOnsiteOperatorSx(const HalfInt magnitude_spin) {

344 static CRS CreateOnsiteOperatorSz(const HalfInt magnitude_spin) {
345 const int dim_onsite = 2*magnitude_spin + 1;
346 CRS matrix(dim_onsite, dim_onsite);
347
348 for (int row = 0; row < dim_onsite; ++row) {
349 const RealType val = magnitude_spin - row;
350 if (val != 0.0) {
351 matrix.val.push_back(val);
352 matrix.col.push_back(row);
353 }
354 matrix.row[row + 1] = matrix.col.size();
355 }
356 matrix.tag = type::CRSTag::BOSON;
357 return matrix;
358 }
359
363 static CRS CreateOnsiteOperatorSp(const HalfInt magnitude_spin) {
364 const int dim_onsite = 2*magnitude_spin + 1;
365 const RealType constant = 2.0;
366 CRS matrix(dim_onsite, dim_onsite);
367 for (int row = 1; row < dim_onsite; ++row) {
368 matrix.val.push_back(std::sqrt((magnitude_spin + 1)*constant*row - row*(row + 1)));
369 matrix.col.push_back(row);
370 matrix.row[row] = matrix.col.size();
371 }
372 matrix.row[dim_onsite] = matrix.col.size();
373 matrix.tag = type::CRSTag::BOSON;
374 return matrix;
375 }
376
380 static CRS CreateOnsiteOperatorSm(const HalfInt magnitude_spin) {
381 const int dim_onsite = 2*magnitude_spin + 1;
382 CRS matrix(dim_onsite, dim_onsite);
383 const RealType constant = 2.0;
384 for (int row = 1; row < dim_onsite; ++row) {
385 matrix.val.push_back(std::sqrt((magnitude_spin + 1)*constant*row - row*(row + 1)));
386 matrix.col.push_back(row - 1);
387 matrix.row[row + 1] = matrix.col.size();
388 }
389 matrix.tag = type::CRSTag::BOSON;
390 return matrix;
391 }
