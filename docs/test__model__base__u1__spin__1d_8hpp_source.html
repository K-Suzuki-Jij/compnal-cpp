15template<typename RealType>
16void TestSpinOneHalf(const compnal::model::BaseU1Spin_1D<RealType> &model) {
17
18 EXPECT_EQ(model.GetDimOnsite(), 2);
19 EXPECT_EQ(model.GetMagnitudeSpin(), 0.5);
20
21 const compnal::sparse_matrix::CRS<RealType> ref_sp ({{+0.0, +1.0}, {+0.0, +0.0}});
22 const compnal::sparse_matrix::CRS<RealType> ref_sm ({{+0.0, +0.0}, {+1.0, +0.0}});
23 const compnal::sparse_matrix::CRS<RealType> ref_sx ({{+0.0, +0.5}, {+0.5, +0.0}});
24 const compnal::sparse_matrix::CRS<RealType> ref_isy({{+0.0, +0.5}, {-0.5, +0.0}});
25 const compnal::sparse_matrix::CRS<RealType> ref_sz ({{+0.5, +0.0}, {+0.0, -0.5}});
26
27 EXPECT_TRUE(compnal::test::ExpectNear(model.GetOnsiteOperatorSp() , ref_sp , std::pow(10, -15)));
28 EXPECT_TRUE(compnal::test::ExpectNear(model.GetOnsiteOperatorSm() , ref_sm , std::pow(10, -15)));
29 EXPECT_TRUE(compnal::test::ExpectNear(model.GetOnsiteOperatorSx() , ref_sx , std::pow(10, -15)));
30 EXPECT_TRUE(compnal::test::ExpectNear(model.GetOnsiteOperatoriSy(), ref_isy, std::pow(10, -15)));
31 EXPECT_TRUE(compnal::test::ExpectNear(model.GetOnsiteOperatorSz() , ref_sz , std::pow(10, -15)));
32}

57
58 const compnal::sparse_matrix::CRS<RealType> ref_isy ({
59 {+0.0, +1.0/std::sqrt(2), +0.0},
60 {-1.0/std::sqrt(2), +0.0, +1.0/std::sqrt(2)},
61 {+0.0, -1.0/std::sqrt(2), +0.0}
62 });
63
64 const compnal::sparse_matrix::CRS<RealType> ref_sz ({
65 {+1.0, +0.0, +0.0},
66 {+0.0, +0.0, +0.0},
67 {+0.0, +0.0, -1.0}
68 });
69
70 EXPECT_TRUE(compnal::test::ExpectNear(model.GetOnsiteOperatorSp() , ref_sp , std::pow(10, -15)));
71 EXPECT_TRUE(compnal::test::ExpectNear(model.GetOnsiteOperatorSm() , ref_sm , std::pow(10, -15)));
