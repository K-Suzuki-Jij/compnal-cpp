25class HalfInt {
26
27public:
28 HalfInt() {};
29
30 template<typename ValueType>
31 HalfInt(ValueType value) {
32 value = 2*value;
33 if (std::floor(value) != value) {
34 throw std::runtime_error("The input number is not half-integer");
35 }
36 integer_ = static_cast<int>(value);
37 };
38
39 int GetInteger() const {
40 return integer_;
41 }
42
43 HalfInt operator-() const {
44 return HalfInt(-0.5*this->integer_);
45 }

47 HalfInt operator+() const {
48 return *this;
49 }
50
51 operator int () const noexcept { return static_cast<int>(0.5*integer_); }
52 operator double() const noexcept { return 0.5*integer_; }

54private:
55 int integer_ = 0;

57};
