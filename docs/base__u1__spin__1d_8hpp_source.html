187 void GenerateBasis(const HalfInt total_sz, const bool flag_display_info = false) {
188 if (!isValidQNumber(total_sz)) {
189 std::stringstream ss;
190 ss << "Error in " << __FUNCTION__ << " at " << __LINE__ << std::endl;
191 ss << "Invalid parameters (system_size or magnitude_spin or total_sz)" << std::endl;
192 throw std::runtime_error(ss.str());
193 }
194
195 const auto start = std::chrono::system_clock::now();
196
197 if (bases_.count(total_sz) != 0) {
198 return;
199 }
200
201 if (flag_display_info) {
202 std::cout << "Generating Basis..." << std::flush;
203 }
204
205 const int shifted_2sz = static_cast<int>(2*(system_size_*magnitude_spin_ - total_sz));
206 const std::int64_t dim_target = CalculateTargetDim(total_sz);
207 std::vector<std::vector<int>> partition_integers;
208 utility::GenerateIntegerPartition(&partition_integers, shifted_2sz, magnitude_spin_);
209
210 std::vector<std::int64_t> site_constant(system_size_);
211 for (int site = 0; site < system_size_; ++site) {
212 site_constant[site] = static_cast<std::int64_t>(std::pow(dim_onsite_, site));
213 }
214
215 std::vector<std::int64_t>().swap(bases_[total_sz]);
216
217 auto &basis_ref = bases_.at(total_sz);
218
219#ifdef _OPENMP
220 const int num_threads = omp_get_max_threads();
221 std::vector<std::vector<std::int64_t>> temp_basis(num_threads);
222 for (auto &&integer_list: partition_integers) {
223 const bool condition1 = (0 < integer_list.size()) && (static_cast<int>(integer_list.size()) <= system_size_);
224 const bool condition2 = (integer_list.size() == 0) && (shifted_2sz == 0);
225 if (condition1 || condition2) {
226 for (int j = static_cast<int>(integer_list.size()); j < system_size_; ++j) {
227 integer_list.push_back(0);
228 }
229
230 const std::int64_t size = utility::CalculateNumCombination(integer_list);
231 std::vector<std::vector<int>> temp_partition_integer(num_threads);
232
233#pragma omp parallel num_threads (num_threads)
234 {
235 const int thread_num = omp_get_thread_num();
236 const std::int64_t loop_begin = thread_num*size/num_threads;
237 const std::int64_t loop_end = (thread_num + 1)*size/num_threads;
238 temp_partition_integer[thread_num] = integer_list;
239 utility::CalculateNthPermutation(&temp_partition_integer[thread_num], loop_begin);
240
241 for (std::int64_t j = loop_begin; j < loop_end; ++j) {
242 std::int64_t basis_global = 0;
243 const auto iter_begin = temp_partition_integer[thread_num].begin();
244 const auto iter_end = temp_partition_integer[thread_num].end();
245 for (auto itr = iter_begin; itr != iter_end; ++itr) {
246 basis_global += *itr*site_constant[std::distance(iter_begin, itr)];
247 }
248 temp_basis[thread_num].push_back(basis_global);
249 std::next_permutation(temp_partition_integer[thread_num].begin(), temp_partition_integer[thread_num].end());
250 }
251 }
252 }
253 }
254
255 for (auto &&basis: temp_basis) {
256 basis_ref.insert(basis_ref.end(), basis.begin(), basis.end());
257 std::vector<std::int64_t>().swap(basis);
258 }
259
260#else
261 basis_ref.reserve(dim_target);
262
263 for (auto &&integer_list: partition_integers) {
264 const bool condition1 = (0 < integer_list.size()) && (static_cast<int>(integer_list.size()) <= system_size_);
265 const bool condition2 = (integer_list.size() == 0) && (shifted_2sz == 0);
266 if (condition1 || condition2) {
267
268 for (std::int64_t j = integer_list.size(); j < system_size_; ++j) {
269 integer_list.push_back(0);
270 }
271
272 std::sort(integer_list.begin(), integer_list.end());
273
274 do {
275 std::int64_t basis_global = 0;
276 for (std::size_t j = 0; j < integer_list.size(); ++j) {
277 basis_global += integer_list[j]*site_constant[j];
278 }
279 basis_ref.push_back(basis_global);
280 } while (std::next_permutation(integer_list.begin(), integer_list.end()));
281 }
282 }
283
284#endif
285
286 if (static_cast<std::int64_t>(basis_ref.size()) != dim_target) {
287 std::stringstream ss;
288 ss << "Unknown error detected in " << __FUNCTION__ << " at " << __LINE__ << std::endl;
289 throw std::runtime_error(ss.str());
290 }
291
292 std::sort(basis_ref.begin(), basis_ref.end());
293
294 bases_inv_[total_sz].clear();
295
296 auto &basis_inv_ref = bases_inv_.at(total_sz);
297
298 for (std::int64_t i = 0; i < dim_target; ++i) {
299 basis_inv_ref[basis_ref[i]] = i;
300 }
301
302 if (flag_display_info) {
303 const auto time_count = std::chrono::duration_cast<std::chrono::microseconds>(std::chrono::system_clock::now() - start).count();
304 const double time_sec = static_cast<double>(time_count)/sparse_matrix::TIME_UNIT_CONSTANT;
305 std::cout << "\rElapsed time of generating basis:" << time_sec << "[sec]" << std::endl;
306 }
307
308 }

314 std::vector<QType> GenerateTargetSector(const CRS &m_1, const CRS &m_2) const {
315
316 std::unordered_set<QType> delta_sector_set_m1;
317 std::unordered_set<QType> delta_sector_set_m2;
318 for (std::int64_t i = 0; i < m_1.row_dim; ++i) {
319 for (std::int64_t j = m_1.row[i]; j < m_1.row[i + 1]; ++j) {
320 if (m_1.val[j] != 0.0) {
321 delta_sector_set_m1.emplace(CalculateQuntumNumberDifference(static_cast<int>(i), static_cast<int>(m_1.col[j])));
322 }
323 }
324 }
325 for (std::int64_t i = 0; i < m_2.row_dim; ++i) {
326 for (std::int64_t j = m_2.row[i]; j < m_2.row[i + 1]; ++j) {
327 if (m_2.val[j] != 0.0) {
328 delta_sector_set_m2.emplace(CalculateQuntumNumberDifference(static_cast<int>(i), static_cast<int>(m_2.col[j])));
329 }
330 }
331 }
332 std::vector<QType> target_sector_set;
333 for (const auto &del_sec_m1: delta_sector_set_m1) {
334 for (const auto &del_sec_m2: delta_sector_set_m2) {
335 const bool c1 = isValidQNumber(del_sec_m1 + total_sz_);
336 if (del_sec_m1 == del_sec_m2 && c1) {
337 target_sector_set.push_back(del_sec_m1 + total_sz_);
338 }
339 }
340 }
341 std::sort(target_sector_set.begin(), target_sector_set.end());
342 target_sector_set.erase(std::unique(target_sector_set.begin(), target_sector_set.end()), target_sector_set.end());
343 return target_sector_set;
344 }

470 static bool isValidQNumber(const int system_size, const HalfInt magnitude_spin, const HalfInt total_sz) {
471 const int total_2sz = 2*total_sz;
472 const int magnitude_2spin = 2*magnitude_spin;
473 const bool c1 = ((system_size*magnitude_2spin - total_2sz)%2 == 0);
474 const bool c2 = (-system_size*magnitude_2spin <= total_2sz);
475 const bool c3 = (total_2sz <= system_size*magnitude_2spin);
476 if (c1 && c2 && c3) {
477 return true;
478 }
479 else {
480 return false;
481 }
482 }

489 static std::int64_t CalculateTargetDim(const int system_size, const HalfInt magnitude_spin, const HalfInt total_sz) {
490 if (!isValidQNumber(system_size, magnitude_spin, total_sz)) {
491 return 0;
492 }
493 if (system_size <= 0) {
494 return 0;
495 }
496 const int magnitude_2spin = 2*magnitude_spin;
497 const int total_2sz = 2*total_sz;
498 const int max_total_2sz = system_size*magnitude_2spin;
499 std::vector<std::vector<std::int64_t>> dim(system_size, std::vector<std::int64_t>(max_total_2sz + 1));
500 for (int s = -magnitude_2spin; s <= magnitude_2spin; s += 2) {
501 dim[0][(s + magnitude_2spin)/2] = 1;
502 }
503 for (int site = 1; site < system_size; site++) {
504 for (int s = -magnitude_2spin; s <= magnitude_2spin; s += 2) {
505 for (int s_prev = -magnitude_2spin*site; s_prev <= magnitude_2spin*site; s_prev += 2) {
506 const std::int64_t a = dim[site ][(s + s_prev + magnitude_2spin*(site + 1))/2];
507 const std::int64_t b = dim[site - 1][(s_prev + magnitude_2spin*site)/2];
508 if (a >= INT64_MAX - b) {
509 throw std::runtime_error("Overflow detected for sumation using uint64_t");
510 }
511 dim[site][(s + s_prev + magnitude_2spin*(site + 1))/2] = a + b;
512 }
513 }
514 }
515 return dim[system_size - 1][(total_2sz + max_total_2sz)/2];
516 }

521 static CRS CreateOnsiteOperatorSx(const HalfInt magnitude_spin) {
522 return 0.5*(CreateOnsiteOperatorSp(magnitude_spin) + CreateOnsiteOperatorSm(magnitude_spin));
523 }
524
528 static CRS CreateOnsiteOperatoriSy(const HalfInt magnitude_spin) {
529 return 0.5*(CreateOnsiteOperatorSp(magnitude_spin) - CreateOnsiteOperatorSm(magnitude_spin));
530 }
531
535 static CRS CreateOnsiteOperatorSz(const HalfInt magnitude_spin) {
536 const int dim_onsite = 2*magnitude_spin + 1;
537 CRS matrix(dim_onsite, dim_onsite);
538
539 for (int row = 0; row < dim_onsite; ++row) {
540 const RealType val = magnitude_spin - row;
541 if (val != 0.0) {
542 matrix.val.push_back(val);
543 matrix.col.push_back(row);
544 }
545 matrix.row[row + 1] = matrix.col.size();
546 }
547 return matrix;
548 }
549
553 static CRS CreateOnsiteOperatorSp(const HalfInt magnitude_spin) {
554 const int dim_onsite = 2*magnitude_spin + 1;
555 CRS matrix(dim_onsite, dim_onsite);
556 for (int row = 1; row < dim_onsite; ++row) {
557 matrix.val.push_back(std::sqrt((magnitude_spin + 1)*2.0*row - row*(row + 1)));
558 matrix.col.push_back(row);
559 matrix.row[row] = matrix.col.size();
560 }
561 matrix.row[dim_onsite] = matrix.col.size();
562 return matrix;
563 }
564
568 static CRS CreateOnsiteOperatorSm(const HalfInt magnitude_spin) {
569 const int dim_onsite = 2*magnitude_spin + 1;
570 CRS matrix(dim_onsite, dim_onsite);
571 for (int row = 1; row < dim_onsite; ++row) {
572 matrix.val.push_back(std::sqrt((magnitude_spin + 1)*2.0*row - row*(row + 1)));
573 matrix.col.push_back(row - 1);
574 matrix.row[row + 1] = matrix.col.size();
575 }
576 return matrix;
577 }
