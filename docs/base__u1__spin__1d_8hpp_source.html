174 void GenerateBasis(const double total_sz) {
175 if (!isValidQNumber(total_sz)) {
176 std::stringstream ss;
177 ss << "Error in " << __FUNCTION__ << std::endl;
178 ss << "Invalid parameters (system_size or magnitude_spin or total_sz)" << std::endl;
179 throw std::runtime_error(ss.str());
180 }
181
182 const auto start = std::chrono::system_clock::now();
183 const int total_2sz = utility::DoubleTheNumber(total_sz);
184
185 if (bases_.count(total_2sz) != 0) {
186 return;
187 }
188
189 std::cout << "Generating Basis..." << std::flush;
190
191 const int shifted_2sz = (system_size_*magnitude_2spin_ - total_2sz)/2;
192 const std::int64_t dim_target = CalculateTargetDim(total_sz);
193 std::vector<std::vector<int>> partition_integers;
194 utility::GenerateIntegerPartition(&partition_integers, shifted_2sz, magnitude_2spin_);
195
196 std::vector<std::int64_t> site_constant(system_size_);
197 for (int site = 0; site < system_size_; ++site) {
198 site_constant[site] = static_cast<std::int64_t>(std::pow(dim_onsite_, site));
199 }
200
201 if (bases_.count(total_2sz) == 0) {
202 bases_[total_2sz] = std::vector<std::int64_t>();
203 }
204
205 std::vector<std::int64_t>().swap(bases_.at(total_2sz));
206
207#ifdef _OPENMP
208 const int num_threads = omp_get_max_threads();
209 std::vector<std::vector<std::int64_t>> temp_basis(num_threads);
210 for (auto &&integer_list: partition_integers) {
211 const bool condition1 = (0 < integer_list.size()) && (static_cast<int>(integer_list.size()) <= system_size_);
212 const bool condition2 = (integer_list.size() == 0) && (shifted_2sz == 0);
213 if (condition1 || condition2) {
214 for (int j = static_cast<int>(integer_list.size()); j < system_size_; ++j) {
215 integer_list.push_back(0);
216 }
217
218 const std::int64_t size = utility::CalculateNumCombination(integer_list);
219 std::vector<std::vector<int>> temp_partition_integer(num_threads);
220
221#pragma omp parallel num_threads (num_threads)
222 {
223 const int thread_num = omp_get_thread_num();
224 const std::int64_t loop_begin = thread_num*size/num_threads;
225 const std::int64_t loop_end = (thread_num + 1)*size/num_threads;
226 temp_partition_integer[thread_num] = integer_list;
227 utility::CalculateNthPermutation(&temp_partition_integer[thread_num], loop_begin);
228
229 for (std::int64_t j = loop_begin; j < loop_end; ++j) {
230 std::int64_t basis_global = 0;
231 const auto iter_begin = temp_partition_integer[thread_num].begin();
232 const auto iter_end = temp_partition_integer[thread_num].end();
233 for (auto itr = iter_begin; itr != iter_end; ++itr) {
234 basis_global += *itr*site_constant[std::distance(iter_begin, itr)];
235 }
236 temp_basis[thread_num].push_back(basis_global);
237 std::next_permutation(temp_partition_integer[thread_num].begin(), temp_partition_integer[thread_num].end());
238 }
239 }
240 }
241 }
242 for (auto &&basis: temp_basis) {
243 bases_.at(total_2sz).insert(bases_.at(total_2sz).end(), basis.begin(), basis.end());
244 std::vector<std::int64_t>().swap(basis);
245 }
246
247#else
248 bases_.at(total_2sz).reserve(dim_target);
249
250 for (auto &&integer_list: partition_integers) {
251 const bool condition1 = (0 < integer_list.size()) && (static_cast<int>(integer_list.size()) <= system_size_);
252 const bool condition2 = (integer_list.size() == 0) && (shifted_2sz == 0);
253 if (condition1 || condition2) {
254
255 for (std::int64_t j = integer_list.size(); j < system_size_; ++j) {
256 integer_list.push_back(0);
257 }
258
259 std::sort(integer_list.begin(), integer_list.end());
260
261 do {
262 std::int64_t basis_global = 0;
263 for (std::size_t j = 0; j < integer_list.size(); ++j) {
264 basis_global += integer_list[j]*site_constant[j];
265 }
266 bases_.at(total_2sz).push_back(basis_global);
267 } while (std::next_permutation(integer_list.begin(), integer_list.end()));
268 }
269 }
270
271#endif
272
273 if (static_cast<std::int64_t>(bases_.at(total_2sz).size()) != dim_target) {
274 std::stringstream ss;
275 ss << "Unknown error detected in " << __FUNCTION__ << std::endl;
276 throw std::runtime_error(ss.str());
277 }
278
279 std::sort(bases_.at(total_2sz).begin(), bases_.at(total_2sz).end());
280
281 if (bases_inv_.count(total_2sz) == 0) {
282 bases_inv_[total_2sz] = std::unordered_map<std::int64_t, std::int64_t>();
283 }
284 bases_inv_.at(total_2sz).clear();
285
286 for (std::int64_t i = 0; i < dim_target; ++i) {
287 bases_inv_.at(total_2sz)[bases_.at(total_2sz)[i]] = i;
288 }
289
290 const auto time_count = std::chrono::duration_cast<std::chrono::microseconds>(std::chrono::system_clock::now() - start).count();
291 const double time_sec = static_cast<double>(time_count)/sparse_matrix::TIME_UNIT_CONSTANT;
292 std::cout << "\rElapsed time of generating basis:" << time_sec << "[sec]" << std::endl;
293 }

299 std::vector<double> GenerateTargetSector(const CRS &m_1, const CRS &m_2) const {
300
301 std::unordered_set<double> delta_sector_set_m1;
302 std::unordered_set<double> delta_sector_set_m2;
303 for (std::int64_t i = 0; i < m_1.row_dim; ++i) {
304 for (std::int64_t j = m_1.row[i]; j < m_1.row[i + 1]; ++j) {
305 if (m_1.val[j] != 0.0) {
306 delta_sector_set_m1.emplace(CalculateQuntumNumberDifference(static_cast<int>(i), static_cast<int>(m_1.col[j])));
307 }
308 }
309 }
310 for (std::int64_t i = 0; i < m_2.row_dim; ++i) {
311 for (std::int64_t j = m_2.row[i]; j < m_2.row[i + 1]; ++j) {
312 if (m_2.val[j] != 0.0) {
313 delta_sector_set_m2.emplace(CalculateQuntumNumberDifference(static_cast<int>(i), static_cast<int>(m_2.col[j])));
314 }
315 }
316 }
317 std::vector<double> target_sector_set;
318 for (const auto &del_sec_m1: delta_sector_set_m1) {
319 for (const auto &del_sec_m2: delta_sector_set_m2) {
320 const bool c1 = isValidQNumber(del_sec_m1 + 0.5*total_2sz_);
321 if (del_sec_m1 == del_sec_m2 && c1) {
322 target_sector_set.push_back(del_sec_m1 + 0.5*total_2sz_);
323 }
324 }
325 }
326 std::sort(target_sector_set.begin(), target_sector_set.end());
327 target_sector_set.erase(std::unique(target_sector_set.begin(), target_sector_set.end()), target_sector_set.end());
328 return target_sector_set;

455 static bool isValidQNumber(const int system_size, const double magnitude_spin, const double total_sz) {
456 const int total_2sz = utility::DoubleTheNumber(total_sz);
457 const int magnitude_2spin = utility::DoubleTheNumber(magnitude_spin);
458 const bool c1 = ((system_size*magnitude_2spin - total_2sz)%2 == 0);
459 const bool c2 = (-system_size*magnitude_2spin <= total_2sz);

461 if (c1 && c2 && c3) {

474 static std::int64_t CalculateTargetDim(const int system_size, const double magnitude_spin, const double total_sz) {
475 const int magnitude_2spin = utility::DoubleTheNumber(magnitude_spin);
476 if (!isValidQNumber(system_size, magnitude_spin, total_sz)) {
477 return 0;
478 }
479 if (system_size <= 0) {
480 return 0;
481 }
482 const int total_2sz = utility::DoubleTheNumber(total_sz);
483 const int max_total_2sz = system_size*magnitude_2spin;
484 std::vector<std::vector<std::int64_t>> dim(system_size, std::vector<std::int64_t>(max_total_2sz + 1));
485 for (int s = -magnitude_2spin; s <= magnitude_2spin; s += 2) {
486 dim[0][(s + magnitude_2spin)/2] = 1;
487 }
488 for (int site = 1; site < system_size; site++) {
489 for (int s = -magnitude_2spin; s <= magnitude_2spin; s += 2) {
490 for (int s_prev = -magnitude_2spin*site; s_prev <= magnitude_2spin*site; s_prev += 2) {
491 const std::int64_t a = dim[site ][(s + s_prev + magnitude_2spin*(site + 1))/2];
492 const std::int64_t b = dim[site - 1][(s_prev + magnitude_2spin*site)/2];
493 if (a >= INT64_MAX - b) {
494 throw std::runtime_error("Overflow detected for sumation using uint64_t");
495 }
496 dim[site][(s + s_prev + magnitude_2spin*(site + 1))/2] = a + b;
497 }
498 }
499 }
500 return dim[system_size - 1][(total_2sz + max_total_2sz)/2];
501 }

506 static CRS CreateOnsiteOperatorSx(const double magnitude_spin) {
507 const int magnitude_2spin = utility::DoubleTheNumber(magnitude_spin);
508 const int dim_onsite = magnitude_2spin + 1;
509 CRS matrix(dim_onsite, dim_onsite);
510 int a = 0;
511 int b = 1;
512
513 matrix.val.push_back(0.5*std::sqrt((magnitude_spin + 1)*(a + b + 1) - (a + 1)*(b + 1)) );
514 matrix.col.push_back(b);
515 matrix.row[1] = matrix.col.size();
516
517 for (int row = 1; row < dim_onsite - 1; ++row) {
518 a = row;
519 b = row - 1;
520 matrix.val.push_back(0.5*std::sqrt((magnitude_spin + 1)*(a + b + 1) - (a + 1)*(b + 1)) );
521 matrix.col.push_back(b);
522
523 a = row;
524 b = row + 1;
525 matrix.val.push_back(0.5*std::sqrt((magnitude_spin + 1)*(a + b + 1) - (a + 1)*(b + 1)) );
526 matrix.col.push_back(b);
527 matrix.row[row + 1] = matrix.col.size();
528 }
529
530 a = dim_onsite - 1;
531 b = dim_onsite - 2;
532
533 matrix.val.push_back(0.5*std::sqrt((magnitude_spin + 1)*(a + b + 1) - (a + 1)*(b + 1)) );
534 matrix.col.push_back(b);
535 matrix.row[dim_onsite] = matrix.col.size();
536
537 return matrix;
538 }
