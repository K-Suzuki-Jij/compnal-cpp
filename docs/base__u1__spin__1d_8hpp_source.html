188 void GenerateBasis(const HalfInt total_sz) {
189 if (!isValidQNumber(total_sz)) {
190 std::stringstream ss;
191 ss << "Error in " << __FUNCTION__ << std::endl;
192 ss << "Invalid parameters (system_size or magnitude_spin or total_sz)" << std::endl;
193 throw std::runtime_error(ss.str());
194 }
195
196 const auto start = std::chrono::system_clock::now();
197 const int total_2sz = utility::DoubleTheNumber(total_sz);
198
199 if (bases_.count(total_2sz) != 0) {
200 return;
201 }
202
203 std::cout << "Generating Basis..." << std::flush;
204
205 const int shifted_2sz = (system_size_*magnitude_2spin_ - total_2sz)/2;
206 const LInt dim_target = CalculateTargetDim(total_sz);
207 std::vector<std::vector<int>> partition_integers;
208 utility::GenerateIntegerPartition(&partition_integers, shifted_2sz, magnitude_2spin_);
209
210 std::vector<LInt> site_constant(system_size_);
211 for (int site = 0; site < system_size_; ++site) {
212 site_constant[site] = static_cast<LInt>(std::pow(dim_onsite_, site));
213 }
214
215 if (bases_.count(total_2sz) == 0) {
216 bases_[total_2sz] = std::vector<LInt>();
217 }
218
219 std::vector<LInt>().swap(bases_.at(total_2sz));
220
221#ifdef _OPENMP
222 const int num_threads = omp_get_max_threads();
223 std::vector<std::vector<LInt>> temp_basis(num_threads);
224 for (auto &&integer_list: partition_integers) {
225 const bool condition1 = (0 < integer_list.size()) && (static_cast<int>(integer_list.size()) <= system_size_);
226 const bool condition2 = (integer_list.size() == 0) && (shifted_2sz == 0);
227 if (condition1 || condition2) {
228 for (int j = static_cast<int>(integer_list.size()); j < system_size_; ++j) {
229 integer_list.push_back(0);
230 }
231
232 const LInt size = utility::CalculateNumCombination(integer_list);
233 std::vector<std::vector<int>> temp_partition_integer(num_threads);
234
235#pragma omp parallel num_threads (num_threads)
236 {
237 const int thread_num = omp_get_thread_num();
238 const LInt loop_begin = thread_num*size/num_threads;
239 const LInt loop_end = (thread_num + 1)*size/num_threads;
240 temp_partition_integer[thread_num] = integer_list;
241 utility::CalculateNthPermutation(&temp_partition_integer[thread_num], loop_begin);
242
243 for (LInt j = loop_begin; j < loop_end; ++j) {
244 LInt basis_global = 0;
245 const auto iter_begin = temp_partition_integer[thread_num].begin();
246 const auto iter_end = temp_partition_integer[thread_num].end();
247 for (auto itr = iter_begin; itr != iter_end; ++itr) {
248 basis_global += *itr*site_constant[std::distance(iter_begin, itr)];
249 }
250 temp_basis[thread_num].push_back(basis_global);
251 std::next_permutation(temp_partition_integer[thread_num].begin(), temp_partition_integer[thread_num].end());
252 }
253 }
254 }
255 }
256 for (auto &&basis: temp_basis) {
257 bases_.at(total_2sz).insert(bases_.at(total_2sz).end(), basis.begin(), basis.end());
258 std::vector<LInt>().swap(basis);
259 }
260
261#else
262 bases_.at(total_2sz).reserve(dim_target);
263
264 for (auto &&integer_list: partition_integers) {
265 const bool condition1 = (0 < integer_list.size()) && (static_cast<int>(integer_list.size()) <= system_size_);
266 const bool condition2 = (integer_list.size() == 0) && (shifted_2sz == 0);
267 if (condition1 || condition2) {
268
269 for (LInt j = integer_list.size(); j < system_size_; ++j) {
270 integer_list.push_back(0);
271 }
272
273 std::sort(integer_list.begin(), integer_list.end());
274
275 do {
276 LInt basis_global = 0;
277 for (std::size_t j = 0; j < integer_list.size(); ++j) {
278 basis_global += integer_list[j]*site_constant[j];
279 }
280 bases_.at(total_2sz).push_back(basis_global);
281 } while (std::next_permutation(integer_list.begin(), integer_list.end()));
282 }
283 }
284
285#endif
286
287 if (static_cast<LInt>(bases_.at(total_2sz).size()) != dim_target) {
288 std::stringstream ss;
289 ss << "Unknown error detected in " << __FUNCTION__ << std::endl;
290 throw std::runtime_error(ss.str());
291 }
292
293 std::sort(bases_.at(total_2sz).begin(), bases_.at(total_2sz).end());
294
295 if (bases_inv_.count(total_2sz) == 0) {
296 bases_inv_[total_2sz] = Map<LInt, LInt>();
297 }
298 bases_inv_.at(total_2sz).clear();
299
300 for (LInt i = 0; i < dim_target; ++i) {
301 bases_inv_.at(total_2sz)[bases_.at(total_2sz)[i]] = i;
302 }
303
304 const auto time_count = std::chrono::duration_cast<std::chrono::microseconds>(std::chrono::system_clock::now() - start).count();
305 const double time_sec = static_cast<double>(time_count)/sparse_matrix::TIME_UNIT_CONSTANT;
306 std::cout << "\rElapsed time of generating basis:" << time_sec << "[sec]" << std::endl;
307 }

313 std::vector<QType> GenerateTargetSector(const CRS &m_1, const CRS &m_2) const {
314
315 std::unordered_set<QType> delta_sector_set_m1;
316 std::unordered_set<QType> delta_sector_set_m2;
317 for (LInt i = 0; i < m_1.row_dim; ++i) {
318 for (LInt j = m_1.row[i]; j < m_1.row[i + 1]; ++j) {
319 if (m_1.val[j] != 0.0) {
320 delta_sector_set_m1.emplace(CalculateQuntumNumberDifference(static_cast<int>(i), static_cast<int>(m_1.col[j])));
321 }
322 }
323 }
324 for (LInt i = 0; i < m_2.row_dim; ++i) {
325 for (LInt j = m_2.row[i]; j < m_2.row[i + 1]; ++j) {
326 if (m_2.val[j] != 0.0) {
327 delta_sector_set_m2.emplace(CalculateQuntumNumberDifference(static_cast<int>(i), static_cast<int>(m_2.col[j])));
328 }
329 }
330 }
331 std::vector<QType> target_sector_set;
332 for (const auto &del_sec_m1: delta_sector_set_m1) {
333 for (const auto &del_sec_m2: delta_sector_set_m2) {
334 const bool c1 = isValidQNumber(del_sec_m1 + 0.5*total_2sz_);
335 if (del_sec_m1 == del_sec_m2 && c1) {
336 target_sector_set.push_back(del_sec_m1 + 0.5*total_2sz_);
337 }
338 }
339 }
340 std::sort(target_sector_set.begin(), target_sector_set.end());
341 target_sector_set.erase(std::unique(target_sector_set.begin(), target_sector_set.end()), target_sector_set.end());
342 return target_sector_set;
343 }

469 static bool isValidQNumber(const int system_size, const HalfInt magnitude_spin, const HalfInt total_sz) {
470 const int total_2sz = utility::DoubleTheNumber(total_sz);
471 const int magnitude_2spin = utility::DoubleTheNumber(magnitude_spin);
472 const bool c1 = ((system_size*magnitude_2spin - total_2sz)%2 == 0);
473 const bool c2 = (-system_size*magnitude_2spin <= total_2sz);
474 const bool c3 = (total_2sz <= system_size*magnitude_2spin);
475 if (c1 && c2 && c3) {
476 return true;
477 }
478 else {
479 return false;
480 }
481 }

488 static LInt CalculateTargetDim(const int system_size, const HalfInt magnitude_spin, const HalfInt total_sz) {
489 const int magnitude_2spin = utility::DoubleTheNumber(magnitude_spin);
490 if (!isValidQNumber(system_size, magnitude_spin, total_sz)) {
491 return 0;
492 }
493 if (system_size <= 0) {
494 return 0;
495 }
496 const int total_2sz = utility::DoubleTheNumber(total_sz);
497 const int max_total_2sz = system_size*magnitude_2spin;
498 std::vector<std::vector<LInt>> dim(system_size, std::vector<LInt>(max_total_2sz + 1));
499 for (int s = -magnitude_2spin; s <= magnitude_2spin; s += 2) {
500 dim[0][(s + magnitude_2spin)/2] = 1;
501 }
502 for (int site = 1; site < system_size; site++) {
503 for (int s = -magnitude_2spin; s <= magnitude_2spin; s += 2) {
504 for (int s_prev = -magnitude_2spin*site; s_prev <= magnitude_2spin*site; s_prev += 2) {
505 const LInt a = dim[site ][(s + s_prev + magnitude_2spin*(site + 1))/2];
506 const LInt b = dim[site - 1][(s_prev + magnitude_2spin*site)/2];
507 if (a >= INT64_MAX - b) {
508 throw std::runtime_error("Overflow detected for sumation using uint64_t");
509 }
510 dim[site][(s + s_prev + magnitude_2spin*(site + 1))/2] = a + b;
511 }
512 }
513 }
514 return dim[system_size - 1][(total_2sz + max_total_2sz)/2];
515 }

520 static CRS CreateOnsiteOperatorSx(const HalfInt magnitude_spin) {
521 const int magnitude_2spin = utility::DoubleTheNumber(magnitude_spin);
522 const int dim_onsite = magnitude_2spin + 1;
523 CRS matrix(dim_onsite, dim_onsite);
524 int a = 0;
525 int b = 1;
526
527 matrix.val.push_back(0.5*std::sqrt((magnitude_spin + 1)*(a + b + 1) - (a + 1)*(b + 1)) );
528 matrix.col.push_back(b);
529 matrix.row[1] = matrix.col.size();
530
531 for (int row = 1; row < dim_onsite - 1; ++row) {
532 a = row;
533 b = row - 1;
534 matrix.val.push_back(0.5*std::sqrt((magnitude_spin + 1)*(a + b + 1) - (a + 1)*(b + 1)) );
535 matrix.col.push_back(b);
536
537 a = row;
538 b = row + 1;
539 matrix.val.push_back(0.5*std::sqrt((magnitude_spin + 1)*(a + b + 1) - (a + 1)*(b + 1)) );
540 matrix.col.push_back(b);
541 matrix.row[row + 1] = matrix.col.size();
542 }
543
544 a = dim_onsite - 1;
545 b = dim_onsite - 2;
546
547 matrix.val.push_back(0.5*std::sqrt((magnitude_spin + 1)*(a + b + 1) - (a + 1)*(b + 1)) );
548 matrix.col.push_back(b);
549 matrix.row[dim_onsite] = matrix.col.size();
550
551 return matrix;
552 }

595 static CRS CreateOnsiteOperatorSz(const HalfInt magnitude_spin) {
596 const int magnitude_2spin = utility::DoubleTheNumber(magnitude_spin);
597 const int dim_onsite = magnitude_2spin + 1;
598 CRS matrix(dim_onsite, dim_onsite);
599
600 for (int row = 0; row < dim_onsite; ++row) {
601 const RealType val = magnitude_spin - row;
602 if (val != 0.0) {
603 matrix.val.push_back(val);
604 matrix.col.push_back(row);
605 }
606 matrix.row[row + 1] = matrix.col.size();
607 }

614 static CRS CreateOnsiteOperatorSp(const HalfInt magnitude_spin) {
615 const int magnitude_2spin = utility::DoubleTheNumber(magnitude_spin);

617 CRS matrix(dim_onsite, dim_onsite);
