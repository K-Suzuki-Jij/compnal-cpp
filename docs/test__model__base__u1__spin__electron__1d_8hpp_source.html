38void TestSpinOneHalf(const BaseU1SpinElectron_1D<RealType> &model) {
39
40 const double threshold = std::pow(10, -15);
41
42 EXPECT_EQ(model.GetDimOnsite(), 4*2);
43 EXPECT_EQ(model.GetMagnitudeLSpin(), 0.5);
44 EXPECT_EQ(model.GetDimOnsiteELectron(), 4);
45 EXPECT_EQ(model.GetDimOnsiteLSpin(), 2);
46
47 const CRS<RealType> ref_c_up ({
48 {0, 0, 1, 0, 0, 0, 0, 0},
49 {0, 0, 0, 1, 0, 0, 0, 0},
50 {0, 0, 0, 0, 0, 0, 0, 0},
51 {0, 0, 0, 0, 0, 0, 0, 0},
52 {0, 0, 0, 0, 0, 0, 1, 0},
53 {0, 0, 0, 0, 0, 0, 0, 1},
54 {0, 0, 0, 0, 0, 0, 0, 0},
55 {0, 0, 0, 0, 0, 0, 0, 0}
56 }, CRSTag::FERMION);
57
58 const CRS<RealType> ref_c_up_d ({
59 {0, 0, 0, 0, 0, 0, 0, 0},
60 {0, 0, 0, 0, 0, 0, 0, 0},
61 {1, 0, 0, 0, 0, 0, 0, 0},
62 {0, 1, 0, 0, 0, 0, 0, 0},
63 {0, 0, 0, 0, 0, 0, 0, 0},
64 {0, 0, 0, 0, 0, 0, 0, 0},
65 {0, 0, 0, 0, 1, 0, 0, 0},
66 {0, 0, 0, 0, 0, 1, 0, 0}
67 }, CRSTag::FERMION);
68
69 const CRS<RealType> ref_c_down ({
70 {0, 0, 0, 0, 1, 0, 0, 0},
71 {0, 0, 0, 0, 0, 1, 0, 0},
72 {0, 0, 0, 0, 0, 0, -1, 0},
73 {0, 0, 0, 0, 0, 0, 0, -1},
74 {0, 0, 0, 0, 0, 0, 0, 0},
75 {0, 0, 0, 0, 0, 0, 0, 0},
76 {0, 0, 0, 0, 0, 0, 0, 0},
77 {0, 0, 0, 0, 0, 0, 0, 0}
78 }, CRSTag::FERMION);
79
80 const CRS<RealType> ref_c_down_d ({
81 {0, 0, 0, 0, 0, 0, 0, 0},
82 {0, 0, 0, 0, 0, 0, 0, 0},
83 {0, 0, 0, 0, 0, 0, 0, 0},
84 {0, 0, 0, 0, 0, 0, 0, 0},

88 {0, 0, 0, -1, 0, 0, 0, 0}
89 }, CRSTag::FERMION);
90
91 ExpectNear(model.GetOnsiteOperatorCUp() , ref_c_up , threshold);
92 ExpectNear(model.GetOnsiteOperatorCUpDagger() , ref_c_up_d , threshold);
93 ExpectNear(model.GetOnsiteOperatorCDown() , ref_c_down , threshold);
94 ExpectNear(model.GetOnsiteOperatorCDownDagger(), ref_c_down_d, threshold);
95
96}
