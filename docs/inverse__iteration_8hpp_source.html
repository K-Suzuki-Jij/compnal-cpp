27template<typename RealType>
28std::pair<int, double> InverseIteration(type::CRS<RealType> *matrix_in,
29 type::BraketVector<RealType> *eigenvector,
30 const RealType eigenvalue,
31 const ParametersII &params = ParametersII(),
32 const std::vector<type::BraketVector<RealType>> &subspace_vectors = {}
33 ) {
34
35 if (matrix_in->row_dim != matrix_in->col_dim) {
36 std::stringstream ss;
37 ss << "Error in " << __func__ << std::endl;
38 ss << "The input matrix is not a square one" << std::endl;
39 ss << "row=" << matrix_in->row_dim << ", col=" << matrix_in->col_dim << std::endl;
40 throw std::runtime_error(ss.str());
41 }
42
43 const auto start = std::chrono::system_clock::now();
44
45 type::BraketVector<RealType> improved_eigenvector;
46 type::BraketVector<RealType> vectors_work(matrix_in->row_dim);
47
48 if (params.cg.flag_use_initial_vec) {
49 if (static_cast<std::int64_t>(eigenvector->val.size()) != matrix_in->row_dim) {
50 std::stringstream ss;
51 ss << "Error in " << __func__ << std::endl;
52 ss << "The dimension of the initial vector is not equal to that of the input matrix." << std::endl;
53 throw std::runtime_error(ss.str());
54 }
55 improved_eigenvector = *eigenvector;
56 }
57
58 matrix_in->DiagonalScaling(params.diag_add - eigenvalue);
59
60 for (int step = 0; step < params.max_step; ++step) {
61 if (params.cg.flag_symmetric_crs) {
62 CalculateSymmetricMatrixVectorProduct(&vectors_work, 1.0, *matrix_in, *eigenvector);
63 }
64 else {
65 CalculateMatrixVectorProduct(&vectors_work, 1.0, *matrix_in, *eigenvector);
66 }
67 const RealType residual_error = CalculateL1Norm(params.diag_add, *eigenvector, 1.0, vectors_work);
68
69 if (params.flag_output_info) {
70 std::cout << "\rII_Step[" << step << "]=" << std::scientific << std::setprecision(1);
71 std::cout << residual_error << std::flush;
72 }
73
74 if (residual_error < params.acc) {
75 matrix_in->DiagonalScaling(-(params.diag_add - eigenvalue));
76 const auto time_count = std::chrono::duration_cast<std::chrono::microseconds>(std::chrono::system_clock::now() - start).count();
77 const double time_sec = static_cast<double>(time_count)/TIME_UNIT_CONSTANT;
78 std::cout << std::defaultfloat << std::setprecision(8) << "\rElapsed time of inverse iteration:" << time_sec << "[sec]";
79 std::cout << " (" << residual_error << ")" << std::flush;
80 std::cout << std::endl;
81 return {step, time_sec};
82 }
83
84 ConjugateGradient(&improved_eigenvector, *matrix_in, *eigenvector, params.cg, subspace_vectors);
85 improved_eigenvector.Normalize();
86 eigenvector->Assign(improved_eigenvector);
87 }
88
89 std::stringstream ss;
90 ss << "Error in " << __func__ << std::endl;
91 ss << "Does not converge" << std::endl;
92 throw std::runtime_error(ss.str());
93
94}
