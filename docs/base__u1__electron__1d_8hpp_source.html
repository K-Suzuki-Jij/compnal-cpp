150 int CalculateNumElectron(const int basis_onsite) const {
151
152 //--------------------------------
153 // # <-> [Cherge ] -- (N, 2*sz)
154 // 0 <-> [ ] -- (0, 0 )
155 // 1 <-> [up ] -- (1, 1 )
156 // 2 <-> [down ] -- (1, -1 )
157 // 3 <-> [up&down ] -- (2, 0 )
158 //--------------------------------
159
160 if (basis_onsite == 0) {
161 return 0;
162 }
163 else if (basis_onsite == 1 || basis_onsite == 2) {
164 return 1;
165 }
166 else if (basis_onsite == 3) {
167 return 2;
168 }
169 else {
170 std::stringstream ss;
171 ss << "Error in " << __FUNCTION__ << " at " << __LINE__ << std::endl;
172 ss << "Invalid onsite basis" << std::endl;
173 throw std::runtime_error(ss.str());
174 }

218 void GenerateBasis(const QType &quantum_number) {
219 if (!isValidQNumber(quantum_number)) {
220 std::stringstream ss;
221 ss << "Error in " << __FUNCTION__ << " at " << __LINE__ << std::endl;
222 ss << "Invalid parameters (system_size or magnitude_spin or total_sz)" << std::endl;
223 throw std::runtime_error(ss.str());
224 }
225
226 const auto start = std::chrono::system_clock::now();
227 const int total_electron = quantum_number.first;
228 const double total_sz = quantum_number.second;
229 const int total_2sz = utility::DoubleHalfInteger(total_sz);
230
231 if (bases_.count({total_electron, total_2sz}) != 0) {
232 return;
233 }
234
235 std::cout << "Generating Basis..." << std::flush;
236 const int max_n_up_down = static_cast<int>(total_electron/2);
237 const std::int64_t dim_target = CalculateTargetDim({total_electron, total_sz});
238
239 std::vector<std::int64_t>().swap(bases_[{total_electron, total_2sz}]);
240 auto &basis_ref = bases_.at({total_electron, total_2sz});
241
242 std::vector<std::int64_t> site_constant(system_size_);
243 for (int site = 0; site < system_size_; ++site) {
244 site_constant[site] = static_cast<std::int64_t>(std::pow(dim_onsite_, site));
245 }
246
247 std::vector<int> basis_list(system_size_);
248
249#ifdef _OPENMP
250 const int num_threads = omp_get_max_threads();
251 std::vector<std::vector<std::int64_t>> temp_basis(num_threads);
252 for (int n_up_down = 0; n_up_down <= max_n_up_down; ++n_up_down) {
253 const int n_up = static_cast<int>((total_electron - 2*n_up_down + total_2sz)/2);
254 const int n_down = static_cast<int>((total_electron - 2*n_up_down - total_2sz)/2);
255 const int n_vac = system_size_ - total_electron + n_up_down;
256 if (0 <= n_up && 0 <= n_down && 0 <= n_vac) {
257 for (int s = 0; s < n_vac; ++s) {
258 basis_list[s] = 0;
259 }
260 for (int s = 0; s < n_up; ++s) {
261 basis_list[s + n_vac] = 1;
262 }
263 for (int s = 0; s < n_down; ++s) {
264 basis_list[s + n_vac + n_up] = 2;
265 }
266 for (int s = 0; s < n_up_down; ++s) {
267 basis_list[s + n_vac + n_up + n_down] = 3;
268 }
269
270 const std::int64_t size = utility::CalculateNumCombination(basis_list);
271 std::vector<std::vector<int>> temp_basis_list(num_threads);
272
273#pragma omp parallel num_threads (num_threads)
274 {
275 const int thread_num = omp_get_thread_num();
276 const std::int64_t loop_begin = thread_num*size/num_threads;
277 const std::int64_t loop_end = (thread_num + 1)*size/num_threads;
278 temp_basis_list[thread_num] = basis_list;
279 utility::CalculateNthPermutation(&temp_basis_list[thread_num], loop_begin);
280
281 for (std::int64_t j = loop_begin; j < loop_end; ++j) {
282 std::int64_t basis_global = 0;
283 for (std::size_t k = 0; k < temp_basis_list[thread_num].size(); ++k) {
284 basis_global += temp_basis_list[thread_num][k]*site_constant[k];
285 }
286 temp_basis[thread_num].push_back(basis_global);
287 std::next_permutation(temp_basis_list[thread_num].begin(), temp_basis_list[thread_num].end());
288 }
289 }
290 }
291 }
292 for (auto &&basis: temp_basis) {
293 basis_ref.insert(basis_ref.end(), basis.begin(), basis.end());
294 std::vector<std::int64_t>().swap(basis);
295 }
296
297#else
298
299 basis_ref.reserve(dim_target);
300 for (int n_up_down = 0; n_up_down <= max_n_up_down; ++n_up_down) {
301 const int n_up = static_cast<int>((total_electron - 2*n_up_down + total_2sz)/2);
302 const int n_down = static_cast<int>((total_electron - 2*n_up_down - total_2sz)/2);
303 const int n_vac = system_size_ - total_electron + n_up_down;
304 if (0 <= n_up && 0 <= n_down && 0 <= n_vac) {
305 for (int s = 0; s < n_vac; ++s) {
306 basis_list[s] = 0;
307 }
308 for (int s = 0; s < n_up; ++s) {
309 basis_list[s + n_vac] = 1;
310 }
311 for (int s = 0; s < n_down; ++s) {
312 basis_list[s + n_vac + n_up] = 2;
313 }
314 for (int s = 0; s < n_up_down; ++s) {
315 basis_list[s + n_vac + n_up + n_down] = 3;
316 }
317
318 do {
319 std::int64_t basis_global = 0;
320 for (std::size_t j = 0; j < basis_list.size(); ++j) {
321 basis_global += basis_list[j]*site_constant[j];
322 }
323 basis_ref.push_back(basis_global);
324 } while (std::next_permutation(basis_list.begin(), basis_list.end()));
325 }
326 }
327#endif
328
329 if (static_cast<std::int64_t>(bases_.at({total_electron, total_2sz}).size()) != dim_target) {
330 std::stringstream ss;
331 ss << "Unknown error detected in " << __FUNCTION__ << " at " << __LINE__ << std::endl;
332 throw std::runtime_error(ss.str());
333 }
334
335 std::sort(basis_ref.begin(), basis_ref.end());
336
337 bases_inv_[{total_electron, total_2sz}].clear();
338
339 auto &basis_inv_ref = bases_inv_.at({total_electron, total_2sz});
340 for (std::int64_t i = 0; i < dim_target; ++i) {
341 basis_inv_ref[basis_ref[i]] = i;
342 }
343
344 if (basis_inv_ref.size() != basis_ref.size()) {
345 std::stringstream ss;
346 ss << "Unknown error detected in " << __FUNCTION__ << " at " << __LINE__ << std::endl;
347 ss << "The same basis has been detected" << std::endl;
348 throw std::runtime_error(ss.str());
349 }
350 const auto time_count = std::chrono::duration_cast<std::chrono::microseconds>(std::chrono::system_clock::now() - start).count();
351 const double time_sec = static_cast<double>(time_count)/blas::TIME_UNIT_CONSTANT;
352 std::cout << "\rElapsed time of generating basis:" << time_sec << "[sec]" << std::endl;
353 }

359 std::vector<QType> GenerateTargetSector(const CRS &m_1, const CRS &m_2) const {
360 std::unordered_set<QType, utility::PairHash> delta_sector_set_m1;
361 std::unordered_set<QType, utility::PairHash> delta_sector_set_m2;
362 for (std::int64_t i = 0; i < m_1.row_dim; ++i) {
363 for (std::int64_t j = m_1.row[i]; j < m_1.row[i + 1]; ++j) {
364 if (m_1.val[j] != 0.0) {
365 delta_sector_set_m1.emplace(CalculateQuntumNumberDifference(static_cast<int>(i), static_cast<int>(m_1.col[j])));
366 }
367 }
368 }
369 for (std::int64_t i = 0; i < m_2.row_dim; ++i) {
370 for (std::int64_t j = m_2.row[i]; j < m_2.row[i + 1]; ++j) {
371 if (m_2.val[j] != 0.0) {
372 delta_sector_set_m2.emplace(CalculateQuntumNumberDifference(static_cast<int>(i), static_cast<int>(m_2.col[j])));
373 }
374 }
375 }
376 std::vector<QType> target_sector_set;
377 for (const auto &del_sec_m1: delta_sector_set_m1) {
378 for (const auto &del_sec_m2: delta_sector_set_m2) {
379 const bool c1 = isValidQNumber(del_sec_m1.first + total_electron_, del_sec_m1.second + 0.5*total_2sz_);
380 if (del_sec_m1 == del_sec_m2 && c1) {
381 target_sector_set.push_back({del_sec_m1.first + total_electron_, del_sec_m1.second + 0.5*total_2sz_});
382 }
383 }
384 }
385 std::sort(target_sector_set.begin(), target_sector_set.end());
386 target_sector_set.erase(std::unique(target_sector_set.begin(), target_sector_set.end()), target_sector_set.end());
387 return target_sector_set;
388 }

396 std::unordered_set<QType, utility::PairHash> delta_sector_set_m1;

400 for (std::int64_t i = 0; i < m_1_bra.row_dim; ++i) {

412 }

415

417 for (std::int64_t j = m_3_ket.row[i]; j < m_3_ket.row[i + 1]; ++j) {
418 if (m_3_ket.val[j] != 0.0) {

420 }
421 }
422 }

425
426 for (const auto &del_sec_m1: delta_sector_set_m1) {
427 for (const auto &del_sec_m2: delta_sector_set_m2) {
428 for (const auto &del_sec_m3: delta_sector_set_m3) {
429 const QType del_sec_m2_m3 = {del_sec_m2.first + del_sec_m3.first, del_sec_m2.second + del_sec_m3.second};

431 const bool c2 = isValidQNumber(del_sec_m3.first + total_electron_, del_sec_m3.second + 0.5*total_2sz_);
432 if (del_sec_m1 == del_sec_m2_m3 && c1 && c2) {
433 target_sector_set.push_back({
434 {del_sec_m1.first + total_electron_, del_sec_m1.second + 0.5*total_2sz_},
435 {del_sec_m3.first + total_electron_, del_sec_m3.second + 0.5*total_2sz_}
436 });
437 }
438 }
439 }
440 }
441 std::sort(target_sector_set.begin(), target_sector_set.end());
442 target_sector_set.erase(std::unique(target_sector_set.begin(), target_sector_set.end()), target_sector_set.end());
443 return target_sector_set;

523 static bool isValidQNumber(const int system_size, const int total_electron, const double total_sz) {
524 const int total_2sz = utility::DoubleHalfInteger(total_sz);
525 const bool c1 = (0 <= total_electron && total_electron <= 2*system_size);
526 const bool c2 = ((total_electron - total_2sz)%2 == 0);
527 const bool c3 = (-total_electron <= total_2sz && total_2sz <= total_electron);
528 if (c1 && c2 && c3) {
529 return true;
530 }
531 else {
532 return false;
533 }

541 static std::int64_t CalculateTargetDim(const int system_size, const int total_electron, const double total_sz) {
542 if (!isValidQNumber(system_size, total_electron, total_sz)) {
543 return 0;
544 }
545 const int total_2sz = utility::DoubleHalfInteger(total_sz);
546 const std::vector<std::vector<std::int64_t>> binom = utility::CalculateBinomialTable(system_size);
547 const int max_n_up_down = static_cast<int>(total_electron/2);
548 std::int64_t dim = 0;
549 for (int n_up_down = 0; n_up_down <= max_n_up_down; ++n_up_down) {
550 const int n_up = static_cast<int>((total_electron - 2*n_up_down + total_2sz)/2);
551 const int n_down = static_cast<int>((total_electron - 2*n_up_down - total_2sz)/2);
552 const int n_vac = system_size - total_electron + n_up_down;
553 if (0 <= n_up && 0 <= n_down && 0 <= n_vac) {
554 // TODO: Detect Overflow
555 dim += binom[system_size][n_up]*binom[system_size - n_up][n_down]*binom[system_size - n_up - n_down][n_up_down];
556 }
557 }
558 return dim;
559 }

563 static CRS CreateOnsiteOperatorCUp() {
564
565 //--------------------------------
566 // # <-> [Cherge ] -- (N, 2*sz)
567 // 0 <-> [ ] -- (0, 0 )
568 // 1 <-> [up ] -- (1, 1 )
569 // 2 <-> [down ] -- (1, -1 )
570 // 3 <-> [up&down ] -- (2, 0 )
571 //--------------------------------
572
573 const int dim_onsite = 4;
574 CRS matrix(dim_onsite, dim_onsite);
575 for (int row = 0; row < dim_onsite; row++) {
576 for (int col = 0; col < dim_onsite; col++) {
577 if ((col == 1 && row == 0) || (col == 3 && row == 2)) {
578 matrix.col.push_back(col);
579 matrix.val.push_back(1.0);
580 }
581 }
582 matrix.row[row + 1] = matrix.col.size();
583 }
584
585 matrix.tag = type::CRSTag::FERMION;
586
587 return matrix;
588 }

592 static CRS CreateOnsiteOperatorCDown() {
593
594 //--------------------------------
595 // # <-> [Cherge ] -- (N, 2*sz)
596 // 0 <-> [ ] -- (0, 0 )
597 // 1 <-> [up ] -- (1, 1 )
598 // 2 <-> [down ] -- (1, -1 )
599 // 3 <-> [up&down ] -- (2, 0 )
600 //--------------------------------
601
602 const int dim_onsite = 4;
603 CRS matrix(dim_onsite, dim_onsite);
604 for (int row = 0; row < dim_onsite; row++) {
605 for (int col = 0; col < dim_onsite; col++) {
606 if (col == 2 && row == 0) {
607 matrix.col.push_back(col);
608 matrix.val.push_back(1.0);
609 }
610 else if (col == 3 && row == 1) {

612 matrix.val.push_back(-1.0);
613 }
614 }
615 matrix.row[row + 1] = matrix.col.size();
616 }
