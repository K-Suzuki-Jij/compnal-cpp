57 void AddPotential(const IndexType &site, const CRS &m, const RealType val = 1.0) {

77 if (potential_list_.count(site) == 0) {
78 potential_list_[site] = val*m;

92 void AddInteraction(const IndexType &site_1, const CRS &m_1, const IndexType &site_2, const CRS &m_2, const RealType val = 1.0) {
93 if (val == 0.0) {
94 return;
95 }
96 if (m_1.row_dim != m_1.col_dim || m_2.row_dim != m_2.col_dim) {
97 std::stringstream ss;
98 ss << "Error at " << __LINE__ << " in " << __FUNCTION__ << " in "<< __FILE__ << std::endl;
99 ss << "The input matrix is not a square matrix." << std::endl;
100 throw std::runtime_error(ss.str());
101 }
102 if (m_1.row_dim != this->GetDimOnsite()) {
103 std::stringstream ss;
104 ss << "Error at " << __LINE__ << " in " << __FUNCTION__ << " in "<< __FILE__ << std::endl;
105 ss << "The input matrix is invalid" << std::endl;
106 throw std::runtime_error(ss.str());
107 }
108
109 if (site_1 == site_2) {
110 AddPotential(site_1, m_1*m_2, val);
111 return;
112 }
113 interaction_list_[site_1][site_2].push_back({val*m_1, m_2});
114 index_list_.emplace(site_1);
115 index_list_.emplace(site_2);
116 }
117
