60 void AddPotential(const IndexType &site, const CRS &m, const RealType val = 1.0) {
61
62 if (val == 0.0) {
63 return;
64 }
65
66 if (m.row_dim != m.col_dim) {
67 std::stringstream ss;
68 ss << "Error at " << __LINE__ << " in " << __FUNCTION__ << " in "<< __FILE__ << std::endl;
69 ss << "The input matrix is not a square matrix." << std::endl;
70 throw std::runtime_error(ss.str());

72
73 if (m.row_dim != this->GetDimOnsite()) {
74 std::stringstream ss;
75 ss << "Error at " << __LINE__ << " in " << __FUNCTION__ << " in "<< __FILE__ << std::endl;
76 ss << "The input matrix is invalid" << std::endl;
77 throw std::runtime_error(ss.str());
78 }
79
80 if (potential_list_.count(site) == 0) {
81 potential_list_[site] = val*m;
82 }
83 else {
84 potential_list_.at(site) = potential_list_.at(site) + val*m;
85 }
86 index_list_.emplace(site);
87 }

95 void AddInteraction(const IndexType &site_1, const CRS &m_1, const IndexType &site_2, const CRS &m_2, const RealType val = 1.0) {
96 if (val == 0.0) {
97 return;
98 }
99 if (m_1.row_dim != m_1.col_dim || m_2.row_dim != m_2.col_dim) {
100 std::stringstream ss;
101 ss << "Error at " << __LINE__ << " in " << __FUNCTION__ << " in "<< __FILE__ << std::endl;
102 ss << "The input matrix is not a square matrix." << std::endl;
103 throw std::runtime_error(ss.str());
104 }
105 if (m_1.row_dim != this->GetDimOnsite()) {
106 std::stringstream ss;
107 ss << "Error at " << __LINE__ << " in " << __FUNCTION__ << " in "<< __FILE__ << std::endl;
108 ss << "The input matrix is invalid" << std::endl;
109 throw std::runtime_error(ss.str());
110 }
111
112 if (site_1 == site_2) {
113 AddPotential(site_1, m_1*m_2, val);
114 return;
115 }
116 interaction_list_[site_1][site_2].push_back({val*m_1, m_2});
117 index_list_.emplace(site_1);
118 index_list_.emplace(site_2);
119 }
120
123 std::int64_t GetSystemSize() const {
124 return static_cast<std::int64_t>(index_list_.size());
125 }
