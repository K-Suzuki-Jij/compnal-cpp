34template<typename RealType>
35std::pair<int, double> ConjugateGradient(type::BraketVector<RealType> *vec_out,
36 const type::CRS<RealType> &matrix_in,
37 const type::BraketVector<RealType> &vec_in,
38 const ParametersCG &params = ParametersCG(),
39 const std::vector<type::BraketVector<RealType>> &subspace_vectors = {}
40 ) {
41
42 if (matrix_in.row_dim != matrix_in.col_dim) {
43 std::stringstream ss;
44 ss << "Error in " << __func__ << std::endl;
45 ss << "The input matrix is not a square one" << std::endl;
46 ss << "row=" << matrix_in.row_dim << ", col=" << matrix_in.col_dim << std::endl;
47 throw std::runtime_error(ss.str());
48 }
49
50 if (static_cast<std::int64_t>(vec_in.val.size()) != matrix_in.row_dim) {
51 std::stringstream ss;
52 ss << "Error in " << __func__ << std::endl;
53 ss << "Matrix vector product (Ax=b) cannot be defined." << std::endl;
54 throw std::runtime_error(ss.str());
55 }
56
57 const auto start = std::chrono::system_clock::now();
58 const std::int64_t dim = matrix_in.row_dim;
59 type::BraketVector<RealType> rrr(dim);
60 type::BraketVector<RealType> ppp(dim);
61 type::BraketVector<RealType> yyy(dim);
62 std::vector<std::vector<RealType>> vectors_work;
63
64 if (params.flag_symmetric_crs) {
65#ifdef _OPENMP
66 vectors_work = std::vector<std::vector<RealType>>(omp_get_max_threads(), std::vector<RealType>(dim));
67#endif
68 }
69
70 if (params.flag_use_initial_vec) {
71 if (static_cast<std::int64_t>(vec_out->val.size()) != dim) {
72 std::stringstream ss;
73 ss << "Error in " << __func__ << std::endl;
74 ss << "The dimension of the initial vector is not equal to that of the input matrix." << std::endl;
75 throw std::runtime_error(ss.str());
76 }
77 }
78 else {
79 std::uniform_real_distribution<RealType> uniform_rand(-1, 1);
80 std::mt19937 random_number_engine;
81 random_number_engine.seed(std::random_device()());
82 vec_out->val.resize(dim);
83 for (std::int64_t i = 0; i < dim; ++i) {
84 vec_out->val[i] = uniform_rand(random_number_engine);
85 }
86 }
87 Orthonormalize(vec_out, subspace_vectors);
88 vec_out->Normalize();
89 if (params.flag_symmetric_crs) {
90 CalculateSymmetricMatrixVectorProduct(&rrr, 1.0, matrix_in, *vec_out, &vectors_work);
91 }
92 else {
93 CalculateMatrixVectorProduct(&rrr, 1.0, matrix_in, *vec_out);
94 }
95
96#pragma omp parallel for
97 for (std::int64_t i = 0; i < dim; ++i) {
98 rrr.val[i] = vec_in.val[i] - rrr.val[i];
99 ppp.val[i] = rrr.val[i];
100 }
101
102 Orthonormalize(&rrr, subspace_vectors);
103 Orthonormalize(&ppp, subspace_vectors);
104
105 for (int step = 0; step < params.max_step; ++step) {
106 if (params.flag_symmetric_crs) {
107 CalculateSymmetricMatrixVectorProduct(&yyy, 1.0, matrix_in, ppp, &vectors_work);
108 }
109 else {
110 CalculateMatrixVectorProduct(&yyy, 1.0, matrix_in, ppp);
111 }
112
113 Orthonormalize(&yyy, subspace_vectors);
114
115 const RealType inner_prod = CalculateInnerProduct(rrr, rrr);
116 const RealType alpha = inner_prod/CalculateInnerProduct(ppp, yyy);
117
118#pragma omp parallel for
119 for (std::int64_t i = 0; i < dim; ++i) {
120 vec_out->val[i] += alpha*ppp.val[i];
121 rrr.val[i] -= alpha*yyy.val[i];
122 }
123
124 Orthonormalize(vec_out, subspace_vectors);
125 Orthonormalize(&rrr , subspace_vectors);
126
127 const RealType residual_error = CalculateInnerProduct(rrr, rrr);
128
129 if (params.flag_output_info) {
130 std::cout << "\rCG_Step[" << step << "]=" << std::scientific << std::setprecision(1);
131 std::cout << residual_error << std::string(5, ' ') << std::flush;
132 }
133
134 if (residual_error < params.acc) {
135 const auto time_count = std::chrono::duration_cast<std::chrono::microseconds>(std::chrono::system_clock::now() - start).count();
136 const double time_sec = static_cast<double>(time_count)/TIME_UNIT_CONSTANT;
137 std::cout << std::defaultfloat << std::setprecision(8) << "\rElapsed time of conjugate_gradient:" << time_sec << "[sec]";
138 std::cout << " (" << residual_error << ")" << std::flush;
139 std::cout << std::endl;
140 return {step, time_sec};
141 }
142
143 const RealType beta = residual_error/inner_prod;
144
145#pragma omp parallel for
146 for (std::int64_t i = 0; i < dim; ++i) {
147 ppp.val[i] = rrr.val[i] + beta*ppp.val[i];
148 }
149 Orthonormalize(&ppp, subspace_vectors);
150 }
151
152 std::stringstream ss;
153 ss << "Error in " << __func__ << std::endl;
154 ss << "Does not converge" << std::endl;
155 throw std::runtime_error(ss.str());
156
157}
