48 void AddOnsitePotential(const CRS &m, const int site, const RealType val = 1.0) {
49 if (val == 0.0) {
50 return;
51 }
52 // TODO: Check of m is valid with BaseClass operator
53 if (m.row_dim != m.col_dim) {
54 std::stringstream ss;

85 void AddInteraction(const CRS &m_1, const int site_1, const CRS &m_2, const int site_2, const RealType val = 1.0) {
86 if (val == 0.0) {
87 return;
88 }
89 if (m_1.row_dim != m_1.col_dim || m_2.row_dim != m_2.col_dim) {
90 std::stringstream ss;

95 if (m_1.row_dim != this->dim_onsite_) {
96 std::stringstream ss;
97 ss << "Error in " << __FUNCTION__ << " at " << __LINE__ << std::endl;
98 ss << "The input matrix is invalid" << std::endl;
99 throw std::runtime_error(ss.str());
100 }
101 if (site_1 < 0 || this->system_size_ <= site_1) {
102 std::stringstream ss;
103 ss << "Error in " << __FUNCTION__ << " at " << __LINE__ << std::endl;
104 ss << "Invalid site: " << site_1 << std::endl;
105 throw std::runtime_error(ss.str());
106 }
107 if (site_2 < 0 || this->system_size_ <= site_2) {
108 std::stringstream ss;
109 ss << "Error in " << __FUNCTION__ << " at " << __LINE__ << std::endl;
110 ss << "Invalid site: " << site_2 << std::endl;
111 throw std::runtime_error(ss.str());
112 }
113 if (site_1 == site_2) {
114 AddOnsitePotential(m_1*m_2, site_1, val);
115 return;
116 }
117 intersite_operator_list_[site_1][site_2].push_back({m_1.MultiplyByScalar(val), m_2});
118 }
