57template<typename IntegerType>
58struct VariantHash {
59 static_assert(std::is_integral<IntegerType>::value, "Template parameter IntegerType must be integer type");
60
61 using VariantVecType = std::vector<std::variant<IntegerType, std::string>>;
62
63 template<class... Types>
64 std::size_t operator() (const std::variant<Types...> &v) const {
65 if (std::holds_alternative<IntegerType>(v)) {
66 return std::hash<IntegerType>()(std::get<IntegerType>(v));
67 }
68 else if (std::holds_alternative<std::string>(v)) {
69 return std::hash<std::string>()(std::get<std::string>(v));
70 }
71 else if (std::holds_alternative<VariantVecType>(v)) {
72 const auto &variant_vec = std::get<VariantVecType>(v);
73 std::size_t hash = variant_vec.size();
74 for (const auto &i : variant_vec) {
75 if (std::holds_alternative<IntegerType>(i)) {
76 hash ^= std::hash<IntegerType>()(std::get<IntegerType>(i)) + 0x9e3779b9 + (hash << 6) + (hash >> 2);
77 }
78 else if (std::holds_alternative<std::string>(i)) {
79 hash ^= std::hash<std::string>()(std::get<std::string>(i)) + 0x9e3779b9 + (hash << 6) + (hash >> 2);
80 }
81 else {
82 throw std::runtime_error("Invalid template parameters");
83 }
84 }
85 return hash;
86 }
87 else {
88 throw std::runtime_error("Invalid template parameters");
89 }
90 }
91};
