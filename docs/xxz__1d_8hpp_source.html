179 static CRS CreateOnsiteOperatorHam(const double magnitude_spin, const RealType h_z = 0.0, const RealType D_z = 0.0) {
180 const int magnitude_2spin = utility::DoubleTheNumber(magnitude_spin);
181 const int dim_onsite = magnitude_2spin + 1;
182 CRS matrix(dim_onsite, dim_onsite);
183
184 for (int row = 0; row < dim_onsite; ++row) {
185 const RealType val = h_z*(magnitude_spin - row) + D_z*D_z*(magnitude_spin - row);
186 if (val != 0.0) {
187 matrix.val.push_back(val);
188 matrix.col.push_back(row);
189 }
190 matrix.row[row + 1] = matrix.col.size();
191 }
192 return matrix;
193 }
