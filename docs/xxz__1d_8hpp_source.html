188 static CRS CreateOnsiteOperatorHam(const double magnitude_spin, const RealType h_z = 0.0, const RealType D_z = 0.0) {
189 const int magnitude_2spin = utility::DoubleTheNumber(magnitude_spin);
190 const int dim_onsite = magnitude_2spin + 1;
191 CRS matrix(dim_onsite, dim_onsite);
192
193 for (int row = 0; row < dim_onsite; ++row) {
194 const RealType val = h_z*(magnitude_spin - row) + D_z*D_z*(magnitude_spin - row);
195 if (val != 0.0) {
196 matrix.val.push_back(val);
197 matrix.col.push_back(row);
198 }
199 matrix.row[row + 1] = matrix.col.size();
200 }
201 return matrix;
202 }
