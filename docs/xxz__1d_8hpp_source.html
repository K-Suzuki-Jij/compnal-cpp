189 static CRS CreateOnsiteOperatorHam(const double magnitude_spin, const RealType h_z = 0.0, const RealType D_z = 0.0) {
190 const int magnitude_2spin = utility::DoubleHalfInteger(magnitude_spin);
191 const int dim_onsite = magnitude_2spin + 1;
192 CRS matrix(dim_onsite, dim_onsite);
193
194 for (int row = 0; row < dim_onsite; ++row) {
195 const RealType val = h_z*(magnitude_spin - row) + D_z*D_z*(magnitude_spin - row);
196 if (val != 0.0) {
197 matrix.val.push_back(val);
198 matrix.col.push_back(row);
199 }
200 matrix.row[row + 1] = matrix.col.size();
201 }
202 return matrix;
203 }
