17template<typename RealType>
18void CalculateMatrixVectorProduct(BraketVector<RealType> *vector_out,
19 const RealType coeef,
20 const CRS<RealType> &matrix_in,
21 const BraketVector<RealType> &vector_in) {
22
23 if (matrix_in.col_dim != static_cast<std::int64_t>(vector_in.val.size())) {
24 std::stringstream ss;
25 ss << "Error in " << __func__ << std::endl;
26 ss << "The column of the input matrix is " << matrix_in.col_dim << std::endl;
27 ss << "The dimension of the input vector is " << vector_in.val.size() << std::endl;
28 ss << "Both must be equal" << std::endl;
29 throw std::runtime_error(ss.str());
30 }
31 vector_out->val.resize(matrix_in.row_dim);
32#pragma omp parallel for
33 for (std::int64_t i = 0; i < matrix_in.row_dim; ++i) {
34 RealType temp = 0.0;
35 for (std::int64_t j = matrix_in.row[i]; j < matrix_in.row[i+1]; ++j) {
36 temp += matrix_in.val[j]*vector_in.val[matrix_in.col[j]];
37 }
38 vector_out->val[i] = temp*coeef;
39 }
40}

42template<typename RealType>
43void CalculateSymmetricMatrixVectorProduct(BraketVector<RealType> *vector_out,
44 const RealType coeef,
45 const CRS<RealType> &matrix_in,
46 const BraketVector<RealType> &vector_in,
47 std::vector<std::vector<RealType>> *vectors_work) {
48
49 if (matrix_in.row_dim != matrix_in.col_dim) {
50 std::stringstream ss;
51 ss << "Error in " << __func__ << std::endl;
52 ss << "The input matrix is not symmetric" << std::endl;
53 throw std::runtime_error(ss.str());
54 }
55
56 if (matrix_in.col_dim != static_cast<std::int64_t>(vector_in.val.size())) {
57 std::stringstream ss;
58 ss << "Error in " << __func__ << std::endl;
59 ss << "The column of the input matrix is " << matrix_in.col_dim << std::endl;
60 ss << "The dimension of the input vector is " << vector_in.val.size() << std::endl;
61 ss << "Both must be equal" << std::endl;
62 throw std::runtime_error(ss.str());
63 }
64 vector_out->val.resize(matrix_in.row_dim);
65
66#ifdef _OPENMP
67 const int num_threads = omp_get_max_threads();
68 if (static_cast<int>(vectors_work->size()) != num_threads) {
69 std::stringstream ss;
70 ss << "Error in " << __func__ << std::endl;
71 ss << "Working vector (vectors_work) must be arrays of the number of parallel threads";
72 throw std::runtime_error(ss.str());
73 }
74
75#pragma omp parallel for schedule (guided)
76 for (std::int64_t i = 0; i < matrix_in.row_dim; ++i) {
77 const int thread_num = omp_get_thread_num();
78 const RealType temp_vec_in = vector_in.val[i];
79 RealType temp_val = matrix_in.val[matrix_in.row[i + 1] - 1]*temp_vec_in;
80 for (std::int64_t j = matrix_in.row[i]; j < matrix_in.row[i + 1] - 1; ++j) {
81 temp_val += matrix_in.val[j]*vector_in.val[matrix_in.col[j]];
82 (*vectors_work)[thread_num][matrix_in.col[j]] += matrix_in.val[j]*temp_vec_in;
83 }
84 (*vectors_work)[thread_num][i] += temp_val;
85 }
86
87#pragma omp parallel for
88 for (std::int64_t i = 0; i < matrix_in.row_dim; ++i) {
89 RealType temp_val = 0.0;
90 for (int thread_num = 0; thread_num < num_threads; ++thread_num) {
91 temp_val += (*vectors_work)[thread_num][i];
92 (*vectors_work)[thread_num][i] = 0.0;
93 }
94 vector_out->val[i] = temp_val*coeef;
95 }
96
97#else
98 vector_out->Fill(0.0);
99 for (std::int64_t i = 0; i < matrix_in.row_dim; ++i) {
100 const RealType temp_vec_in = vector_in.val[i];
101 RealType temp_val = matrix_in.val[matrix_in.row[i + 1] - 1]*temp_vec_in;
102 for (std::int64_t j = matrix_in.row[i]; j < matrix_in.row[i + 1] - 1; ++j) {
103 temp_val += matrix_in.val[j]*vector_in.val[matrix_in.col[j]];
104 vector_out->val[matrix_in.col[j]] += matrix_in.val[j]*temp_vec_in;
105 }
106 vector_out->val[i] += temp_val*coeef;
107 }
108
109#endif
110}
