194 static CRS CreateOnsiteOperatorHam(const RealType h_z = 0.0, const RealType U = 0.0, const RealType mu = 0.0) {
195 const CRS m_s_z = BaseU1Electron_1D<RealType>::CreateOnsiteOperatorSz();
196 const CRS m_n_up = BaseU1Electron_1D<RealType>::CreateOnsiteOperatorNCUp();
197 const CRS m_n_down = BaseU1Electron_1D<RealType>::CreateOnsiteOperatorNCDown();
198 const CRS m_mu = BaseU1Electron_1D<RealType>::CreateOnsiteOperatorNC();
199 return h_z*m_s_z + U*m_n_up*m_n_down + mu*m_mu;
200 }
