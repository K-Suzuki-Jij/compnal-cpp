205 static CRS CreateOnsiteOperatorHam(const RealType h_z = 0.0, const RealType U = 0.0, const RealType mu = 0.0) {
206 const CRS m_s_z = BaseU1Electron_1D<RealType>::CreateOnsiteOperatorSz();
207 const CRS m_n_up = BaseU1Electron_1D<RealType>::CreateOnsiteOperatorNCUp();
208 const CRS m_n_down = BaseU1Electron_1D<RealType>::CreateOnsiteOperatorNCDown();
209 const CRS m_mu = BaseU1Electron_1D<RealType>::CreateOnsiteOperatorNC();
210 return h_z*m_s_z + U*m_n_up*m_n_down + mu*m_mu;
211 }
