206 static std::int64_t CalculateTargetDim(const int system_size,
207 const double magnitude_lspin,
208 const std::vector<int> &total_electron,
209 const double total_sz) {
210 std::vector<std::vector<std::vector<int>>> electron_configuration_list;
211 std::vector<int> length_list;
212 std::int64_t length = 1;
213 for (const auto num_electron: total_electron) {
214 const auto electron_configuration = GenerateElectronConfigurations(system_size, num_electron);
215 electron_configuration_list.push_back(electron_configuration);
216 length_list.push_back(electron_configuration[0].size());
217 length *= electron_configuration[0].size();
218 }
219 const int total_2sz = utility::DoubleTheNumber(total_sz);
220 const std::vector<std::vector<std::int64_t>> binom = utility::CalculateBinomialTable(system_size);
221 std::int64_t dim = 0;
222 for (std::int64_t i = 0; i < length; ++i) {
223 int electron_2sz = 0;
224 std::int64_t electron_dim = 1;
225 for (std::size_t j = 0; j < length_list.size(); ++j) {
226 std::int64_t prod = 1;
227 for (std::size_t k = j + 1; k < length_list.size(); ++k) {
228 prod *= length_list[k];
229 }
230 const std::size_t index = (i/prod)%length_list[j];
231 const int n_up_down = electron_configuration_list[j][0][index];
232 const int n_up = electron_configuration_list[j][1][index];
233 const int n_down = electron_configuration_list[j][2][index];
234 electron_2sz += n_up - n_down;
235 electron_dim *= binom[system_size][n_up]*binom[system_size - n_up][n_down]*binom[system_size - n_up - n_down][n_up_down];
236 }
237 const int spin_2sz = total_2sz - electron_2sz;
238 if (BaseU1Spin_1D<RealType>::isValidQNumber(system_size, magnitude_lspin, 0.5*spin_2sz)) {
239 dim += electron_dim*BaseU1Spin_1D<RealType>::CalculateTargetDim(system_size, magnitude_lspin, 0.5*spin_2sz);
240 }
241 }
242 return dim;
243 }

251 static CRS CreateOnsiteOperatorCUp(const double magnitude_lspin, const int orbital, const int num_orbital) {
252 const int magnitude_2lspin = utility::DoubleTheNumber(magnitude_lspin);
253 const int dim_onsite_lspin = magnitude_2lspin + 1;
254 const int dim_onsite = dim_onsite_lspin*static_cast<int>(std::pow(4, num_orbital));
255
256 //--------------------------------
257 // # <-> [Cherge ] -- (N, 2*sz)
258 // 0 <-> [ ] -- (0, 0 )
259 // 1 <-> [up ] -- (1, 1 )
260 // 2 <-> [down ] -- (1, -1 )
261 // 3 <-> [up&down ] -- (2, 0 )
262 //--------------------------------
263
264 CRS matrix(dim_onsite, dim_onsite);
265
266 for (int row = 0; row < dim_onsite; ++row) {
267 int num_electron = 0;
268 for (int o = 0; o < orbital; ++o) {
269 const int basis_electron_onsite = CalculateBasisOnsiteElectron(row, magnitude_lspin, o, num_orbital);
270 num_electron += CalculateNumElectron(basis_electron_onsite);
271 }
272 int sign = 1;
273 if (num_electron%2 == 1) {
274 sign = -1;
275 }
276
277 for (int col = 0; col < dim_onsite; col++) {
278 bool c_1 = true;
279 for (int o = 0; o < num_orbital; ++o) {
280 const int basis_1 = CalculateBasisOnsiteElectron(row, magnitude_lspin, o, num_orbital);
281 const int basis_2 = CalculateBasisOnsiteElectron(col, magnitude_lspin, o, num_orbital);
282 if (o != orbital && basis_1 != basis_2) {
283 c_1 = false;
284 }
285 }
286
287 int basis_row_electron = CalculateBasisOnsiteElectron(row, magnitude_lspin, orbital, num_orbital);
288 int basis_col_electron = CalculateBasisOnsiteElectron(col, magnitude_lspin, orbital, num_orbital);
289
290 const bool c_2 = (basis_row_electron == 0 && basis_col_electron == 1);
291 const bool c_3 = (basis_row_electron == 2 && basis_col_electron == 3);
292
293 if (c_1 && row%dim_onsite_lspin == col%dim_onsite_lspin && (c_2 || c_3)) {
294 matrix.val.push_back(sign);
295 matrix.col.push_back(col);
296 }
297
298 }
299 matrix.row.push_back(matrix.col.size());
300 }
301 return matrix;
302 }

372 static CRS CreateOnsiteOperatorSzL(const double magnitude_lspin, const int num_orbital) {
373 const int magnitude_2lspin = utility::DoubleTheNumber(magnitude_lspin);
374 const int dim_onsite_lspin = magnitude_2lspin + 1;
375 const int dim_onsite = dim_onsite_lspin*static_cast<int>(std::pow(4, num_orbital));
376 CRS matrix(dim_onsite, dim_onsite);
377 for (int row = 0; row < dim_onsite; ++row) {
378 for (int col = 0; col < dim_onsite; ++col) {
379 const int basis_row_lspin = row%dim_onsite_lspin;
380 const int basis_col_lspin = col%dim_onsite_lspin;
381 const int basis_row_total_electron = row/dim_onsite_lspin;
382 const int basis_col_total_electron = col/dim_onsite_lspin;
383 if (basis_row_total_electron == basis_col_total_electron && basis_row_lspin == basis_col_lspin) {
384 matrix.val.push_back(magnitude_lspin - basis_row_lspin);
385 matrix.col.push_back(col);
386 }
387 }
388 matrix.row.push_back(matrix.col.size());
389 }
390 return matrix;
391 }

397 static CRS CreateOnsiteOperatorSpL(const double magnitude_lspin, const int num_orbital) {
398 const int magnitude_2lspin = utility::DoubleTheNumber(magnitude_lspin);
399 const int dim_onsite_lspin = magnitude_2lspin + 1;
400 const int dim_onsite = dim_onsite_lspin*static_cast<int>(std::pow(4, num_orbital));
401 CRS matrix(dim_onsite, dim_onsite);
402 for (int row = 0; row < dim_onsite; ++row) {
403 for (int col = 0; col < dim_onsite; ++col) {
404 const int a = row%dim_onsite_lspin + 1;
405 const int b = col%dim_onsite_lspin + 1;
406 const int basis_row_total_electron = row/dim_onsite_lspin;
407 const int basis_col_total_electron = col/dim_onsite_lspin;
408 if (basis_row_total_electron == basis_col_total_electron && a + 1 == b) {
409 matrix.val.push_back(std::sqrt( (magnitude_lspin + 1.0)*(a + b - 1) - a*b ) );
410 matrix.col.push_back(col);
411 }
412 }
413 matrix.row.push_back(matrix.col.size());
414 }
415 return matrix;
416 }

569 static CRS CreateOnsiteOperatorSCSL(const double magnitude_lspin, const int orbital, const int num_orbital) {
570 const CRS spc = CreateOnsiteOperatorSpC(magnitude_lspin, orbital, num_orbital);
571 const CRS smc = CreateOnsiteOperatorSmC(magnitude_lspin, orbital, num_orbital);
572 const CRS szc = CreateOnsiteOperatorSzC(magnitude_lspin, orbital, num_orbital);

574 const CRS sml = CreateOnsiteOperatorSmL(magnitude_lspin, num_orbital);

824 const int max_n_up_down = static_cast<int>(total_electron/2);
825 std::vector<std::vector<int>> ele_configurations(4);
826 for (int n_up_down = 0; n_up_down <= max_n_up_down; ++n_up_down) {
827 for (int n_up = 0; n_up <= total_electron - 2*n_up_down; ++n_up) {
828 const int n_down = total_electron - 2*n_up_down - n_up;
829 const int n_vac = system_size - n_up - n_down - n_up_down;
830 if (0 <= n_up && 0 <= n_down && 0 <= n_vac) {
831 ele_configurations[0].push_back(n_up_down);
832 ele_configurations[1].push_back(n_up );
833 ele_configurations[2].push_back(n_down );
834 ele_configurations[3].push_back(n_vac );
835 }
836 }
837 }
838 return ele_configurations;
