478 void GenerateBasis(const std::vector<int> &total_electron, const double total_sz) {
479 const auto start = std::chrono::system_clock::now();
480 const int total_2sz = utility::DoubleHalfInteger(total_sz);
481 if (bases_.count({total_electron, total_2sz}) != 0) {
482 return;
483 }
484 std::cout << "Generating Basis..." << std::flush;
485
486 std::vector<std::vector<std::vector<int>>> electron_configuration_list;
487 std::vector<int> length_list;
488 std::int64_t length = 1;
489 for (const auto num_electron: total_electron) {
490 const auto electron_configuration = GenerateElectronConfigurations(system_size_, num_electron);
491 electron_configuration_list.push_back(electron_configuration);
492 length_list.push_back(static_cast<int>(electron_configuration[0].size()));
493 length *= electron_configuration[0].size();
494 }
495 const std::vector<std::vector<std::int64_t>> binom = utility::CalculateBinomialTable(system_size_);
496 const std::int64_t dim_target_global = CalculateTargetDim(total_electron, total_sz);
497
498 std::vector<std::vector<int>> q_number_n_up_down;
499 std::vector<std::vector<int>> q_number_n_up;
500 std::vector<std::vector<int>> q_number_n_down;
501 std::vector<std::vector<int>> q_number_n_vac;
502 std::vector<std::vector<int>> q_number_ele_list;
503 std::vector<int> q_number_spin_vec;
504 std::vector<std::int64_t> total_electron_dim_list;
505 std::vector<std::vector<std::int64_t>> each_electron_basis_dim_prod;
506 std::vector<std::vector<std::int64_t>> each_electron_basis_dim;
507 std::vector<std::int64_t> bias_basis;
508 bias_basis.push_back(0);
509
510 for (std::int64_t i = 0; i < length; ++i) {
511 int electron_2sz = 0;
512 std::vector<int> temp_n_up_down;
513 std::vector<int> temp_n_up;
514 std::vector<int> temp_n_down;
515 std::vector<int> temp_n_vac;
516 std::int64_t electron_dim = 1;
517 for (std::size_t j = 0; j < length_list.size(); ++j) {
518 std::int64_t prod = 1;
519 for (std::size_t k = j + 1; k < length_list.size(); ++k) {
520 prod *= length_list[k];
521 }
522 const std::size_t index = (i/prod)%length_list[j];
523 const int n_up_down = electron_configuration_list[j][0][index];
524 const int n_up = electron_configuration_list[j][1][index];
525 const int n_down = electron_configuration_list[j][2][index];
526 const int n_vac = electron_configuration_list[j][3][index];
527 electron_2sz += n_up - n_down;
528 temp_n_up_down.push_back(n_up_down);
529 temp_n_up .push_back(n_up );
530 temp_n_down .push_back(n_down );
531 temp_n_vac .push_back(n_vac );
532 electron_dim *= binom[system_size_][n_up]*binom[system_size_ - n_up][n_down]*binom[system_size_ - n_up - n_down][n_up_down];
533 }
534 const int spin_2sz = total_2sz - electron_2sz;
535 if (BaseU1Spin_1D<RealType>::isValidQNumber(system_size_, 0.5*magnitude_2lspin_, 0.5*spin_2sz)) {
536 q_number_n_up_down.push_back(temp_n_up_down);
537 q_number_n_up .push_back(temp_n_up );
538 q_number_n_down .push_back(temp_n_down );
539 q_number_n_vac .push_back(temp_n_vac );
540 q_number_spin_vec .push_back(spin_2sz);
541 total_electron_dim_list.push_back(electron_dim);
542 each_electron_basis_dim.emplace_back();
543 bias_basis.push_back(electron_dim*BaseU1Spin_1D<RealType>::CalculateTargetDim(system_size_, 0.5*magnitude_2lspin_, 0.5*spin_2sz));
544 for (std::size_t j = 0; j < length_list.size(); ++j) {
545 q_number_ele_list.push_back({temp_n_up_down[j], temp_n_up[j], temp_n_down[j], temp_n_vac[j]});
546 each_electron_basis_dim.back().push_back(binom[system_size_][temp_n_up[j]]*
547 binom[system_size_ - temp_n_up[j]][temp_n_down[j]]*
548 binom[system_size_ - temp_n_up[j] - temp_n_down[j]][temp_n_up_down[j]]);
549
550 }
551 }
552 }
553
554 each_electron_basis_dim_prod.resize(each_electron_basis_dim.size());
555 for (std::size_t i = 0; i < each_electron_basis_dim_prod.size(); ++i) {
556 each_electron_basis_dim_prod[i].resize(total_electron.size());
557 for (std::size_t j = 0; j < total_electron.size(); ++j) {
558 std::int64_t prod = 1;
559 for (std::size_t k = j + 1; k < total_electron.size(); ++k) {
560 prod *= each_electron_basis_dim[i][k];
561 }
562 each_electron_basis_dim_prod[i][j] = prod;
563 }
564 }
565
566 for (std::size_t i = 1; i < bias_basis.size(); ++i) {
567 bias_basis[i] += bias_basis[i - 1];
568 }
569
570 if (bias_basis.back() != dim_target_global) {
571 std::stringstream ss;
572 ss << "Unknown error in " << __FUNCTION__ << " at " << __LINE__ << std::endl;
573 throw std::runtime_error(ss.str());
574 }
575
576 std::vector<std::int64_t> site_constant_global(system_size_);
577 std::vector<std::int64_t> ele_constant_global(system_size_);
578
579 for (int site = 0; site < system_size_; ++site) {
580 site_constant_global[site] = static_cast<std::int64_t>(std::pow(dim_onsite_, site));
581 }
582
583 for (int o = 0; o < static_cast<int>(total_electron.size()); ++o) {
584 ele_constant_global[o] = static_cast<std::int64_t>(std::pow(dim_onsite_electron_, o));
585 }
586
587 //Generate spin bases
588 std::vector<int> temp_q_number_spin_vec = q_number_spin_vec;
589 std::sort(temp_q_number_spin_vec.begin(), temp_q_number_spin_vec.end());
590 temp_q_number_spin_vec.erase(std::unique(temp_q_number_spin_vec.begin(), temp_q_number_spin_vec.end()), temp_q_number_spin_vec.end());
591 std::unordered_map<int, std::vector<std::int64_t>> spin_bases;
592#pragma omp parallel for

763 static std::int64_t CalculateTargetDim(const int system_size,
764 const double magnitude_lspin,
765 const std::vector<int> &total_electron,
766 const double total_sz) {
767 std::vector<std::vector<std::vector<int>>> electron_configuration_list;
768 std::vector<int> length_list;
769 std::int64_t length = 1;
770 for (const auto num_electron: total_electron) {
771 const auto electron_configuration = GenerateElectronConfigurations(system_size, num_electron);
772 electron_configuration_list.push_back(electron_configuration);
773 length_list.push_back(static_cast<int>(electron_configuration[0].size()));
774 length *= electron_configuration[0].size();
775 }
776 const int total_2sz = utility::DoubleHalfInteger(total_sz);
777 const std::vector<std::vector<std::int64_t>> binom = utility::CalculateBinomialTable(system_size);
778 std::int64_t dim = 0;
779 for (std::int64_t i = 0; i < length; ++i) {
780 int electron_2sz = 0;
781 std::int64_t electron_dim = 1;
782 for (std::size_t j = 0; j < length_list.size(); ++j) {
783 std::int64_t prod = 1;
784 for (std::size_t k = j + 1; k < length_list.size(); ++k) {
785 prod *= length_list[k];
786 }
787 const std::size_t index = (i/prod)%length_list[j];
788 const int n_up_down = electron_configuration_list[j][0][index];
789 const int n_up = electron_configuration_list[j][1][index];
790 const int n_down = electron_configuration_list[j][2][index];
791 electron_2sz += n_up - n_down;
792 electron_dim *= binom[system_size][n_up]*binom[system_size - n_up][n_down]*binom[system_size - n_up - n_down][n_up_down];
793 }
794 const int spin_2sz = total_2sz - electron_2sz;
795 if (BaseU1Spin_1D<RealType>::isValidQNumber(system_size, magnitude_lspin, 0.5*spin_2sz)) {
796 dim += electron_dim*BaseU1Spin_1D<RealType>::CalculateTargetDim(system_size, magnitude_lspin, 0.5*spin_2sz);
797 }
798 }
799 return dim;
800 }

837 // 0 <-> [ ] -- (0, 0 )
