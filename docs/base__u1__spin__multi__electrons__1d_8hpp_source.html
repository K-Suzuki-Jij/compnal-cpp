191 static std::int64_t CalculateTargetDim(const int system_size,
192 const double magnitude_lspin,
193 const std::vector<int> &total_electron,
194 const double total_sz) {
195 std::vector<std::vector<std::vector<int>>> electron_configuration_list;
196 std::vector<int> length_list;
197 std::int64_t length = 1;
198 for (const auto num_electron: total_electron) {
199 const auto electron_configuration = GenerateElectronConfigurations(system_size, num_electron);
200 electron_configuration_list.push_back(electron_configuration);
201 length_list.push_back(electron_configuration[0].size());
202 length *= electron_configuration[0].size();
203 }
204 const int total_2sz = utility::DoubleTheNumber(total_sz);
205 const std::vector<std::vector<std::int64_t>> binom = utility::CalculateBinomialTable(system_size);
206 std::int64_t dim = 0;
207 for (std::int64_t i = 0; i < length; ++i) {
208 int electron_2sz = 0;
209 std::int64_t electron_dim = 1;
210 for (std::size_t j = 0; j < length_list.size(); ++j) {
211 std::int64_t prod = 1;
212 for (std::size_t k = j + 1; k < length_list.size(); ++k) {
213 prod *= length_list[k];
214 }
215 const std::size_t index = (i/prod)%length_list[j];
216 const int n_up_down = electron_configuration_list[j][0][index];
217 const int n_up = electron_configuration_list[j][1][index];
218 const int n_down = electron_configuration_list[j][2][index];
219 electron_2sz += n_up - n_down;
220 electron_dim *= binom[system_size][n_up]*binom[system_size - n_up][n_down]*binom[system_size - n_up - n_down][n_up_down];
221 }
222 const int spin_2sz = total_2sz - electron_2sz;
223 if (BaseU1Spin_1D<RealType>::isValidQNumber(system_size, magnitude_lspin, 0.5*spin_2sz)) {
224 dim += electron_dim*BaseU1Spin_1D<RealType>::CalculateTargetDim(system_size, magnitude_lspin, 0.5*spin_2sz);
225 }
226 }
227 return dim;
228 }

236 static CRS CreateOnsiteOperatorCUp(const double magnitude_lspin, const int orbital, const int num_orbital) {
237 const int magnitude_2lspin = utility::DoubleTheNumber(magnitude_lspin);
238 const int dim_onsite_lspin = magnitude_2lspin + 1;
239 const int dim_onsite = dim_onsite_lspin*static_cast<int>(std::pow(4, num_orbital));
240
241 //--------------------------------
242 // # <-> [Cherge ] -- (N, 2*sz)
243 // 0 <-> [ ] -- (0, 0 )
244 // 1 <-> [up ] -- (1, 1 )
245 // 2 <-> [down ] -- (1, -1 )
246 // 3 <-> [up&down ] -- (2, 0 )
247 //--------------------------------
248
249 CRS matrix(dim_onsite, dim_onsite);
250
251 for (int row = 0; row < dim_onsite; ++row) {
252 int num_electron = 0;
253 for (int o = 0; o < orbital; ++o) {
254 const int basis_electron_onsite = CalculateBasisOnsiteElectron(row, magnitude_lspin, o, num_orbital);
255 num_electron += CalculateNumElectron(basis_electron_onsite);
256 }
257 int sign = 1;
258 if (num_electron%2 == 1) {
259 sign = -1;
260 }
261
262 for (int col = 0; col < dim_onsite; col++) {
263 bool c_1 = true;
264 for (int o = 0; o < num_orbital; ++o) {
265 const int basis_1 = CalculateBasisOnsiteElectron(row, magnitude_lspin, o, num_orbital);
266 const int basis_2 = CalculateBasisOnsiteElectron(col, magnitude_lspin, o, num_orbital);
267 if (o != orbital && basis_1 != basis_2) {
268 c_1 = false;
269 }
270 }
271
272 int basis_row_electron = CalculateBasisOnsiteElectron(row, magnitude_lspin, orbital, num_orbital);
273 int basis_col_electron = CalculateBasisOnsiteElectron(col, magnitude_lspin, orbital, num_orbital);
274
275 const bool c_2 = (basis_row_electron == 0 && basis_col_electron == 1);
276 const bool c_3 = (basis_row_electron == 2 && basis_col_electron == 3);
277
278 if (c_1 && row%dim_onsite_lspin == col%dim_onsite_lspin && (c_2 || c_3)) {
279 matrix.val.push_back(sign);
280 matrix.col.push_back(col);
281 }
282
283 }
284 matrix.row.push_back(matrix.col.size());
285 }
286 return matrix;
287 }

357 static CRS CreateOnsiteOperatorSzL(const double magnitude_lspin, const int num_orbital) {
358 const int magnitude_2lspin = utility::DoubleTheNumber(magnitude_lspin);
359 const int dim_onsite_lspin = magnitude_2lspin + 1;
360 const int dim_onsite = dim_onsite_lspin*static_cast<int>(std::pow(4, num_orbital));
361 CRS matrix(dim_onsite, dim_onsite);
362 for (int row = 0; row < dim_onsite; ++row) {
363 for (int col = 0; col < dim_onsite; ++col) {
364 const int basis_row_lspin = row%dim_onsite_lspin;
365 const int basis_col_lspin = col%dim_onsite_lspin;
366 const int basis_row_total_electron = row/dim_onsite_lspin;
367 const int basis_col_total_electron = col/dim_onsite_lspin;
368 if (basis_row_total_electron == basis_col_total_electron && basis_row_lspin == basis_col_lspin) {
369 matrix.val.push_back(magnitude_lspin - basis_row_lspin);
370 matrix.col.push_back(col);
371 }
372 }
373 matrix.row.push_back(matrix.col.size());
374 }
375 return matrix;
376 }

382 static CRS CreateOnsiteOperatorSpL(const double magnitude_lspin, const int num_orbital) {
383 const int magnitude_2lspin = utility::DoubleTheNumber(magnitude_lspin);
384 const int dim_onsite_lspin = magnitude_2lspin + 1;
385 const int dim_onsite = dim_onsite_lspin*static_cast<int>(std::pow(4, num_orbital));
386 CRS matrix(dim_onsite, dim_onsite);
387 for (int row = 0; row < dim_onsite; ++row) {
388 for (int col = 0; col < dim_onsite; ++col) {
389 const int a = row%dim_onsite_lspin + 1;
390 const int b = col%dim_onsite_lspin + 1;
391 const int basis_row_total_electron = row/dim_onsite_lspin;
392 const int basis_col_total_electron = col/dim_onsite_lspin;
393 if (basis_row_total_electron == basis_col_total_electron && a + 1 == b) {
394 matrix.val.push_back(std::sqrt( (magnitude_lspin + 1.0)*(a + b - 1) - a*b ) );
395 matrix.col.push_back(col);
396 }
397 }
398 matrix.row.push_back(matrix.col.size());
399 }
400 return matrix;
401 }

810 std::vector<std::vector<int>> ele_configurations(4);
811 for (int n_up_down = 0; n_up_down <= max_n_up_down; ++n_up_down) {
812 for (int n_up = 0; n_up <= total_electron - 2*n_up_down; ++n_up) {

814 const int n_vac = system_size - n_up - n_down - n_up_down;
815 if (0 <= n_up && 0 <= n_down && 0 <= n_vac) {
816 ele_configurations[0].push_back(n_up_down);
817 ele_configurations[1].push_back(n_up );
818 ele_configurations[2].push_back(n_down );
819 ele_configurations[3].push_back(n_vac );
820 }
821 }
822 }
