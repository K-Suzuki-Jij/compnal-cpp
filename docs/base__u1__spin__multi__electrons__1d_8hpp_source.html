38 static CRS CreateOnsiteOperatorCUp(const double magnitude_lspin, const int orbital, const int num_orbital) {
39 const int magnitude_2lspin = utility::DoubleTheNumber(magnitude_lspin);
40 const int dim_onsite_lspin = magnitude_2lspin + 1;
41 const int dim_onsite = dim_onsite_lspin*static_cast<int>(std::pow(4, num_orbital));
42
43 //--------------------------------
44 // # <-> [Cherge ] -- (N, 2*sz)
45 // 0 <-> [ ] -- (0, 0 )
46 // 1 <-> [up ] -- (1, 1 )
47 // 2 <-> [down ] -- (1, -1 )
48 // 3 <-> [up&down ] -- (2, 0 )
49 //--------------------------------
50
51 CRS matrix(dim_onsite, dim_onsite);
52
53 for (int row = 0; row < dim_onsite; ++row) {
54 int num_electron = 0;
55 for (int o = 0; o < orbital; ++o) {
56 const int basis_electron_onsite = CalculateBasisOnsiteElectron(row, magnitude_lspin, o, num_orbital);
57 num_electron += CalculateNumElectron(basis_electron_onsite);
58 }
59 int sign = 1;
60 if (num_electron%2 == 1) {
61 sign = -1;
62 }
63
64 for (int col = 0; col < dim_onsite; col++) {
65 bool c_1 = true;
66 for (int o = 0; o < num_orbital; ++o) {
67 const int basis_1 = CalculateBasisOnsiteElectron(row, magnitude_lspin, o, num_orbital);
68 const int basis_2 = CalculateBasisOnsiteElectron(col, magnitude_lspin, o, num_orbital);
69 if (o != orbital && basis_1 != basis_2) {
70 c_1 = false;
71 }
72 }
73
74 int basis_row_electron = CalculateBasisOnsiteElectron(row, magnitude_lspin, orbital, num_orbital);
75 int basis_col_electron = CalculateBasisOnsiteElectron(col, magnitude_lspin, orbital, num_orbital);
76
77 const bool c_2 = (basis_row_electron == 0 && basis_col_electron == 1);
78 const bool c_3 = (basis_row_electron == 2 && basis_col_electron == 3);
79
80 if (c_1 && row%dim_onsite_lspin == col%dim_onsite_lspin && (c_2 || c_3)) {
81 matrix.val.push_back(sign);
82 matrix.col.push_back(col);
83 }
84
85 }
86 matrix.row.push_back(matrix.col.size());
87 }
88 return matrix;
89 }

159 static CRS CreateOnsiteOperatorSzL(const double magnitude_lspin, const int num_orbital) {
160 const int magnitude_2lspin = utility::DoubleTheNumber(magnitude_lspin);
161 const int dim_onsite_lspin = magnitude_2lspin + 1;
162 const int dim_onsite = dim_onsite_lspin*static_cast<int>(std::pow(4, num_orbital));
163 CRS matrix(dim_onsite, dim_onsite);
164 for (int row = 0; row < dim_onsite; ++row) {
165 for (int col = 0; col < dim_onsite; ++col) {
166 const int basis_row_lspin = row%dim_onsite_lspin;
167 const int basis_col_lspin = col%dim_onsite_lspin;
168 const int basis_row_total_electron = row/dim_onsite_lspin;
169 const int basis_col_total_electron = col/dim_onsite_lspin;
170 if (basis_row_total_electron == basis_col_total_electron && basis_row_lspin == basis_col_lspin) {
171 matrix.val.push_back(magnitude_lspin - basis_row_lspin);
172 matrix.col.push_back(col);
173 }
174 }
175 matrix.row.push_back(matrix.col.size());
176 }
177 return matrix;
178 }

184 static CRS CreateOnsiteOperatorSpL(const double magnitude_lspin, const int num_orbital) {
185 const int magnitude_2lspin = utility::DoubleTheNumber(magnitude_lspin);
186 const int dim_onsite_lspin = magnitude_2lspin + 1;
187 const int dim_onsite = dim_onsite_lspin*static_cast<int>(std::pow(4, num_orbital));
188 CRS matrix(dim_onsite, dim_onsite);
189 for (int row = 0; row < dim_onsite; ++row) {
190 for (int col = 0; col < dim_onsite; ++col) {
191 const int a = row%dim_onsite_lspin + 1;
192 const int b = col%dim_onsite_lspin + 1;
193 const int basis_row_total_electron = row/dim_onsite_lspin;
194 const int basis_col_total_electron = col/dim_onsite_lspin;
195 if (basis_row_total_electron == basis_col_total_electron && a + 1 == b) {
196 matrix.val.push_back(std::sqrt( (magnitude_lspin + 1.0)*(a + b - 1) - a*b ) );
197 matrix.col.push_back(col);
198 }
199 }
200 matrix.row.push_back(matrix.col.size());
201 }
202 return matrix;
203 }

356 static CRS CreateOnsiteOperatorSCSL(const double magnitude_lspin, const int orbital, const int num_orbital) {
357 const CRS spc = CreateOnsiteOperatorSpC(magnitude_lspin, orbital, num_orbital);
358 const CRS smc = CreateOnsiteOperatorSmC(magnitude_lspin, orbital, num_orbital);
359 const CRS szc = CreateOnsiteOperatorSzC(magnitude_lspin, orbital, num_orbital);
360 const CRS spl = CreateOnsiteOperatorSpL(magnitude_lspin, num_orbital);
361 const CRS sml = CreateOnsiteOperatorSmL(magnitude_lspin, num_orbital);
362 const CRS szl = CreateOnsiteOperatorSzL(magnitude_lspin, num_orbital);
363 return szc*szl + 0.5*(spc*sml + smc*spl);

372 static CRS CreateOnsiteOperatorSCSL(const double magnitude_lspin, const int num_orbital) {
373 const CRS spl = CreateOnsiteOperatorSpL(magnitude_lspin, num_orbital);
374 const CRS sml = CreateOnsiteOperatorSmL(magnitude_lspin, num_orbital);
375 const CRS szl = CreateOnsiteOperatorSzL(magnitude_lspin, num_orbital);
376 CRS out(szl.row_dim, szl.col_dim);
377 for (int o = 0; o < num_orbital; ++o) {
378 const CRS spc = CreateOnsiteOperatorSpC(magnitude_lspin, o, num_orbital);
379 const CRS smc = CreateOnsiteOperatorSmC(magnitude_lspin, o, num_orbital);
380 const CRS szc = CreateOnsiteOperatorSzC(magnitude_lspin, o, num_orbital);
381 out = out + szc*szl + 0.5*(spc*sml + smc*spl);
382 }
383 return out;

554 inline int CalculateBasisOnsiteElectron(const int basis_onsite, const int orbital) const {
555 const int num_inner_electron = num_electron_orbital_ - orbital - 1;
556 int basis_electron_onsite = basis_onsite/dim_onsite_lspin_;
557 for (int i = 0; i < num_inner_electron; ++i) {
558 basis_electron_onsite /= dim_onsite_electron_;
559 }
560 return basis_electron_onsite%dim_onsite_electron_;
561 }

569 inline static int CalculateBasisOnsiteElectron(const int basis_onsite, const double magnitude_lspin, const int orbital, const int num_orbital) {
570 const int dim_onsite_lspin = utility::DoubleTheNumber(magnitude_lspin) + 1;
571 const int num_inner_electron = num_orbital - orbital - 1;
572 int dim_onsite_electron = 1;
573 for (int i = 0; i < num_orbital; ++i) {
574 dim_onsite_electron *= 4;
575 }
576 int basis_electron_onsite = basis_onsite/dim_onsite_lspin;
577 for (int i = 0; i < num_inner_electron; ++i) {
578 basis_electron_onsite /= dim_onsite_electron;
579 }
580 return basis_electron_onsite%dim_onsite_electron;
581 }
