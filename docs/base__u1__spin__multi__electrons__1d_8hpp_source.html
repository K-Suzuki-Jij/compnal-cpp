552 basis_global += integer_list[j]*site_constant_global[j];
553 }
554 spin_basis.push_back(basis_global);
555 } while (std::next_permutation(integer_list.begin(), integer_list.end()));
556 }
557 }
558 if (static_cast<std::int64_t>(spin_basis.size()) != dim_target_lspin) {
559 std::stringstream ss;
560 ss << "Unknown error detected in " << __FUNCTION__ << std::endl;
561 throw std::runtime_error(ss.str());
562 }
563 std::sort(spin_basis.begin(), spin_basis.end());
564 }

566 //Generate electron bases
567 const std::int64_t loop_size = static_cast<std::int64_t>(q_number_spin_vec.size());
568 std::unordered_map<std::vector<int>, std::vector<std::int64_t>, utility::VectorHash> electron_bases;
569 std::vector<std::vector<int>> temp_q_number_ele_list = q_number_ele_list;
570 std::sort(temp_q_number_ele_list.begin(), temp_q_number_ele_list.end());
571 temp_q_number_ele_list.erase(std::unique(temp_q_number_ele_list.begin(), temp_q_number_ele_list.end()), temp_q_number_ele_list.end());
572
573#pragma omp parallel for
574 for (std::size_t i = 0; i < temp_q_number_ele_list.size(); ++i) {
575 const int n_up_down = temp_q_number_ele_list[i][0];
576 const int n_down = temp_q_number_ele_list[i][1];
577 const int n_up = temp_q_number_ele_list[i][2];
578 const int n_vac = temp_q_number_ele_list[i][3];
579 std::vector<int> basis_list_electron(system_size_);
580 for (int s = 0; s < n_vac; ++s) {
581 basis_list_electron[s] = 0;
582 }
583 for (int s = 0; s < n_up; ++s) {
584 basis_list_electron[s + n_vac] = 1;
585 }
586 for (int s = 0; s < n_down; ++s) {
587 basis_list_electron[s + n_vac + n_up] = 2;
588 }
589 for (int s = 0; s < n_up_down; ++s) {
590 basis_list_electron[s + n_vac + n_up + n_down] = 3;
591 }
592 do {
593 std::int64_t basis_global_electron = 0;
594 for (std::size_t j = 0; j < basis_list_electron.size(); ++j) {
595 basis_global_electron += basis_list_electron[j]*site_constant_global[j]*dim_onsite_lspin_;
596 }
597 } while (std::next_permutation(basis_list_electron.begin(), basis_list_electron.end()));
598 }
599
600 //Generate global bases
601 std::vector<std::int64_t>().swap(bases_[{total_electron, total_2sz}]);
602 auto &global_basis_ref = bases_.at({total_electron, total_2sz});
603 global_basis_ref.resize(dim_target_global);
604
605#pragma omp parallel for
606 for (std::int64_t i = 0; i < loop_size; ++i) {
607 std::int64_t count = bias_basis[i];
608 const int total_2sz_lspin = q_number_spin_vec[i];
609 const std::int64_t total_electron_dim = total_electron_dim_list[i];
610 for (const auto &spin_basis: spin_bases.at(total_2sz_lspin)) {
611
612 for (std::int64_t j = 0; j < total_electron_dim; ++j) {
613 std::int64_t global_basis = spin_basis;
614 for (std::size_t k = 0; k < total_electron.size(); ++k) {
615 const int n_up_down = q_number_n_up_down[i][k];
616 const int n_up = q_number_n_up [i][k];
617 const int n_down = q_number_n_down [i][k];
618 const int n_vac = q_number_n_vac [i][k];
619 std::size_t index = (j/each_electron_basis_dim_prod[i][k])%each_electron_basis_dim[i][k];
620 global_basis += electron_bases.at({n_up_down, n_up, n_down, n_vac})[index];
621 }
622 global_basis_ref[count++] = global_basis;
623 }
624 }
625 }
626
627 if (static_cast<std::int64_t>(bases_.at({total_electron, total_2sz}).size()) != dim_target_global) {
628 std::stringstream ss;
629 ss << "Unknown error detected in " << __FUNCTION__ << std::endl;
630 throw std::runtime_error(ss.str());
631 }
632
633 std::sort(global_basis_ref.begin(), global_basis_ref.end());
634
635 bases_inv_[{total_electron, total_2sz}].clear();
636
637 auto &basis_inv_ref = bases_inv_.at({total_electron, total_2sz});
638 for (std::int64_t i = 0; i < dim_target_global; ++i) {
639 basis_inv_ref[global_basis_ref[i]] = i;
640 }
641
642 if (basis_inv_ref.size() != global_basis_ref.size()) {
643 std::stringstream ss;
644 ss << "Unknown error detected in " << __FUNCTION__ << std::endl;
645 ss << "The same basis has been detected" << std::endl;
646 throw std::runtime_error(ss.str());
647 }

649 const double time_sec = static_cast<double>(time_count)/sparse_matrix::TIME_UNIT_CONSTANT;
650 std::cout << "\rElapsed time of generating basis:" << time_sec << "[sec]" << std::endl;

663 static std::int64_t CalculateTargetDim(const int system_size,
664 const double magnitude_lspin,
665 const std::vector<int> &total_electron,
666 const double total_sz) {
667 std::vector<std::vector<std::vector<int>>> electron_configuration_list;
668 std::vector<int> length_list;
669 std::int64_t length = 1;
670 for (const auto num_electron: total_electron) {
671 const auto electron_configuration = GenerateElectronConfigurations(system_size, num_electron);
672 electron_configuration_list.push_back(electron_configuration);
673 length_list.push_back(electron_configuration[0].size());
674 length *= electron_configuration[0].size();
675 }
676 const int total_2sz = utility::DoubleTheNumber(total_sz);
677 const std::vector<std::vector<std::int64_t>> binom = utility::CalculateBinomialTable(system_size);
678 std::int64_t dim = 0;
679 for (std::int64_t i = 0; i < length; ++i) {
680 int electron_2sz = 0;
681 std::int64_t electron_dim = 1;
682 for (std::size_t j = 0; j < length_list.size(); ++j) {
683 std::int64_t prod = 1;
684 for (std::size_t k = j + 1; k < length_list.size(); ++k) {
685 prod *= length_list[k];
686 }
687 const std::size_t index = (i/prod)%length_list[j];
688 const int n_up_down = electron_configuration_list[j][0][index];
689 const int n_up = electron_configuration_list[j][1][index];
690 const int n_down = electron_configuration_list[j][2][index];
691 electron_2sz += n_up - n_down;
692 electron_dim *= binom[system_size][n_up]*binom[system_size - n_up][n_down]*binom[system_size - n_up - n_down][n_up_down];
693 }
694 const int spin_2sz = total_2sz - electron_2sz;
695 if (BaseU1Spin_1D<RealType>::isValidQNumber(system_size, magnitude_lspin, 0.5*spin_2sz)) {
696 dim += electron_dim*BaseU1Spin_1D<RealType>::CalculateTargetDim(system_size, magnitude_lspin, 0.5*spin_2sz);
697 }
698 }
699 return dim;
700 }

730 static CRS CreateOnsiteOperatorCUp(const double magnitude_lspin, const int orbital, const int num_orbital) {
731 const int magnitude_2lspin = utility::DoubleTheNumber(magnitude_lspin);
732 const int dim_onsite_lspin = magnitude_2lspin + 1;
733 const int dim_onsite = dim_onsite_lspin*static_cast<int>(std::pow(4, num_orbital));
734
735 //--------------------------------
736 // # <-> [Cherge ] -- (N, 2*sz)
737 // 0 <-> [ ] -- (0, 0 )
738 // 1 <-> [up ] -- (1, 1 )
739 // 2 <-> [down ] -- (1, -1 )
740 // 3 <-> [up&down ] -- (2, 0 )
741 //--------------------------------
742
743 CRS matrix(dim_onsite, dim_onsite);
744
745 for (int row = 0; row < dim_onsite; ++row) {
746 int num_electron = 0;
747 for (int o = 0; o < orbital; ++o) {
748 const int basis_electron_onsite = CalculateBasisOnsiteElectron(row, magnitude_lspin, o, num_orbital);
749 num_electron += CalculateNumElectron(basis_electron_onsite);
750 }
751 int sign = 1;
752 if (num_electron%2 == 1) {
753 sign = -1;
754 }
755
756 for (int col = 0; col < dim_onsite; col++) {
757 bool c_1 = true;
758 for (int o = 0; o < num_orbital; ++o) {
759 const int basis_1 = CalculateBasisOnsiteElectron(row, magnitude_lspin, o, num_orbital);
760 const int basis_2 = CalculateBasisOnsiteElectron(col, magnitude_lspin, o, num_orbital);
761 if (o != orbital && basis_1 != basis_2) {
762 c_1 = false;
763 break;
764 }
765 }
766
767 int basis_row_electron = CalculateBasisOnsiteElectron(row, magnitude_lspin, orbital, num_orbital);
768 int basis_col_electron = CalculateBasisOnsiteElectron(col, magnitude_lspin, orbital, num_orbital);
769
770 const bool c_2 = (basis_row_electron == 0 && basis_col_electron == 1);
771 const bool c_3 = (basis_row_electron == 2 && basis_col_electron == 3);
772
773 if (c_1 && row%dim_onsite_lspin == col%dim_onsite_lspin && (c_2 || c_3)) {
774 matrix.val.push_back(sign);
775 matrix.col.push_back(col);
776 }
777
778 }
779 matrix.row[row + 1] = matrix.col.size();
780 }
781 matrix.tag = sparse_matrix::CRSTag::FERMION;
782 return matrix;
