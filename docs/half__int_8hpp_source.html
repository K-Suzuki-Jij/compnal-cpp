40 template<typename ValueType>
41 HalfInt(ValueType value) {
42 value = 2*value;
43 if (std::floor(value) != value) {
44 throw std::runtime_error("The input number is not half-integer");
45 }
46 integer_ = static_cast<int>(value);
47 };
48
49 //------------------------------------------------------------------
50 //----------------------Public Member Functions---------------------
51 //------------------------------------------------------------------
53 int GetInteger() const {
54 return integer_;
55 }
56
57 //------------------------------------------------------------------
58 //-----------------------Operator Overloading-----------------------
59 //------------------------------------------------------------------
61 HalfInt operator+() const {
62 return *this;
63 }
64
66 HalfInt operator-() const {
67 return HalfInt(-0.5*this->integer_);
68 }
69
72 template<typename ValueType>
73 operator ValueType() const noexcept {
74 return static_cast<ValueType>(0.5*integer_);
75 }

77private:
