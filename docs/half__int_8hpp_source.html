39 template<typename ValueType>
40 HalfInt(ValueType value) {
41 value = 2*value;
42 if (std::floor(value) != value) {
43 throw std::runtime_error("The input number is not half-integer");
44 }
45 integer_ = static_cast<int>(value);
46 };
