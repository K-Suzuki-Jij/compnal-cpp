40 template<typename ValueType>
41 HalfInt(ValueType value) {
42 value = 2*value;
43 if (std::floor(value) != value) {
44 throw std::runtime_error("The input number is not half-integer");
45 }
46 integer_ = static_cast<int>(value);
47 };
