244 template<typename ReturnType>
245 ReturnType CalculateL1Norm() const {
246 ReturnType norm = 0.0;
247#pragma omp parallel for reduction (+:norm)
248 for (std::size_t i = 0; i < this->value_list.size(); ++i) {
249 norm += std::abs(this->value_list[i]);
250 }
251 return norm;
252 }

257 template<typename ReturnType>
258 ReturnType CalculateL2Norm() const {
259 ReturnType inner_product = 0.0;
260#pragma omp parallel for reduction (+:inner_product)
261 for (std::size_t i = 0; i < this->value_list.size(); ++i) {
262 inner_product += this->value_list[i]*this->value_list[i];
263 }
264 return std::sqrt(inner_product);
265 }

368template<typename T1, typename T2, typename T3, typename T4>
369auto CalculateVectorVectorSum(const T1 coeff_1,
370 const BraketVector<T2> &braket_vector_1,
371 const T3 coeff_2,
372 const BraketVector<T4> &braket_vector_2) {
373 if (braket_vector_1.value_list.size() != braket_vector_2.value_list.size()) {
374 std::stringstream ss;
375 ss << "Error at " << __LINE__ << " in " << __func__ << " in "<< __FILE__ << std::endl;
376 ss << "BraketVector types do not match each other" << std::endl;
377 ss << "dim_1 = " << braket_vector_1.value_list.size() << ", dim_2 = " << braket_vector_2.value_list.size() << std::endl;
378 throw std::runtime_error(ss.str());
379 }
380 BraketVector<decltype(T1{0}*T2{0}+T3{0}*T4{0})> vector_out;
381 vector_out.Resize(braket_vector_1.value_list.size());
382 const std::int64_t size = braket_vector_1.value_list.size();
383#pragma omp parallel for
384 for (std::int64_t i = 0; i < size; ++i) {
385 vector_out.value_list[i] = coeff_1*braket_vector_1.value_list[i] + coeff_2*braket_vector_2.value_list[i];
386 }
387 return vector_out;
388}

396template<typename T1, typename T2>
397auto CalculateVectorVectorProduct(const BraketVector<T1> &braket_vector_1,
398 const BraketVector<T2> &braket_vector_2) {
399 if (braket_vector_1.value_list.size() != braket_vector_2.value_list.size()) {
400 std::stringstream ss;
401 ss << "Error at " << __LINE__ << " in " << __func__ << " in "<< __FILE__ << std::endl;
402 ss << "BraketVector types do not match each other" << std::endl;
403 ss << "dim_1 = " << braket_vector_1.value_list.size() << ", dim_2 = " << braket_vector_2.value_list.size() << std::endl;
404 throw std::runtime_error(ss.str());
405 }
406 decltype(T1{0}*T2{0}) val_out = 0.0;
407 const std::int64_t size = braket_vector_1.value_list.size();
408#pragma omp parallel for reduction (+: val_out)
409 for (std::int64_t i = 0; i < size; ++i) {
410 val_out += braket_vector_1.value_list[i]*braket_vector_2.value_list[i];
411 }
412 return val_out;
413}
