133 static CRS CreateOnsiteOperatorHam(const RealType h_z = 0.0, const RealType U = 0.0) {
134 const CRS m_s_z = BaseU1Electron_1D<RealType>::CreateOnsiteOperatorSz();
135 const CRS m_n_up = BaseU1Electron_1D<RealType>::CreateOnsiteOperatorNCUp();
136 const CRS m_n_down = BaseU1Electron_1D<RealType>::CreateOnsiteOperatorNCDown();
137 const CRS m_n_up_down = sparse_matrix::CalculateMatrixMatrixProduct(1.0, m_n_up, 1.0, m_n_down);
138 return sparse_matrix::CalculateMatrixMatrixSum(U, m_n_up_down, h_z, m_s_z);
139 }
