116 void AddInteraction(const IndexType &site_1, const CRS &m_1, const IndexType &site_2, const CRS &m_2, const RealType val = 1.0) {
117 if (val == 0.0) {
118 return;
119 }
120 if (m_1.row_dim != m_1.col_dim || m_2.row_dim != m_2.col_dim) {
121 std::stringstream ss;
122 ss << "Error at " << __LINE__ << " in " << __FUNCTION__ << " in "<< __FILE__ << std::endl;
123 ss << "The input matrix is not a square matrix." << std::endl;
124 throw std::runtime_error(ss.str());
125 }
126 if (m_1.row_dim != this->GetDimOnsite()) {
127 std::stringstream ss;
128 ss << "Error at " << __LINE__ << " in " << __FUNCTION__ << " in "<< __FILE__ << std::endl;
129 ss << "The input matrix is invalid" << std::endl;
130 throw std::runtime_error(ss.str());
131 }
132
133 if (site_1 == site_2) {
134 AddPotential(site_1, m_1*m_2, val);
135 return;
136 }
137 interaction_list_[site_1][site_2].push_back({val*m_1, m_2});
138 index_list_.emplace(site_1);
139 index_list_.emplace(site_2);
140 }
