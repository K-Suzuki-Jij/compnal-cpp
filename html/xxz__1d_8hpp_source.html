134 static CRS CreateOnsiteOperatorHam(const double magnitude_spin, const RealType h_z = 0.0, const RealType D_z = 0.0) {
135 const int magnitude_2spin = utility::DoubleTheNumber(magnitude_spin);
136 const int dim_onsite = magnitude_2spin + 1;
137 CRS matrix(dim_onsite, dim_onsite);
138
139 for (int row = 0; row < dim_onsite; ++row) {
140 const RealType val = h_z*(magnitude_spin - row) + D_z*D_z*(magnitude_spin - row);
141 if (val != 0.0) {
142 matrix.val.push_back(val);
143 matrix.col.push_back(row);
144 }
145 matrix.row[row + 1] = matrix.col.size();
146 }
147 return matrix;
148 }
