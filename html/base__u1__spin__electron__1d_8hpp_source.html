175 std::vector<int> q_number_n_up_vec;

177 std::vector<int> q_number_n_up_down_vec;
178 std::vector<std::int64_t> bias_basis;
179 bias_basis.push_back(0);
180 for (int n_up_down = 0; n_up_down <= max_n_up_down; ++n_up_down) {
181 for (int n_up = 0; n_up <= total_electron - 2*max_n_up_down; ++n_up) {
182 const int n_down = total_electron - 2*max_n_up_down - n_up;
183 const int n_vac = system_size_ - n_up - n_down - n_up_down;

185 const double total_sz_electron = 0.5*(n_up - n_down);
186 if (BaseU1Spin_1D<RealType>::isValidQNumber(system_size_, 0.5*magnitude_2lspin_, total_sz - total_sz_electron)) {
187 q_number_spin_vec.push_back(total_2sz - (n_up - n_down));
188 q_number_n_vac_vec.push_back(n_vac);
189 q_number_n_up_vec.push_back(n_up );
190 q_number_n_down_vec.push_back(n_down);
191 q_number_n_up_down_vec.push_back(n_up_down);
192 const std::int64_t dim_electron = binom[system_size_][n_up]*binom[system_size_ - n_up][n_down]*binom[system_size_ - n_up - n_down][n_up_down];
193 const std::int64_t dim_lspin = BaseU1Spin_1D<RealType>::CalculateTargetDim(system_size_, 0.5*magnitude_2lspin_, total_sz - total_sz_electron);

223 spin_basis.reserve(dim_target_lspin);
224 for (auto &&integer_list: partition_integers) {
225 const bool condition1 = (0 < integer_list.size()) && (static_cast<int>(integer_list.size()) <= system_size_);
226 const bool condition2 = (integer_list.size() == 0) && (shifted_2sz == 0);
227 if (condition1 || condition2) {
228 for (int j = static_cast<int>(integer_list.size()); j < system_size_; ++j) {

231 std::sort(integer_list.begin(), integer_list.end());
232 do {
233 std::int64_t basis_global = 0;
234 for (std::size_t j = 0; j < integer_list.size(); ++j) {
235 basis_global += integer_list[j]*site_constant_global[j];
236 }
237 spin_basis.push_back(basis_global);
238 } while (std::next_permutation(integer_list.begin(), integer_list.end()));
239 }
240 }
241 if (static_cast<std::int64_t>(spin_basis.size()) != dim_target_lspin) {
242 std::stringstream ss;
243 ss << "Unknown error detected in " << __FUNCTION__ << std::endl;
244 throw std::runtime_error(ss.str());
245 }
246 std::sort(spin_basis.begin(), spin_basis.end());

249 //Generate global bases
250 const std::int64_t loop_size = static_cast<std::int64_t>(q_number_spin_vec.size());
251 global_basis_ref.resize(dim_target_global);
252#pragma omp parallel for
253 for (std::int64_t i = 0; i < loop_size; ++i) {
254 const int n_vac = q_number_n_vac_vec[i];
255 const int n_up = q_number_n_up_vec[i];
256 const int n_down = q_number_n_down_vec[i];
257 const int n_up_down = q_number_n_up_down_vec[i];
258 const int total_2sz_lspin = q_number_spin_vec[i];

261 for (int s = 0; s < n_vac; ++s) {
262 basis_list_electron[s] = 0;
263 }
264 for (int s = 0; s < n_up; ++s) {
265 basis_list_electron[s + n_vac] = 1;
266 }
267 for (int s = 0; s < n_down; ++s) {
268 basis_list_electron[s + n_vac + n_up] = 2;
269 }
270 for (int s = 0; s < n_up_down; ++s) {
271 basis_list_electron[s + n_vac + n_up + n_down] = 3;
272 }

275 std::int64_t basis_global_electron = 0;
276 for (std::size_t j = 0; j < basis_list_electron.size(); ++j) {
277 basis_global_electron += basis_list_electron[j]*site_constant_global[j]*dim_onsite_lspin_;
278 }
279 for (const auto &basis_global_lspin: spin_bases.at(total_2sz_lspin)) {
280 global_basis_ref[count++] = basis_global_electron + basis_global_lspin;
281
282 }
283 } while (std::next_permutation(basis_list_electron.begin(), basis_list_electron.end()));
284 }
285
286 if (static_cast<std::int64_t>(bases_.at({total_electron, total_2sz}).size()) != dim_target_global) {
287 std::stringstream ss;
288 ss << "Unknown error detected in " << __FUNCTION__ << std::endl;
289 throw std::runtime_error(ss.str());
290 }

292 std::sort(global_basis_ref.begin(), global_basis_ref.end());
293
294 bases_inv_[{total_electron, total_2sz}].clear();
295
296 auto &basis_inv_ref = bases_inv_.at({total_electron, total_2sz});
297 for (std::int64_t i = 0; i < dim_target_global; ++i) {
298 basis_inv_ref[global_basis_ref[i]] = i;

302 std::stringstream ss;
303 ss << "Unknown error detected in " << __FUNCTION__ << std::endl;

313 static bool isValidQNumber(const int system_size, const double magnitude_lspin, const int total_electron, const double total_sz) {
314 const int total_2sz = utility::DoubleTheNumber(total_sz);
315 const int magnitude_2lspin = utility::DoubleTheNumber(magnitude_lspin);
316 const bool c1 = (0 <= total_electron && total_electron <= 2*system_size);
317 const bool c2 = ((total_electron + system_size*magnitude_2lspin - total_2sz)%2 == 0);
318 const bool c3 = (-(total_electron + system_size*magnitude_2lspin) <= total_2sz);
319 const bool c4 = (total_2sz <= total_electron + system_size*magnitude_2lspin);
320 if (c1 && c2 && c3 && c4) {
321 return true;
322 }
323 else {
324 return false;
325 }

328 static std::int64_t CalculateTargetDim(const int system_size, const double magnitude_lspin, const int total_electron, const double total_sz) {
329 if (!isValidQNumber(system_size, magnitude_lspin, total_electron, total_sz)) {
330 return 0;
331 }
332 const std::vector<std::vector<std::int64_t>> binom = utility::CalculateBinomialTable(system_size);
333 const int max_n_up_down = static_cast<int>(total_electron/2);
334 std::int64_t dim = 0;
335 for (int n_up_down = 0; n_up_down <= max_n_up_down; ++n_up_down) {
336 for (int n_up = 0; n_up <= total_electron - 2*max_n_up_down; ++n_up) {
337 const int n_down = total_electron - 2*max_n_up_down - n_up;
338 const int n_vac = system_size - n_up - n_down - n_up_down;
339 if (0 <= n_up && 0 <= n_down && 0 <= n_vac) {
340 const double total_sz_electron = 0.5*(n_up - n_down);
341 const std::int64_t dim_electron = binom[system_size][n_up]*binom[system_size - n_up][n_down]*binom[system_size - n_up - n_down][n_up_down];
342 const std::int64_t dim_lspin = BaseU1Spin_1D<RealType>::CalculateTargetDim(system_size, magnitude_lspin, total_sz - total_sz_electron);
343 dim += dim_electron*dim_lspin;
344 }
345 }
346 }
347 return dim;
