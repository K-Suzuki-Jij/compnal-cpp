206 void GenerateBasis(const std::pair<int, double> &quantum_number) {
207 if (!isValidQNumber(quantum_number)) {
208 std::stringstream ss;
209 ss << "Error in " << __FUNCTION__ << std::endl;
210 ss << "Invalid parameters (system_size or magnitude_spin or total_sz)" << std::endl;
211 throw std::runtime_error(ss.str());
212 }
213
214 const auto start = std::chrono::system_clock::now();
215 const int total_electron = quantum_number.first;
216 const double total_sz = quantum_number.second;
217 const int total_2sz = utility::DoubleTheNumber(total_sz);
218
219 if (bases_.count({total_electron, total_2sz}) != 0) {
220 return;
221 }
222
223 std::cout << "Generating Basis..." << std::flush;
224 const int max_n_up_down = static_cast<int>(total_electron/2);
225 const std::int64_t dim_target = CalculateTargetDim({total_electron, total_sz});
226
227 std::vector<std::int64_t>().swap(bases_[{total_electron, total_2sz}]);
228 auto &basis_ref = bases_.at({total_electron, total_2sz});
229
230 std::vector<std::int64_t> site_constant(system_size_);
231 for (int site = 0; site < system_size_; ++site) {
232 site_constant[site] = static_cast<std::int64_t>(std::pow(dim_onsite_, site));
233 }
234
235 std::vector<int> basis_list(system_size_);
236
237#ifdef _OPENMP
238 const int num_threads = omp_get_max_threads();
239 std::vector<std::vector<std::int64_t>> temp_basis(num_threads);
240 for (int n_up_down = 0; n_up_down <= max_n_up_down; ++n_up_down) {
241 const int n_up = static_cast<int>((total_electron - 2*n_up_down + total_2sz)/2);
242 const int n_down = static_cast<int>((total_electron - 2*n_up_down - total_2sz)/2);
243 const int n_vac = system_size_ - total_electron + n_up_down;
244 if (0 <= n_up && 0 <= n_down && 0 <= n_vac) {
245 for (int s = 0; s < n_vac; ++s) {
246 basis_list[s] = 0;
247 }
248 for (int s = 0; s < n_up; ++s) {
249 basis_list[s + n_vac] = 1;
250 }
251 for (int s = 0; s < n_down; ++s) {
252 basis_list[s + n_vac + n_up] = 2;
253 }
254 for (int s = 0; s < n_up_down; ++s) {
255 basis_list[s + n_vac + n_up + n_down] = 3;
256 }
257
258 const std::int64_t size = utility::CalculateNumCombination(basis_list);
259 std::vector<std::vector<int>> temp_basis_list(num_threads);
260
261#pragma omp parallel num_threads (num_threads)
262 {
263 const int thread_num = omp_get_thread_num();
264 const std::int64_t loop_begin = thread_num*size/num_threads;
265 const std::int64_t loop_end = (thread_num + 1)*size/num_threads;
266 temp_basis_list[thread_num] = basis_list;
267 utility::CalculateNthPermutation(&temp_basis_list[thread_num], loop_begin);
268
269 for (std::int64_t j = loop_begin; j < loop_end; ++j) {
270 std::int64_t basis_global = 0;
271 for (std::size_t k = 0; k < temp_basis_list[thread_num].size(); ++k) {
272 basis_global += temp_basis_list[thread_num][k]*site_constant[k];
273 }
274 temp_basis[thread_num].push_back(basis_global);
275 std::next_permutation(temp_basis_list[thread_num].begin(), temp_basis_list[thread_num].end());
276 }
277 }
278 }
279 }
280 for (auto &&basis: temp_basis) {
281 basis_ref.insert(basis_ref.end(), basis.begin(), basis.end());
282 std::vector<std::int64_t>().swap(basis);
283 }
284
285#else
286
287 basis_ref.reserve(dim_target);
288 for (int n_up_down = 0; n_up_down <= max_n_up_down; ++n_up_down) {
289 const int n_up = static_cast<int>((total_electron - 2*n_up_down + total_2sz)/2);
290 const int n_down = static_cast<int>((total_electron - 2*n_up_down - total_2sz)/2);
291 const int n_vac = system_size_ - total_electron + n_up_down;
292 if (0 <= n_up && 0 <= n_down && 0 <= n_vac) {
293 for (int s = 0; s < n_vac; ++s) {
294 basis_list[s] = 0;
295 }
296 for (int s = 0; s < n_up; ++s) {
297 basis_list[s + n_vac] = 1;
298 }
299 for (int s = 0; s < n_down; ++s) {
300 basis_list[s + n_vac + n_up] = 2;
301 }
302 for (int s = 0; s < n_up_down; ++s) {
303 basis_list[s + n_vac + n_up + n_down] = 3;
304 }
305
306 do {
307 std::int64_t basis_global = 0;
308 for (std::size_t j = 0; j < basis_list.size(); ++j) {
309 basis_global += basis_list[j]*site_constant[j];
310 }
311 basis_ref.push_back(basis_global);
312 } while (std::next_permutation(basis_list.begin(), basis_list.end()));
313 }
314 }
315#endif
316
317 if (static_cast<std::int64_t>(bases_.at({total_electron, total_2sz}).size()) != dim_target) {
318 std::stringstream ss;
319 ss << "Unknown error detected in " << __FUNCTION__ << std::endl;
320 throw std::runtime_error(ss.str());
321 }
322
323 std::sort(basis_ref.begin(), basis_ref.end());
324
325 bases_inv_[{total_electron, total_2sz}].clear();
326
327 auto &basis_inv_ref = bases_inv_.at({total_electron, total_2sz});
328 for (std::int64_t i = 0; i < dim_target; ++i) {
329 basis_inv_ref[basis_ref[i]] = i;
330 }
331
332 if (basis_inv_ref.size() != basis_ref.size()) {
333 std::stringstream ss;
334 ss << "Unknown error detected in " << __FUNCTION__ << std::endl;
335 ss << "The same basis has been detected" << std::endl;
336 throw std::runtime_error(ss.str());
337 }
338 const auto time_count = std::chrono::duration_cast<std::chrono::microseconds>(std::chrono::system_clock::now() - start).count();
339 const double time_sec = static_cast<double>(time_count)/sparse_matrix::TIME_UNIT_CONSTANT;
340 std::cout << "\rElapsed time of generating basis:" << time_sec << "[sec]" << std::endl;
341 }

505 static bool isValidQNumber(const int system_size, const int total_electron, const double total_sz) {
506 const int total_2sz = utility::DoubleTheNumber(total_sz);
507 const bool c1 = (0 <= total_electron && total_electron <= 2*system_size);
508 const bool c2 = ((total_electron - total_2sz)%2 == 0);
509 const bool c3 = (-total_electron <= total_2sz && total_2sz <= total_electron);
510 if (c1 && c2 && c3) {
511 return true;
512 }
513 else {
514 return false;
515 }

523 static std::int64_t CalculateTargetDim(const int system_size, const int total_electron, const double total_sz) {
524 if (!isValidQNumber(system_size, total_electron, total_sz)) {
525 return 0;
526 }
527 const int total_2sz = utility::DoubleTheNumber(total_sz);
528 const std::vector<std::vector<std::int64_t>> binom = utility::CalculateBinomialTable(system_size);
529 const int max_n_up_down = static_cast<int>(total_electron/2);
530 std::int64_t dim = 0;
531 for (int n_up_down = 0; n_up_down <= max_n_up_down; ++n_up_down) {
532 const int n_up = static_cast<int>((total_electron - 2*n_up_down + total_2sz)/2);
533 const int n_down = static_cast<int>((total_electron - 2*n_up_down - total_2sz)/2);
534 const int n_vac = system_size - total_electron + n_up_down;
535 if (0 <= n_up && 0 <= n_down && 0 <= n_vac) {
536 // TODO: Detect Overflow
537 dim += binom[system_size][n_up]*binom[system_size - n_up][n_down]*binom[system_size - n_up - n_down][n_up_down];
538 }
539 }
540 return dim;
541 }

545 static CRS CreateOnsiteOperatorCUp() {
546
547 //--------------------------------
548 // # <-> [Cherge ] -- (N, 2*sz)
549 // 0 <-> [ ] -- (0, 0 )
550 // 1 <-> [up ] -- (1, 1 )
551 // 2 <-> [down ] -- (1, -1 )
552 // 3 <-> [up&down ] -- (2, 0 )
553 //--------------------------------
554
555 const int dim_onsite = 4;
556 CRS matrix(dim_onsite, dim_onsite);
557 for (int row = 0; row < dim_onsite; row++) {
558 for (int col = 0; col < dim_onsite; col++) {
559 if ((col == 1 && row == 0) || (col == 3 && row == 2)) {
560 matrix.col.push_back(col);
561 matrix.val.push_back(1.0);
562 }
563 }
564 matrix.row[row + 1] = matrix.col.size();
565 }
566
567 matrix.tag = sparse_matrix::CRSTag::FERMION;
568
569 return matrix;
570 }
