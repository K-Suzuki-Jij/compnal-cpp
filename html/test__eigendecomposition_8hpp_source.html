14template<typename RealType>
15compnal::sparse_matrix::CRS<RealType> GenerateTightBindingHamiltonian(const int dim, const bool flag_symmetric = false) {
16 if (dim <= 2) {
17 throw std::runtime_error("Too small system size");
18 }
19 compnal::sparse_matrix::CRS<double> matrix;
20 if (flag_symmetric) {
21 matrix.val.push_back(0.0);
22 matrix.col.push_back(0);
23 matrix.val.push_back(-1.0);
24 matrix.col.push_back(1);
25 matrix.val.push_back(-1.0);
26 matrix.col.push_back(dim - 1);
27 matrix.UpdateRow();
28 for (int i = 1; i < dim; ++i) {
29 matrix.val.push_back(0.0);
30 matrix.col.push_back(i);
31 if (i + 1 < dim) {
32 matrix.val.push_back(-1.0);
33 matrix.col.push_back(i + 1);
34 }
35 matrix.UpdateRow();
36 }
37 }
38 else {
39 matrix.val.push_back(-1.0);
40 matrix.col.push_back(1);
41 matrix.val.push_back(-1.0);
42 matrix.col.push_back(dim - 1);
43 matrix.UpdateRow();
44 for (int i = 0; i < dim - 1; ++i) {
45 if (i == dim - 2) {
46 matrix.val.push_back(-1.0);
47 matrix.col.push_back(0);
48 }
49 matrix.val.push_back(-1.0);
50 matrix.col.push_back(i);
51 if (i + 2 < dim) {
52 matrix.val.push_back(-1.0);
53 matrix.col.push_back(i + 2);
54 }
55 matrix.UpdateRow();
56 }
57 }
58 matrix.row_dim = dim;
59 matrix.col_dim = dim;
60 return matrix;
61}
